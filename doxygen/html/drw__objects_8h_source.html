55class DRW_TableEntry {
56public:
57 //initializes default values
58 DRW_TableEntry()
59 : tType(DRW::UNKNOWNT),
60 handle(),
61 parentHandle(0),
62 name(),
63 flags(0),
64 extData(),
65 curr(nullptr)
66 {}
67
68 virtual~DRW_TableEntry() {
69 for (std::vector<DRW_Variant*>::iterator it=extData.begin(); it!=extData.end(); ++it)
70 delete *it;
71
72 extData.clear();
73 }
74
75 DRW_TableEntry(const DRW_TableEntry& e)
76 : tType(e.tType),
77 handle(e.handle),
78 parentHandle(e.parentHandle),
79 name(e.name),
80 flags(e.flags),
81 extData(),
82 curr(e.curr)
83 {
84 for (std::vector<DRW_Variant*>::const_iterator it=e.extData.begin(); it!=e.extData.end(); ++it){
85 extData.push_back(new DRW_Variant(*(*it)));
86 }
87 }
88
89protected:
90 void parseCode(int code, dxfReader *reader);
91 void reset(){
92 flags = 0;
93 for (std::vector<DRW_Variant*>::iterator it=extData.begin(); it!=extData.end(); ++it)
94 delete *it;
95 extData.clear();
96 }
97
98public:
99 enum DRW::TTYPE tType;
100 duint32 handle;
101 int parentHandle;
102 UTF8STRING name;
103 int flags;
104 std::vector<DRW_Variant*> extData;
106private:
107 DRW_TableEntry &operator=(const DRW_TableEntry &) Q_DECL_EQ_DELETE;
108 // cppcheck-suppress unsafeClassCanLeak
109 DRW_Variant* curr;
110};
