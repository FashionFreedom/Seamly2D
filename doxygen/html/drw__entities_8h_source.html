97class DRW_Entity {
98 SETENTFRIENDS
99public:
100 //initializes default values
101 DRW_Entity()
102 : eType(DRW::UNKNOWN),
103 handle(DRW::NoHandle),
104 appData(),
105 parentHandle(DRW::NoHandle),
106 space(DRW::ModelSpace),
107 layer("0"),
108 lineType("BYLAYER"),
109 material(DRW::MaterialByLayer),
110 color(DRW::ColorByLayer),
111 lWeight(DRW_LW_Conv::widthByLayer),
112 ltypeScale(1.0),
113 visible(true),
114 numProxyGraph(0),
115 proxyGraphics(),
116 color24(-1),
117 colorName(),
118 transparency(DRW::Opaque),
119 plotStyle(DRW::DefaultPlotStyle),
120 shadow(DRW::CastAndReceieveShadows),
121 haveExtrusion(false),
122 extData(),
123 extAxisX(),
124 extAxisY(),
125 curr(nullptr)
126 {}
127
128 DRW_Entity(const DRW_Entity& e)
129 : eType(e.eType),
130 handle(e.handle),
131 appData(),
132 parentHandle(e.parentHandle),
133 space(e.space),
134 layer(e.layer),
135 lineType(e.lineType),
136 material(e.material),
137 color(e.color),
138 lWeight(e.lWeight),
139 ltypeScale(e.ltypeScale),
140 visible(e.visible),
141 numProxyGraph(e.numProxyGraph),
142 proxyGraphics(),
143 color24(e.color24),
144 colorName(),
145 transparency(e.transparency),
146 plotStyle(e.plotStyle),
147 shadow(e.shadow),
148 haveExtrusion(e.haveExtrusion),
149 extData(),
150 extAxisX(),
151 extAxisY(),
152 curr(nullptr/*e.curr*/)
153 {
154 for (std::vector<DRW_Variant*>::const_iterator it=e.extData.begin(); it!=e.extData.end(); ++it){
155 extData.push_back(new DRW_Variant(*(*it)));
156 }
157 }
158
159 virtual ~DRW_Entity() {
160 for (std::vector<DRW_Variant*>::iterator it=extData.begin(); it!=extData.end(); ++it)
161 delete *it;
162
163 extData.clear();
164 }
165
166 void reset(){
167 for (std::vector<DRW_Variant*>::iterator it=extData.begin(); it!=extData.end(); ++it)
168 delete *it;
169 extData.clear();
170 }
171
172 virtual void applyExtrusion() = 0;
173
174 void setWidthMm(double millimeters) {
175 if(millimeters < 0.0) {
176 lWeight = DRW_LW_Conv::widthByLayer;
177 return;
178 }
179 if(millimeters > 2.11) millimeters = 2.11;
180 lWeight = DRW_LW_Conv::dxfInt2lineWidth(int(floor(millimeters * 100.0)));
181 }
182
183protected:
184 //parses dxf pair to read entity
185 bool parseCode(int code, dxfReader *reader);
186 //calculates extrusion axis (normal vector)
187 void calculateAxis(DRW_Coord extPoint);
188 //apply extrusion to @extPoint and return data in @point
189 void extrudePoint(DRW_Coord extPoint, DRW_Coord *point) const;
190
191 //parses dxf 102 groups to read entity
192 bool parseDxfGroups(int code, dxfReader *reader);
193
194public:
195 enum DRW::ETYPE eType;
196 duint32 handle;
197 std::list<std::list<DRW_Variant> > appData;
198 duint32 parentHandle;
199 DRW::Space space;
200 UTF8STRING layer;
201 UTF8STRING lineType;
202 duint32 material;
203 int color;
204 enum DRW_LW_Conv::lineWidth lWeight;
205 double ltypeScale;
206 bool visible;
207 int numProxyGraph;
208 std::string proxyGraphics;
209 int color24;
210 std::string colorName;
211 int transparency;
212 int plotStyle;
213 DRW::ShadowMode shadow;
214 bool haveExtrusion;
215 std::vector<DRW_Variant*> extData;
217private:
218 DRW_Entity &operator=(const DRW_Entity &) Q_DECL_EQ_DELETE;
219 DRW_Coord extAxisX;
220 DRW_Coord extAxisY;
221 DRW_Variant* curr;
222};
