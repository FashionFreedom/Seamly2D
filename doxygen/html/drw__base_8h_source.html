210class DRW_Coord {
211public:
212 DRW_Coord()
213 : x(0),
214 y(0),
215 z(0)
216 {}
217 DRW_Coord(double ix, double iy, double iz)
218 : x(ix),
219 y(iy),
220 z(iz)
221 {}
222
223 DRW_Coord(const DRW_Coord &data)
224 : x(data.x),
225 y(data.y),
226 z(data.z)
227 {}
228
229 DRW_Coord &operator = (const DRW_Coord& data)
230 {
231 if ( &data == this )
232 {
233 return *this;
234 }
235
236 x = data.x;
237 y = data.y;
238 z = data.z;
239 return *this;
240 }
242 void unitize(){
243 double dist;
244 dist = sqrt(x*x + y*y + z*z);
245 if (dist > 0.0) {
246 x= x/dist;
247 y= y/dist;
248 z= z/dist;
249 }
250 }
251
252public:
253 double x;
254 double y;
255 double z;
256};
