83 QVariant VPE::LineTypeProperty::data(int column, int role) const
84 {
85  if (m_lineTypes.empty())
86  {
87  return QVariant();
88  }
89 
90  int tmpIndex = VProperty::d_ptr->VariantValue.toInt();
91 
92  if (tmpIndex < 0 || tmpIndex >= m_indexList.count())
93  {
94  tmpIndex = 0;
95  }
96 
97  if (column == DPC_Data && Qt::DisplayRole == role)
98  {
99  return m_indexList.at(tmpIndex);
100  }
101  else if (column == DPC_Data && Qt::EditRole == role)
102  {
103  return tmpIndex;
104  }
105  else
106  {
107  return VProperty::data(column, role);

129  QMap<QString, QString>::const_iterator i = m_lineTypes.constBegin();
