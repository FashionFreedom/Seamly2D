247 bool TMainWindow::LoadFile(const QString &path)
248 {
249  if (individualMeasurements == nullptr)
250  {
251  if (!QFileInfo(path).exists())
252  {
253  qCCritical(tMainWindow, "%s", qUtf8Printable(tr("File '%1' doesn't exist!").arg(path)));
254  if (qApp->IsTestMode())
255  {
256  qApp->exit(V_EX_NOINPUT);
257  }
258  return false;
259  }
260 
261  // Check if file already opened
262  QList<TMainWindow*>list = qApp->MainWindows();
263  for (int i = 0; i < list.size(); ++i)
264  {
265  if (list.at(i)->CurrentFile() == path)
266  {
267  list.at(i)->activateWindow();
268  close();
269  return false;
270  }
271  }
272 
273  VlpCreateLock(lock, path);
274 
275  if (!lock->IsLocked())
276  {
277  if (!IgnoreLocking(lock->GetLockError(), path))
278  {
279  return false;
280  }
281  }
282 
283  try
284  {
285  data = new VContainer(qApp->TrVars(), &mUnit);
286 
287  individualMeasurements = new MeasurementDoc(data);
288  individualMeasurements->setSize(&currentSize);
289  individualMeasurements->setHeight(&currentHeight);
290  individualMeasurements->setXMLContent(path);
291 
292  mType = individualMeasurements->Type();
293 
294  if (mType == MeasurementsType::Unknown)
295  {
296  VException e(tr("File has unknown format."));
297  throw e;
298  }
299 
300  if (mType == MeasurementsType::Multisize)
301  {
302  MultiSizeConverter converter(path);
303  m_curFileFormatVersion = converter.GetCurrentFormatVarsion();
304  m_curFileFormatVersionStr = converter.GetVersionStr();
305  individualMeasurements->setXMLContent(converter.Convert());// Read again after conversion
306  }
307  else
308  {
309  IndividualSizeConverter converter(path);
310  m_curFileFormatVersion = converter.GetCurrentFormatVarsion();
311  m_curFileFormatVersionStr = converter.GetVersionStr();
312  individualMeasurements->setXMLContent(converter.Convert());// Read again after conversion
313  }
314 
315  if (!individualMeasurements->eachKnownNameIsValid())
316  {
317  VException e(tr("File contains invalid known measurement(s)."));
318  throw e;
319  }
320 
321  mUnit = individualMeasurements->MUnit();
322  pUnit = mUnit;
323 
324  currentSize = individualMeasurements->BaseSize();
325  currentHeight = individualMeasurements->BaseHeight();
326 
327  ui->labelToolTip->setVisible(false);
328  ui->tabWidget->setVisible(true);
329 
330  mIsReadOnly = individualMeasurements->IsReadOnly();
331  UpdatePadlock(mIsReadOnly);
332 
333  SetCurrentFile(path);
334 
335  InitWindow();
336 
337  const bool freshCall = true;
338  RefreshData(freshCall);
339 
340  if (ui->tableWidget->rowCount() > 0)
341  {
342  ui->tableWidget->selectRow(0);
343  }
344 
345  MeasurementGUI();
346  }
347  catch (VException &exception)
348  {
349  qCCritical(tMainWindow, "%s\n\n%s\n\n%s", qUtf8Printable(tr("File error.")),
350  qUtf8Printable(exception.ErrorMessage()), qUtf8Printable(exception.DetailedInformation()));
351 
352  ui->labelToolTip->setVisible(true);
353  ui->tabWidget->setVisible(false);
354  delete individualMeasurements;
355  individualMeasurements = nullptr;
356  delete data;
357  data = nullptr;
358  lock.reset();
359 
360  if (qApp->IsTestMode())
361  {
362  qApp->exit(V_EX_NOINPUT);
363  }
364  return false;
365  }
366  }
367  else
368  {
369  qApp->NewMainWindow();
370  return qApp->MainWindow()->LoadFile(path);
371  }
372 
373  return true;
374 }

663 bool TMainWindow::eventFilter(QObject *object, QEvent *event)
664 {
665  if (QPlainTextEdit *plainTextEdit = qobject_cast<QPlainTextEdit *>(object))
666  {
667  if (event->type() == QEvent::KeyPress)
668  {
669  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
670  if ((keyEvent->key() == Qt::Key_Enter) || (keyEvent->key() == Qt::Key_Return))
671  {
672  // Ignore Enter key
673  return true;
674  }
675  else if ((keyEvent->key() == Qt::Key_Period) && (keyEvent->modifiers() & Qt::KeypadModifier))
676  {
677  if (qApp->Settings()->GetOsSeparator())
678  {
679  plainTextEdit->insertPlainText(QLocale().decimalPoint());
680  }
681  else
682  {
683  plainTextEdit->insertPlainText(QLocale::c().decimalPoint());
684  }
685  return true;
686  }
687  }
688  }
689  else if (QLineEdit *textEdit = qobject_cast<QLineEdit *>(object))
690  {
691  if (event->type() == QEvent::KeyPress)
692  {
693  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
694  if ((keyEvent->key() == Qt::Key_Period) && (keyEvent->modifiers() & Qt::KeypadModifier))
695  {
696  if (qApp->Settings()->GetOsSeparator())
697  {
698  textEdit->insert(QLocale().decimalPoint());
699  }
700  else
701  {
702  textEdit->insert(QLocale::c().decimalPoint());
703  }
704  return true;
705  }
706  }
707  }
708  else
709  {
710  // pass the event on to the parent class
711  return QMainWindow::eventFilter(object, event);
712  }
713  return false;// pass the event to the widget
714 }
