315 void PreferencesPatternPage::initNotches()
316 {
317  const Unit units = StrToUnits(qApp->Seamly2DSettings()->getUnit());
318  switch (units)
319  {
320  case Unit::Cm:
321  {
322  ui->defaultNotchLength_DoubleSpinBox->setMaximum(4);
323  ui->defaultNotchWidth_DoubleSpinBox->setMaximum(1.25);
324  break;
325  }
326  case Unit::Mm:
327  {
328  ui->defaultNotchLength_DoubleSpinBox->setMaximum(40);
329  ui->defaultNotchWidth_DoubleSpinBox->setMaximum(12.50);
330  break;
331  }
332  case Unit::Inch:
333  default:
334  {
335  ui->defaultNotchLength_DoubleSpinBox->setMaximum(1.50);
336  ui->defaultNotchWidth_DoubleSpinBox->setMaximum(.50);
337  break;
338  }
339  }
340 
341  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("Slit"), "slit");
342  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("T Notch"), "tNotch");
343  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("U Notch"), "uNotch");
344  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("V Internal"), "vInternal");
345  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("V External"), "vExternal");
346  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("Castle"), "castle");
347  ui->defaultNotchType_ComboBox->addItem(QIcon(), tr("Diamond"), "diamond");
348 
349  //----------------------- Get Default Notch
350  int index = ui->defaultNotchType_ComboBox->findData(qApp->Seamly2DSettings()->getDefaultNotchType());
351  if (index != -1)
352  {
353  ui->defaultNotchType_ComboBox->setCurrentIndex(index);
354  }
355 
356  index = ui->defaultNotchColor_ComboBox->findData(qApp->Seamly2DSettings()->getDefaultNotchColor());
357  if (index != -1)
358  {
359  ui->defaultNotchColor_ComboBox->setCurrentIndex(index);
360  }
361  ui->showSeamlineNotch_CheckBox->setChecked(qApp->Seamly2DSettings()->showSeamlineNotch());
362  ui->showSeamAllowanceNotch_CheckBox->setChecked(qApp->Seamly2DSettings()->showSeamAllowanceNotch());
363  ui->defaultNotchLength_DoubleSpinBox->setValue(qApp->Seamly2DSettings()->getDefaultNotchLength());
364  ui->defaultNotchLength_DoubleSpinBox->setSuffix(" " + UnitsToStr(units, true));
365  ui->defaultNotchWidth_DoubleSpinBox->setValue(qApp->Seamly2DSettings()->getDefaultNotchWidth());
366  ui->defaultNotchWidth_DoubleSpinBox->setSuffix(" " + UnitsToStr(units, true));
367 }
