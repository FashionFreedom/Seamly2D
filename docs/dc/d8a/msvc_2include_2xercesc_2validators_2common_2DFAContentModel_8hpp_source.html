38 //
39 // DFAContentModel is the heavy weight derivative of ContentModel that does
40 // all of the non-trivial element content validation. This guy does the full
41 // bore regular expression to DFA conversion to create a DFA that it then
42 // uses in its validation algorithm.
43 //
44 // NOTE: Upstream work insures that this guy will never see a content model
45 // with PCDATA in it. Any model with PCDATA is 'mixed' and is handled
46 // via the MixedContentModel class, since mixed models are very
47 // constrained in form and easily handled via a special case. This
48 // also makes our life much easier here.
49 //

147  // -----------------------------------------------------------------------
148  // Private data members
149  //
150  // fElemMap
151  // fElemMapSize
152  // This is the map of unique input symbol elements to indices into
153  // each state's per-input symbol transition table entry. This is part
154  // of the built DFA information that must be kept around to do the
155  // actual validation.
156  //
157  // fElemMapType
158  // This is a map of whether the element map contains information
159  // related to ANY models.
160  //
161  // fEmptyOk
162  // This is an optimization. While building the transition table we
163  // can see whether this content model would approve of an empty
164  // content (which could happen if everything was optional.) So we
165  // set this flag and short circuit that check, which would otherwise
166  // be ugly and time consuming if we tried to determine it at each
167  // validation call.
168  //
169  // fEOCPos
170  // The NFA position of the special EOC (end of content) node. This
171  // is saved away since its used during the DFA build.
172  //
173  // fFinalStateFlags
174  // This is an array of booleans, one per state (there are
175  // fTransTableSize states in the DFA) that indicates whether that
176  // state is a final state.
177  //
178  // fFollowList
179  // The list of follow positions for each NFA position (i.e. for each
180  // non-epsilon leaf node.) This is only used during the building of
181  // the DFA, and is let go afterwards.
182  //
183  // fHeadNode
184  // This is the head node of our intermediate representation. It is
185  // only non-null during the building of the DFA (just so that it
186  // does not have to be passed all around.) Once the DFA is built,
187  // this is no longer required so its deleted.
188  //
189  // fLeafCount
190  // The count of leaf nodes. This is an important number that set some
191  // limits on the sizes of data structures in the DFA process.
192  //
193  // fLeafList
194  // An array of non-epsilon leaf nodes, which is used during the DFA
195  // build operation, then dropped. These are just references to nodes
196  // pointed to by fHeadNode, so we don't have to clean them up, just
197  // the actually leaf list array itself needs cleanup.
198  //
199  // fLeafListType
200  // Array mapping ANY types to the leaf list.
201  //
202  // fTransTable
203  // fTransTableSize
204  // This is the transition table that is the main by product of all
205  // of the effort here. It is an array of arrays of ints. The first
206  // dimension is the number of states we end up with in the DFA. The
207  // second dimensions is the number of unique elements in the content
208  // model (fElemMapSize). Each entry in the second dimension indicates
209  // the new state given that input for the first dimension's start
210  // state.
211  //
212  // The fElemMap array handles mapping from element indexes to
213  // positions in the second dimension of the transition table.
214  //
215  // fTransTableSize is the number of valid entries in the transition
216  // table, and in the other related tables such as fFinalStateFlags.
217  //
218  // fCountingStates
219  // This is the table holding the minOccurs/maxOccurs for elements
220  // that can be repeated a finite number of times.
221  //
222  // fDTD
223  // Boolean to allow DTDs to validate even with namespace support.
224  //
225  // fIsMixed
226  // DFA ContentModel with mixed PCDATA.
227  // -----------------------------------------------------------------------
228  QName** fElemMap;

249 inline unsigned int
250 DFAContentModel::getNextState(unsigned int currentState,
251  XMLSize_t elementIndex) const {
252 
253  if (currentState == XMLContentModel::gInvalidTrans) {
254  return XMLContentModel::gInvalidTrans;
255  }
256 
257  if (currentState >= fTransTableSize || elementIndex >= fElemMapSize) {
258  ThrowXMLwithMemMgr(ArrayIndexOutOfBoundsException, XMLExcepts::Array_BadIndex, fMemoryManager);
259  }
260 
261  return fTransTable[currentState][elementIndex];
262 }
