117 template <class TVal, class THasher>
118 void ValueHashTableOf<TVal, THasher>::removeAll()
119 {
120  if(isEmpty())
121  return;
122 
123  // Clean up the buckets first
124  for (XMLSize_t buckInd = 0; buckInd < fHashModulus; buckInd++)
125  {
126  // Get the bucket list head for this entry
127  ValueHashTableBucketElem<TVal>* curElem = fBucketList[buckInd];
128  ValueHashTableBucketElem<TVal>* nextElem;
129  while (curElem)
130  {
131  // Save the next element before we hose this one
132  nextElem = curElem->fNext;
133 
134  // delete the current element and move forward
135  // destructor is empty...
136  // curElem->~ValueHashTableBucketElem();
137  fMemoryManager->deallocate(curElem);
138  curElem = nextElem;
139  }
140 
141  // Clean out this entry
142  fBucketList[buckInd] = 0;
143  }
144  fCount = 0;
145 }

178 template <class TVal, class THasher>
179 void ValueHashTableOf<TVal, THasher>::put(void* key, const TVal& valueToAdopt)
180 {
181  // Apply 0.75 load factor to find threshold.
182  XMLSize_t threshold = fHashModulus * 3 / 4;
183 
184  // If we've grown too big, expand the table and rehash.
185  if (fCount >= threshold)
186  rehash();
187 
188  // First see if the key exists already
189  XMLSize_t hashVal;
190  ValueHashTableBucketElem<TVal>* newBucket = findBucketElem(key, hashVal);
191 
192  //
193  // If so,then update its value. If not, then we need to add it to
194  // the right bucket
195  //
196  if (newBucket)
197  {
198  newBucket->fData = valueToAdopt;
199  newBucket->fKey = key;
200  }
201  else
202  {
203  newBucket =
204  new (fMemoryManager->allocate(sizeof(ValueHashTableBucketElem<TVal>)))
205  ValueHashTableBucketElem<TVal>(key, valueToAdopt, fBucketList[hashVal]);
206  fBucketList[hashVal] = newBucket;
207  fCount++;
208  }
209 }

216 template <class TVal, class THasher>
217 void ValueHashTableOf<TVal, THasher>::rehash()
218 {
219  const XMLSize_t newMod = (fHashModulus * 2) + 1;
220 
221  ValueHashTableBucketElem<TVal>** newBucketList =
222  (ValueHashTableBucketElem<TVal>**) fMemoryManager->allocate
223  (
224  newMod * sizeof(ValueHashTableBucketElem<TVal>*)
225  );//new RefHashTableBucketElem<TVal>*[newMod];
226 
227  // Make sure the new bucket list is destroyed if an
228  // exception is thrown.
229  ArrayJanitor<ValueHashTableBucketElem<TVal>*> guard(newBucketList, fMemoryManager);
230 
231  memset(newBucketList, 0, newMod * sizeof(newBucketList[0]));
232 
233 
234  // Rehash all existing entries.
235  for (XMLSize_t index = 0; index < fHashModulus; index++)
236  {
237  // Get the bucket list head for this entry
238  ValueHashTableBucketElem<TVal>* curElem = fBucketList[index];
239 
240  while (curElem)
241  {
242  // Save the next element before we detach this one
243  ValueHashTableBucketElem<TVal>* const nextElem = curElem->fNext;
244 
245  const XMLSize_t hashVal = fHasher.getHashVal(curElem->fKey, newMod);
246  assert(hashVal < newMod);
247 
248  ValueHashTableBucketElem<TVal>* const newHeadElem = newBucketList[hashVal];
249 
250  // Insert at the start of this bucket's list.
251  curElem->fNext = newHeadElem;
252  newBucketList[hashVal] = curElem;
253 
254  curElem = nextElem;
255  }
256  }
257 
258  ValueHashTableBucketElem<TVal>** const oldBucketList = fBucketList;
259 
260  // Everything is OK at this point, so update the
261  // member variables.
262  fBucketList = guard.release();
263  fHashModulus = newMod;
264 
265  // Delete the old bucket list.
266  fMemoryManager->deallocate(oldBucketList);//delete[] oldBucketList;
267 
268 }

271 inline ValueHashTableBucketElem<TVal>* ValueHashTableOf<TVal, THasher>::
272 findBucketElem(const void* const key, XMLSize_t& hashVal)
273 {
274  // Hash the key
275  hashVal = fHasher.getHashVal(key, fHashModulus);
276  assert(hashVal < fHashModulus);
277 
278  // Search that bucket for the key
279  ValueHashTableBucketElem<TVal>* curElem = fBucketList[hashVal];
280  while (curElem)
281  {
282  if (fHasher.equals(key, curElem->fKey))
283  return curElem;
284 
285  curElem = curElem->fNext;
286  }
287  return 0;
288 }

311 template <class TVal, class THasher>
312 void ValueHashTableOf<TVal, THasher>::
313 removeBucketElem(const void* const key, XMLSize_t& hashVal)
314 {
315  // Hash the key
316  hashVal = fHasher.getHashVal(key, fHashModulus);
317  assert(hashVal < fHashModulus);
318 
319  //
320  // Search the given bucket for this key. Keep up with the previous
321  // element so we can patch around it.
322  //
323  ValueHashTableBucketElem<TVal>* curElem = fBucketList[hashVal];
324  ValueHashTableBucketElem<TVal>* lastElem = 0;
325 
326  while (curElem)
327  {
328  if (fHasher.equals(key, curElem->fKey))
329  {
330  if (!lastElem)
331  {
332  // It was the first in the bucket
333  fBucketList[hashVal] = curElem->fNext;
334  }
335  else
336  {
337  // Patch around the current element
338  lastElem->fNext = curElem->fNext;
339  }
340 
341  // Delete the current element
342  // delete curElem;
343  // destructor is empty...
344  // curElem->~ValueHashTableBucketElem();
345  fMemoryManager->deallocate(curElem);
346 
347  fCount--;
348 
349  return;
350  }
351 
352  // Move both pointers upwards
353  lastElem = curElem;
354  curElem = curElem->fNext;
355  }
356 
357  // We never found that key
358  ThrowXMLwithMemMgr(NoSuchElementException, XMLExcepts::HshTbl_NoSuchKeyExists, fMemoryManager);
359 }

458 template <class TVal, class THasher>
459 void ValueHashTableOfEnumerator<TVal, THasher>::findNext()
460 {
461  //
462  // If there is a current element, move to its next element. If this
463  // hits the end of the bucket, the next block will handle the rest.
464  //
465  if (fCurElem)
466  fCurElem = fCurElem->fNext;
467 
468  //
469  // If the current element is null, then we have to move up to the
470  // next hash value. If that is the hash modulus, then we cannot
471  // go further.
472  //
473  if (!fCurElem)
474  {
475  if (++fCurHash == fToEnum->fHashModulus)
476  return;
477 
478  // Else find the next non-empty bucket
479  while (fToEnum->fBucketList[fCurHash]==0)
480  {
481  // Bump to the next hash value. If we max out return
482  if (++fCurHash == fToEnum->fHashModulus)
483  return;
484  }
485  fCurElem = fToEnum->fBucketList[fCurHash];
486  }
487 }
