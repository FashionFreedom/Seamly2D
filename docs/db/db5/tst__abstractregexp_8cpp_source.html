79 int TST_AbstractRegExp::LoadVariables(const QString &checkedLocale)
80 {
81  const QString path = TranslationsPath();
82  const QString file = QString("seamly2d_%1.qm").arg(checkedLocale);
83 
84  if (QFileInfo(path+QLatin1String("/")+file).size() <= 34)
85  {
86  const QString message = QString("Translation variables for locale = %1 is empty. \nFull path: %2/%3")
87  .arg(checkedLocale)
88  .arg(path)
89  .arg(file);
90  QWARN(qUtf8Printable(message));
91 
92  return ErrorSize;
93  }
94 
95  m_vTranslator = new QTranslator(this);
96 
97  if (not m_vTranslator->load(file, path))
98  {
99  const QString message = QString("Can't load translation variables for locale = %1. \nFull path: %2/%3")
100  .arg(checkedLocale)
101  .arg(path)
102  .arg(file);
103  QWARN(qUtf8Printable(message));
104 
105  delete m_vTranslator;
106 
107  return ErrorLoad;
108  }
109 
110  if (not QCoreApplication::installTranslator(m_vTranslator))
111  {
112  const QString message = QString("Can't install translation variables for locale = %1. \nFull path: %2/%3")
113  .arg(checkedLocale)
114  .arg(path)
115  .arg(file);
116  QWARN(qUtf8Printable(message));
117 
118  delete m_vTranslator;
119 
120  return ErrorInstall;
121  }
122 
123  return NoError;
124 }
