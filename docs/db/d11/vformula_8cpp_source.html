250 //---------------------------------------------------------------------------------------------------------------------
251 void VFormula::Eval()
252 {
253  if (data == nullptr)
254  {
255  return;
256  }
257  if (formula.isEmpty())
258  {
259  value = tr("Error");
260  _error = true;
261  dValue = 0;
262  }
263  else
264  {
265  try
266  {
267  QScopedPointer<Calculator> cal(new Calculator());
268  QString expression = qApp->translateVariables()->FormulaFromUser(formula, qApp->Settings()->getOsSeparator());
269  const qreal result = cal->EvalFormula(data->DataVariables(), expression);
270 
271  if (qIsInf(result) || qIsNaN(result))
272  {
273  value = tr("Error");
274  _error = true;
275  dValue = 0;
276  }
277  else
278  {
279  //if result equal 0
280  if (checkZero && qFuzzyIsNull(result))
281  {
282  value = QString("0");
283  _error = true;
284  dValue = 0;
285  }
286  else
287  {
288  dValue = result;
289  value = QString(qApp->LocaleToString(result) + " " + postfix);
290  _error = false;
291  }
292  }
293  }
294  catch (qmu::QmuParserError &error)
295  {
296  value = tr("Error");
297  _error = true;
298  dValue = 0;
299  qDebug() << "\nMath parser error:\n"
300  << "--------------------------------------\n"
301  << "Message: " << error.GetMsg() << "\n"
302  << "Expression: " << error.GetExpr() << "\n"
303  << "--------------------------------------";
304  }
305  }
306 }
