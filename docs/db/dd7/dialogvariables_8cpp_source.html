189  * @brief fillCustomVariables fill data for variables table
190  */
191 void DialogVariables::fillCustomVariables(bool freshCall)
192 {
193  ui->variables_TableWidget->blockSignals(true);
194  ui->variables_TableWidget->clearContents();
195 
196  const QMap<QString, QSharedPointer<CustomVariable> > variables = data->variablesData();
197  QMap<QString, QSharedPointer<CustomVariable> >::const_iterator i;
198  QMap<quint32, QString> map;
199  //Sorting QHash by id
200  for (i = variables.constBegin(); i != variables.constEnd(); ++i)
201  {
202  QSharedPointer<CustomVariable> variable = i.value();
203  map.insert(variable->getIndex(), i.key());
204  }
205 
206  qint32 currentRow = -1;
207  QMapIterator<quint32, QString> iMap(map);
208  ui->variables_TableWidget->setRowCount ( variables.size() );
209  while (iMap.hasNext())
210  {
211  iMap.next();
212  QSharedPointer<CustomVariable> variable = variables.value(iMap.value());
213  currentRow++;
214 
215  addCell(ui->variables_TableWidget, variable->GetName(), currentRow, 0, Qt::AlignVCenter); // name
216  addCell(ui->variables_TableWidget, qApp->LocaleToString(*variable->GetValue()), currentRow, 1,
217  Qt::AlignHCenter | Qt::AlignVCenter, variable->IsFormulaOk()); // calculated value
218 
219  QString formula;
220  try
221  {
222  formula = qApp->translateVariables()->FormulaToUser(variable->GetFormula(), qApp->Settings()->getOsSeparator());
223  }
224  catch (qmu::QmuParserError &error)
225  {
226  Q_UNUSED(error)
227  formula = variable->GetFormula();
228  }
229 
230  addCell(ui->variables_TableWidget, formula, currentRow, 2, Qt::AlignVCenter); // formula
231  }
232 
233  if (freshCall)
234  {
235  ui->variables_TableWidget->resizeColumnsToContents();
236  ui->variables_TableWidget->resizeRowsToContents();
237  }
238 
239  ui->variables_TableWidget->setColumnWidth(0, 350);
240  ui->variables_TableWidget->horizontalHeader()->setStretchLastSection(true);
241  ui->variables_TableWidget->blockSignals(false);
242 }

246 void DialogVariables::fillTable(const QMap<QString, T> &varTable, QTableWidget *table)
247 {
248  SCASSERT(table != nullptr)
249 
250  qint32 currentRow = -1;
251  QMapIterator<QString, T> i(varTable);
252  while (i.hasNext())
253  {
254  i.next();
255  qreal length = *i.value()->GetValue();
256  currentRow++;
257  table->setRowCount ( varTable.size() );
258 
259  QTableWidgetItem *item = new QTableWidgetItem(i.key());
260  item->setTextAlignment(Qt::AlignLeft);
261  table->setItem(currentRow, 0, item);
262 
263  item = new QTableWidgetItem(qApp->LocaleToString(length));
264  item->setTextAlignment(Qt::AlignHCenter);
265  table->setItem(currentRow, 1, item);
266  }
267  table->resizeColumnsToContents();
268  table->resizeRowsToContents();
269  table->verticalHeader()->setDefaultSectionSize(20);
270 }

389 bool DialogVariables::evalVariableFormula(const QString &formula, bool fromUser, VContainer *data, QLabel *label)
390 {
391  const QString postfix = UnitsToStr(qApp->patternUnit());//Show unit in dialog label (cm, mm or inch)
392  if (formula.isEmpty())
393  {
394  label->setText(tr("Error") + " (" + postfix + "). " + tr("Empty field."));
395  label->setToolTip(tr("Empty field"));
396  return false;
397  }
398  else
399  {
400  try
401  {
402  QString f;
403  // Replace line return character with spaces for calc if exist
404  if (fromUser)
405  {
406  f = qApp->translateVariables()->FormulaFromUser(formula, qApp->Settings()->getOsSeparator());
407  }
408  else
409  {
410  f = formula;
411  }
412  f.replace("\n", " ");
413  QScopedPointer<Calculator> cal(new Calculator());
414  const qreal result = cal->EvalFormula(data->DataVariables(), f);
415 
416  if (qIsInf(result) || qIsNaN(result))
417  {
418  label->setText(tr("Error") + " (" + postfix + ").");
419  label->setToolTip(tr("Invalid result. Value is infinite or NaN. Please, check your calculations."));
420  return false;
421  }
422 
423  label->setText(qApp->LocaleToString(result) + " " + postfix);
424  label->setToolTip(tr("Value"));
425  return true;
426  }
427  catch (qmu::QmuParserError &error)
428  {
429  label->setText(tr("Error") + " (" + postfix + "). " + tr("Parser error: %1").arg(error.GetMsg()));
430  label->setToolTip(tr("Parser error: %1").arg(error.GetMsg()));
431  return false;
432  }
433  }
434 }

531 bool DialogVariables::variableUsed(const QString &name) const
532 {
533  const QVector<VFormulaField> expressions = doc->ListExpressions();
534 
535  for(int i = 0; i < expressions.size(); ++i)
536  {
537  if (expressions.at(i).expression.indexOf(name) != -1)
538  {
539  // Eval formula
540  try
541  {
542  QScopedPointer<qmu::QmuTokenParser> cal(new qmu::QmuTokenParser(expressions.at(i).expression, false,
543  false));
544 
545  // Tokens (variables, measurements)
546  if (cal->GetTokens().values().contains(name))
547  {
548  return true;
549  }
550  }
551  catch (const qmu::QmuParserError &)
552  {
553  // Do nothing. Because we not sure if used. A formula is broken.
554  }
555  }
556  }
557  return false;
558 }
