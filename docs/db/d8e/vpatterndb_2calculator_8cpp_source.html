85 /**
86  * @brief eval calculate formula.
87  *
88  * First we try eval expression without adding variables. If it fail, we take tokens from expression and add variables
89  * to parser and try again.
90  *
91  * @param formula string of formula.
92  * @return value of formula.
93  */
94 qreal Calculator::EvalFormula(const QHash<QString, QSharedPointer<VInternalVariable>> *vars, const QString &formula)
95 {
96  // Parser doesn't know any variable on this stage. So, we just use variable factory that for each unknown variable
97  // set value to 0.
98  SetVarFactory(AddVariable, this);
99  SetSepForEval();//Reset separators options
100 
101  SetExpr(formula);
102 
103  qreal result = 0;
104  result = Eval();
105 
106  QMap<int, QString> tokens = this->GetTokens();
107 
108  // Remove "-" from tokens list if exist. If don't do that unary minus operation will broken.
109  RemoveAll(tokens, QStringLiteral("-"));
110 
111  for (int i = 0; i < builInFunctions.size(); ++i)
112  {
113  if (tokens.isEmpty())
114  {
115  break;
116  }
117  RemoveAll(tokens, builInFunctions.at(i));
118  }
119 
120  if (tokens.isEmpty())
121  {
122  return result; // We have found only numbers in expression.
123  }
124 
125  // Add variables to parser because we have deal with expression with variables.
126  InitVariables(vars, tokens, formula);
127  return Eval();
128 }

131 /**
132  * @brief Calculator::InitVariables add variables to parser.
133  *
134  * For optimization purpose we try don't add variables that we don't need.
135  *
136  * @param vars list of variables.
137  * @param tokens all tokens (measurements names, variables with lengths) that parser have found in expression.
138  * @param formula expression, need for throwing better error message.
139  */
140 void Calculator::InitVariables(const QHash<QString, QSharedPointer<VInternalVariable> > *vars,
141  const QMap<int, QString> &tokens, const QString &formula)
142 {
143  QMap<int, QString>::const_iterator i = tokens.constBegin();
144  while (i != tokens.constEnd())
145  {
146  bool found = false;
147  if (vars->contains(i.value()))
148  {
149  DefineVar(i.value(), vars->value(i.value())->GetValue());
150  found = true;
151  }
152 
153  if (found == false && builInFunctions.contains(i.value()))
154  {// We have found built-in function
155  found = true;
156  }
157 
158  if (found == false)
159  {
160  throw qmu::QmuParserError (qmu::ecUNASSIGNABLE_TOKEN, i.value(), formula, i.key());
161  }
162  ++i;
163  }
164 }
