142 //---------------------------------------------------------------------------------------------------------------------
143 IntersectCirclesTool *IntersectCirclesTool::Create(const quint32 _id, const QString &pointName,
144  quint32 firstCircleCenterId,
145  quint32 secondCircleCenterId,
146  QString &firstCircleRadius,
147  QString &secondCircleRadius,
148  CrossCirclesPoint crossPoint,
149  qreal mx, qreal my,
150  bool showPointName,
151  VMainGraphicsScene *scene,
152  VAbstractPattern *doc, VContainer *data,
153  const Document &parse,
154  const Source &typeCreation)
155 {
156  const qreal calcC1Radius = qApp->toPixel(CheckFormula(_id, firstCircleRadius, data));
157  const qreal calcC2Radius = qApp->toPixel(CheckFormula(_id, secondCircleRadius, data));
158 
159  const VPointF c1Point = *data->GeometricObject<VPointF>(firstCircleCenterId);
160  const VPointF c2Point = *data->GeometricObject<VPointF>(secondCircleCenterId);
161 
162  const QPointF point = FindPoint(static_cast<QPointF>(c1Point), static_cast<QPointF>(c2Point), calcC1Radius,
163  calcC2Radius, crossPoint);
164  if (point == QPointF())
165  {
166  const QString msg = tr("<b><big>Can't find intersection point %1 of Circles</big></b><br>"
167  "Using origin point as a place holder until pattern is corrected.")
168  .arg(pointName);
169 
170  QMessageBox msgBox(qApp->getMainWindow());
171  msgBox.setWindowTitle(tr("Point Intersect Circles"));
172  msgBox.setWindowFlags(msgBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
173  msgBox.setWindowIcon(QIcon(":/toolicon/32x32/point_of_intersection_circles.png"));
174  msgBox.setIcon(QMessageBox::Warning);
175  msgBox.setText(msg);
176  msgBox.setStandardButtons(QMessageBox::Ok);
177  msgBox.exec();
178  }
179 
180  quint32 id = _id;
181 
182  VPointF *p = new VPointF(point, pointName, mx, my);
183  p->setShowPointName(showPointName);
184 
185  if (typeCreation == Source::FromGui)
186  {
187  id = data->AddGObject(p);
188  }
189  else
190  {
191  data->UpdateGObject(id, p);
192  if (parse != Document::FullParse)
193  {
194  doc->UpdateToolData(id, data);
195  }
196  }
197 
198  if (parse == Document::FullParse)
199  {
200  VDrawTool::AddRecord(id, Tool::PointOfIntersectionCircles, doc);
201  IntersectCirclesTool *point = new IntersectCirclesTool(doc, data, id, firstCircleCenterId,
202  secondCircleCenterId,
203  firstCircleRadius,
204  secondCircleRadius, crossPoint,
205  typeCreation);
206  scene->addItem(point);
207  InitToolConnections(scene, point);
208  VAbstractPattern::AddTool(id, point);
209  doc->IncrementReferens(c1Point.getIdTool());
210  doc->IncrementReferens(c2Point.getIdTool());
211  return point;
212  }
213  return nullptr;
214 }

217 QPointF IntersectCirclesTool::FindPoint(const QPointF &c1Point, const QPointF &c2Point, qreal c1Radius,
218  qreal c2Radius, const CrossCirclesPoint crossPoint)
219 {
220  QPointF p1, p2;
221  const int res = VGObject::IntersectionCircles(c1Point, c1Radius, c2Point, c2Radius, p1, p2);
222 
223  switch(res)
224  {
225  case 2:
226  if (crossPoint == CrossCirclesPoint::FirstPoint)
227  {
228  return p1;
229  }
230  else
231  {
232  return p2;
233  }
234  case 1:
235  return p1;
236  case 3:
237  case 0:
238  default:
239  return QPointF();
240  }
241 }
