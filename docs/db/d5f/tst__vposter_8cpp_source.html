111 QRect TST_VPoster::PageRect(const QPrinter &printer) const
112 {
113  // Because the Point unit is defined to be 1/72th of an inch
114  // we can't use method pageRect(QPrinter::Point). Our dpi different can be different.
115  // We convert value yourself to pixels.
116  const QRectF rect = printer.pageRect(QPrinter::Millimeter);
117  QRect pageRect(qFloor(ToPixel(rect.x())), qFloor(ToPixel(rect.y())),
118  qFloor(ToPixel(rect.width())), qFloor(ToPixel(rect.height())));
119  return pageRect;
120 }
