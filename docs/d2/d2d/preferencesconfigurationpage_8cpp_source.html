245 void PreferencesConfigurationPage::Apply()
246 {
247  if(!ui->email_LineEdit->text().isEmpty() && !ui->email_LineEdit->hasAcceptableInput())
248  {
249  QMessageBox::warning(this, tr("Email verification"),
250  tr("Email format is not valid."), QMessageBox::Ok);
251 
252  ui->email_LineEdit->setText(QString(""));
253  }
254 
255  VSettings *settings = qApp->Seamly2DSettings();
256 
257  //Designer Info
258  settings->setCompanyName(ui->companyName_LineEdit->text());
259  settings->setContact(ui->contact_LineEdit->text());
260  settings->setAddress(ui->address_LineEdit->text());
261  settings->setCity(ui->city_LineEdit->text());
262  settings->setState(ui->state_LineEdit->text());
263  settings->setZipcode(ui->zipcode_LineEdit->text());
264  settings->setCountry(ui->country_LineEdit->text());
265  settings->setTelephone(ui->telephone_LineEdit->text());
266  settings->setFax(ui->fax_LineEdit->text());
267  settings->setEmail(ui->email_LineEdit->text());
268  settings->setWebsite(ui->website_LineEdit->text());
269 
270  /* Maximum number of commands in undo stack may only be set when the undo stack is empty, since setting it on a
271  * non-empty stack might delete the command at the current index. Calling setUndoLimit() on a non-empty stack
272  * prints a warning and does nothing.*/
273  settings->setUndoCount(ui->undoCount_SpinBox->value());
274  if (m_selectionSoundChanged)
275  {
276  const QString locale = qvariant_cast<QString>(ui->selectionSound_ComboBox->currentText());
277  settings->setSelectionSound(locale);
278  m_selectionSoundChanged = false;
279  }
280  settings->setConfirmItemDelete(ui->confirmItemDelete_CheckBox->isChecked());
281  settings->setConfirmFormatRewriting(ui->confirmFormatRewriting_CheckBox->isChecked());
282 
283  settings->setAutosaveState(ui->autoSave_CheckBox->isChecked());
284  settings->setAutosaveInterval(ui->autoInterval_Spinbox->value());
285 
286  QTimer *autoSaveTimer = qApp->getAutoSaveTimer();
287  SCASSERT(autoSaveTimer)
288 
289  ui->autoSave_CheckBox->isChecked() ? autoSaveTimer->start(ui->autoInterval_Spinbox->value()*60000) : autoSaveTimer->stop();
290 
291  settings->setUseModeType(ui->useModeType_CheckBox->isChecked());
292  settings->setUseLastExportFormat(ui->uselastExportFormat_CheckBox->isChecked());
293  if (m_defaultExportFormatChanged)
294  {
295  const QString format = qvariant_cast<QString>(ui->defaultExportFormat_ComboBox->currentText());
296  settings->setExportFormat(format);
297  m_defaultExportFormatChanged = false;
298  }
299 
300  settings->setShowWelcome(ui->showWelcome_CheckBox->isChecked());
301  settings->setOsSeparator(ui->osOptionCheck->isChecked());
302 
303  if (m_langChanged)
304  {
305  const QString locale = qvariant_cast<QString>(ui->langCombo->currentData());
306  settings->setLocale(locale);
307  m_langChanged = false;
308 
309  qApp->loadTranslations(locale);
310  }
311  if (m_unitChanged)
312  {
313  const QString unit = qvariant_cast<QString>(ui->unitCombo->currentData());
314  settings->SetUnit(unit);
315  m_unitChanged = false;
316  const QString text = tr("The Default unit has been updated and will be used as the default for the next "
317  "pattern you create.");
318  QMessageBox::information(this, QCoreApplication::applicationName(), text);
319  }
320  if (m_labelLangChanged)
321  {
322  const QString locale = qvariant_cast<QString>(ui->labelCombo->currentData());
323  settings->setPointNameLanguage(locale);
324  m_labelLangChanged = false;
325  }
326  if (m_moveSuffixChanged)
327  {
328  const QString locale = qvariant_cast<QString>(ui->moveSuffix_ComboBox->currentData());
329  settings->setMoveSuffix(locale);
330  m_moveSuffixChanged = false;
331  }
332  if (m_rotateSuffixChanged)
333  {
334  const QString locale = qvariant_cast<QString>(ui->rotateSuffix_ComboBox->currentData());
335  settings->setRotateSuffix(locale);
336  m_rotateSuffixChanged = false;
337  }
338  if (m_mirrorByAxisSuffixChanged)
339  {
340  const QString locale = qvariant_cast<QString>(ui->mirrorByAxisSuffix_ComboBox->currentData());
341  settings->setMirrorByAxisSuffix(locale);
342  m_mirrorByAxisSuffixChanged = false;
343  }
344  if (m_mirrorByLineSuffixChanged)
345  {
346  const QString locale = qvariant_cast<QString>(ui->mirrorByLineSuffix_ComboBox->currentData());
347  settings->setMirrorByLineSuffix(locale);
348  m_mirrorByLineSuffixChanged = false;
349  }
350 }
