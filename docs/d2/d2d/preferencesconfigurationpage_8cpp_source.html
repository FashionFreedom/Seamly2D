242 void PreferencesConfigurationPage::Apply()
243 {
244  VSettings *settings = qApp->Seamly2DSettings();
245  /* Maximum number of commands in undo stack may only be set when the undo stack is empty, since setting it on a
246  * non-empty stack might delete the command at the current index. Calling setUndoLimit() on a non-empty stack
247  * prints a warning and does nothing.*/
248  settings->SetUndoCount(ui->undoCount_SpinBox->value());
249  if (m_selectionSoundChanged)
250  {
251  const QString locale = qvariant_cast<QString>(ui->selectionSound_ComboBox->currentText());
252  settings->setSelectionSound(locale);
253  m_selectionSoundChanged = false;
254  }
255  settings->setConfirmItemDelete(ui->confirmItemDelete_CheckBox->isChecked());
256  settings->setConfirmFormatRewriting(ui->confirmFormatRewriting_CheckBox->isChecked());
257 
258  settings->SetAutosaveState(ui->autoSave_CheckBox->isChecked());
259  settings->setAutosaveInterval(ui->autoInterval_Spinbox->value());
260 
261  QTimer *autoSaveTimer = qApp->getAutoSaveTimer();
262  SCASSERT(autoSaveTimer)
263 
264  ui->autoSave_CheckBox->isChecked() ? autoSaveTimer->start(ui->autoInterval_Spinbox->value()*60000) : autoSaveTimer->stop();
265 
266  settings->setUseModeType(ui->useModeType_CheckBox->isChecked());
267  settings->setUseLastExportFormat(ui->uselastExportFormat_CheckBox->isChecked());
268  if (m_defaultExportFormatChanged)
269  {
270  const QString format = qvariant_cast<QString>(ui->defaultExportFormat_ComboBox->currentText());
271  settings->setExportFormat(format);
272  m_defaultExportFormatChanged = false;
273  }
274 
275  settings->SetOsSeparator(ui->osOptionCheck->isChecked());
276  //settings->SetSendReportState(ui->sendReportCheck->isChecked());
277 
278  if (m_langChanged || m_systemChanged)
279  {
280  const QString locale = qvariant_cast<QString>(ui->langCombo->currentData());
281  settings->SetLocale(locale);
282  m_langChanged = false;
283 
284  const QString code = qvariant_cast<QString>(ui->systemCombo->currentData());
285  settings->SetPMSystemCode(code);
286  m_systemChanged = false;
287 
288  qApp->loadTranslations(locale);
289  }
290  if (m_unitChanged)
291  {
292  const QString unit = qvariant_cast<QString>(ui->unitCombo->currentData());
293  settings->SetUnit(unit);
294  m_unitChanged = false;
295  const QString text = tr("The Default unit has been updated and will be used as the default for the next "
296  "pattern you create.");
297  QMessageBox::information(this, QCoreApplication::applicationName(), text);
298  }
299  if (m_labelLangChanged)
300  {
301  const QString locale = qvariant_cast<QString>(ui->labelCombo->currentData());
302  settings->SetLabelLanguage(locale);
303  m_labelLangChanged = false;
304  }
305  if (m_moveSuffixChanged)
306  {
307  const QString locale = qvariant_cast<QString>(ui->moveSuffix_ComboBox->currentData());
308  settings->setMoveSuffix(locale);
309  m_moveSuffixChanged = false;
310  }
311  if (m_rotateSuffixChanged)
312  {
313  const QString locale = qvariant_cast<QString>(ui->rotateSuffix_ComboBox->currentData());
314  settings->setRotateSuffix(locale);
315  m_rotateSuffixChanged = false;
316  }
317  if (m_mirrorByAxisSuffixChanged)
318  {
319  const QString locale = qvariant_cast<QString>(ui->mirrorByAxisSuffix_ComboBox->currentData());
320  settings->setMirrorByAxisSuffix(locale);
321  m_mirrorByAxisSuffixChanged = false;
322  }
323  if (m_mirrorByLineSuffixChanged)
324  {
325  const QString locale = qvariant_cast<QString>(ui->mirrorByLineSuffix_ComboBox->currentData());
326  settings->setMirrorByLineSuffix(locale);
327  m_mirrorByLineSuffixChanged = false;
328  }
329 }
