183 QString PageFormatCombobox::MakeHelpTemplateList()
184 {
185  QString out = "\n";
186 
187  auto cntr = static_cast<int>(PaperSizeFormat::A0);
188  foreach(auto& v, initFormats())
189  {
190  if (cntr <= static_cast<int>(PaperSizeFormat::Roll44in))// Don't include custom template
191  {
192  out += QLatin1String("\t") + v.first + QLatin1String(" = ") +
193  QString::number(static_cast<int>(v.second)) + QLatin1String("\n");
194  }
195  }
196  return out;
