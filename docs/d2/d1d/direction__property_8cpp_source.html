53 QVariant VPE::DirectionProperty::data(int column, int role) const
54 {
55  if (m_directions.empty())
56  {
57  return QVariant();
58  }
59 
60  int tmpIndex = VProperty::d_ptr->VariantValue.toInt();
61 
62  if (tmpIndex < 0 || tmpIndex >= m_indexList.count())
63  {
64  tmpIndex = 0;
65  }
66 
67  if (column == DPC_Data && Qt::DisplayRole == role)
68  {
69  return m_indexList.at(tmpIndex);

71  else if (column == DPC_Data && Qt::EditRole == role)
72  {
73  return tmpIndex;
74  }
75  else
76  {
77  return VProperty::data(column, role);
