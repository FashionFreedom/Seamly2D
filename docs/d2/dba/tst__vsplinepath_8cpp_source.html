126 void TST_VSplinePath::TestRotation()
127 {
128  QFETCH(QVector<VSplinePoint>, originPoints);
129  QFETCH(QPointF, originPoint);
130  QFETCH(qreal, degrees);
131  QFETCH(QString, prefix);
132  QFETCH(QVector<VSplinePoint>, rotatedPoints);
133 
134  const VSplinePath rotatedPath(rotatedPoints);
135 
136  const VSplinePath originPath(originPoints);
137  const VSplinePath res = originPath.Rotate(originPoint, degrees, prefix);
138 
139  QCOMPARE(originPath.GetLength(), res.GetLength());
140  QCOMPARE(rotatedPath.GetLength(), res.GetLength());
141  QCOMPARE(rotatedPath.CountPoints(), res.CountPoints());
142 
143  const QString errorMsg = QString("The name doesn't contain the prefix '%1'.").arg(prefix);
144  QVERIFY2(res.name().endsWith(prefix), qUtf8Printable(errorMsg));
145 
146  const QVector<VSplinePoint> resPoints = res.GetSplinePath();
147 
148  for (int i=0; i < resPoints.size(); ++i)
149  {
150  const VSplinePoint resPoint = resPoints.at(i);
151  const VSplinePoint rotatedPoint = rotatedPoints.at(i);
152 
153  QCOMPARE(rotatedPoint.Angle1(), resPoint.Angle1());
154  QCOMPARE(rotatedPoint.Angle2(), resPoint.Angle2());
155  QCOMPARE(rotatedPoint.Length1(), resPoint.Length1());
156  QCOMPARE(rotatedPoint.Length2(), resPoint.Length2());
157  QCOMPARE(rotatedPoint.P().toQPointF().toPoint(), resPoint.P().toQPointF().toPoint());
158  }
159 }
