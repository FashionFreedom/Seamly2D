164 /**
165  * @brief ContextMenu show context menu for tool.
166  * @param itemId id of point.
167  * @param event context menu event.
168  * @param showRemove true - tool enable option delete.
169  * @param ref true - do not ignore referens value.
170  */
171 void VDrawTool::ContextMenu(QGraphicsSceneContextMenuEvent *event, quint32 itemId,
172  const RemoveOption &showRemove, const Referens &ref)
173 {
174  SCASSERT(event != nullptr)
175 
176  if (m_suppressContextMenu)
177  {
178  return;
179  }
180 
181  GOType itemType = GOType::Unknown;
182  if(itemId != NULL_ID)
183  {
184  try
185  {
186  itemType = data.GetGObject(itemId)->getType();
187  }
188  catch (const VExceptionBadId &error)
189  { // Possible case. Parent was deleted, but the node object is still here.
190  qWarning() << qUtf8Printable(error.ErrorMessage());
191  }
192  }
193 
194  qCDebug(vTool, "Creating tool context menu.");
195  QMenu menu;
196  QAction *actionOption = menu.addAction(QIcon::fromTheme("preferences-other"), tr("Properties"));
197 
198  // Show object name menu item
199  QAction *actionShowPointName = menu.addAction(QIcon("://icon/16x16/open_eye.png"), tr("Show Point Name"));
200  actionShowPointName->setCheckable(true);
201 
202  if (itemType == GOType::Point)
203  {
204  actionShowPointName->setChecked(isPointNameVisible(itemId));
205  }
206  else
207  {
208  actionShowPointName->setVisible(false);
209  }
210 
211  QAction *actionDelete = menu.addAction(QIcon::fromTheme("edit-delete"), tr("Delete"));
212  if (showRemove == RemoveOption::Enable)
213  {
214  if (ref == Referens::Follow)
215  {
216  if (_referens > 1)
217  {
218  qCDebug(vTool, "Delete disabled. Tool has children.");
219  actionDelete->setEnabled(false);
220  }
221  else
222  {
223  qCDebug(vTool, "Delete enabled. Tool has no children.");
224  actionDelete->setEnabled(true);
225  }
226  }
227  else
228  {
229  qCDebug(vTool, "Delete enabled. Ignore referens value.");
230  actionDelete->setEnabled(true);
231  }
232  }
233  else
234  {
235  qCDebug(vTool, "Delete disabled.");
236  actionDelete->setEnabled(false);
237  }
238 
239  // Add Group Item menu item
240  QMap<quint32,QString> groupsNotContainingItem = doc->getGroupsContainingItem(this->getId(), itemId, false);
241  QActionGroup* actionAddGroupMenu= new QActionGroup(this);
242 
243  if(not groupsNotContainingItem.empty())
244  {
245  QMenu *menuAddGroupItem = menu.addMenu(QIcon("://icon/32x32/add.png"), tr("Add Group Object"));
246  QStringList list = QStringList(groupsNotContainingItem.values());
247  list.sort(Qt::CaseInsensitive);
248 
249  for(int i=0; i<list.count(); ++i)
250  {
251  QAction *actionAddGroupItem = menuAddGroupItem->addAction(list[i]);
252  actionAddGroupMenu->addAction(actionAddGroupItem);
253  const quint32 groupId = groupsNotContainingItem.key(list[i]);
254  actionAddGroupItem->setData(groupId);
255  groupsNotContainingItem.remove(groupId); // delete any duplicate groups
256  }
257  }
258 
259  // Remove Group Item menu item
260  QMap<quint32,QString> groupsContainingItem = doc->getGroupsContainingItem(this->getId(), itemId, true);
261  QActionGroup* actionDeleteGroupMenu = new QActionGroup(this);
262 
263  if(not groupsContainingItem.empty())
264  {
265  QMenu *menuRemoveGroupItem = menu.addMenu(QIcon("://icon/32x32/remove.png"), tr("Remove Group Object"));
266 
267  QStringList list = QStringList(groupsContainingItem.values());
268  list.sort(Qt::CaseInsensitive);
269 
270  for(int i=0; i<list.count(); ++i)
271  {
272  QAction *actionRemoveGroupItem = menuRemoveGroupItem->addAction(list[i]);
273  actionDeleteGroupMenu->addAction(actionRemoveGroupItem);
274  const quint32 groupId = groupsContainingItem.key(list[i]);
275  actionRemoveGroupItem->setData(groupId);
276  groupsContainingItem.remove(groupId);
277  }
278  }
279 
280  QAction *selectedAction = menu.exec(event->screenPos());
281  if(selectedAction == nullptr)
282  {
283  return;
284  }
285  if (selectedAction == actionOption)
286  {
287  qCDebug(vTool, "Show options.");
288  qApp->getSceneView()->itemClicked(nullptr);
289  m_dialog = QSharedPointer<Dialog>(new Dialog(getData(), m_id, qApp->getMainWindow()));
290  m_dialog->setModal(true);
291 
292  connect(m_dialog.data(), &DialogTool::DialogClosed, this, &VDrawTool::FullUpdateFromGuiOk);
293  connect(m_dialog.data(), &DialogTool::DialogApplied, this, &VDrawTool::FullUpdateFromGuiApply);
294 
295  this->setDialog();
296 
297  m_dialog->show();
298  }
299  else if (selectedAction == actionDelete)
300  {
301  qCDebug(vTool, "Deleting Object.");
302  deleteTool(); // do not catch exception here
303  return; //Leave this method immediately after call!!!
304  }
305  else if (selectedAction == actionShowPointName)
306  {
307  updatePointNameVisibility(itemId, selectedAction->isChecked());
308  }
309  else if (selectedAction->actionGroup() == actionAddGroupMenu)
310  {
311  quint32 groupId = selectedAction->data().toUInt();
312  QDomElement item = doc->addGroupItem(this->getId(), itemId, groupId);
313 
314  VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
315  SCASSERT(scene != nullptr)
316  scene->clearSelection();
317 
318  VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
319  SCASSERT(window != nullptr)
320  {
321  AddGroupItem *command = new AddGroupItem(item, doc, groupId);
322  connect(command, &AddGroupItem::updateGroups, window, &VAbstractMainWindow::updateGroups);
323  qApp->getUndoStack()->push(command);
324  }
325  }
326  else if (selectedAction->actionGroup() == actionDeleteGroupMenu)
327  {
328  quint32 groupId = selectedAction->data().toUInt();
329  QDomElement item = doc->removeGroupItem(this->getId(), itemId, groupId);
330 
331  VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
332  SCASSERT(scene != nullptr)
333  scene->clearSelection();
334 
335  VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
336  SCASSERT(window != nullptr)
337  {
338  RemoveGroupItem *command = new RemoveGroupItem(item, doc, groupId);
339  connect(command, &RemoveGroupItem::updateGroups, window, &VAbstractMainWindow::updateGroups);
340  qApp->getUndoStack()->push(command);
341  }
342  }
343 }
