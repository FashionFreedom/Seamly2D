246 void DialogMove::ShowDialog(bool click)
247 {
248  if (stage1 && not click)
249  {
250  if (m_objects.isEmpty())
251  {
252  return;
253  }
254 
255  stage1 = false;
256  prepare = true;
257 
258  VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
259  SCASSERT(scene != nullptr)
260  scene->clearSelection();
261 
262  VisToolMove *operation = qobject_cast<VisToolMove *>(vis);
263  SCASSERT(operation != nullptr)
264  operation->setObjects(sourceToObjects(m_objects));
265  operation->VisualMode();
266 
267  VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
268  SCASSERT(window != nullptr)
269  connect(operation, &VisToolMove::ToolTip, window, &VAbstractMainWindow::ShowToolTip);
270 
271  scene->ToggleArcSelection(false);
272  scene->ToggleElArcSelection(false);
273  scene->ToggleSplineSelection(false);
274  scene->ToggleSplinePathSelection(false);
275 
276  scene->ToggleArcHover(false);
277  scene->ToggleElArcHover(false);
278  scene->ToggleSplineHover(false);
279  scene->ToggleSplinePathHover(false);
280 
281  qApp->getSceneView()->allowRubberBand(false);
282  }
283  else if (!stage2 && !stage1 && prepare && click)
284  {
285  VisToolMove *operation = qobject_cast<VisToolMove *>(vis);
286  SCASSERT(operation != nullptr)
287 
288  if (operation->LengthValue() > 0)
289  {
290  angleFormula = qApp->TrVars()->FormulaToUser(operation->Angle(), qApp->Settings()->getOsSeparator());
291  lengthFormula = qApp->TrVars()->FormulaToUser(operation->Length(), qApp->Settings()->getOsSeparator());
292  operation->SetAngle(angleFormula);
293  operation->SetLength(lengthFormula);
294 
295  operation->RefreshGeometry();
296  emit ToolTip(operation->CurrentToolTip());
297  stage2 = true;
298  }
299  }
300  else if (!stage1 && stage2 && prepare && click)
301  {
302  VisToolMove *operation = qobject_cast<VisToolMove *>(vis);
303  SCASSERT(operation != nullptr)
304 
305  if (QGuiApplication::keyboardModifiers() == Qt::ControlModifier)
306  {
307  if (!useOriginPoint)
308  {
309  operation->setOriginPointId(NULL_ID);
310  SetObject(NULL_ID, ui->rotationPoint_ComboBox, QString());
311  operation->RefreshGeometry();
312  }
313  useOriginPoint = false;
314  }
315  else
316  {
317  SetAngle(operation->Angle());
318  SetLength(operation->Length());
319  setRotation(operation->Rotation());
320  setModal(true);
321  emit ToolTip("");
322  angleTimer->start();
323  lengthTimer->start();
324  rotationTimer->start();
325  show();
326  }
327  }
328 }

351 void DialogMove::SelectedObject(bool selected, quint32 id, quint32 tool)
352 {
353  Q_UNUSED(tool)
354  if (stage1)
355  {
356  auto object = std::find_if(m_objects.begin(), m_objects.end(),
357  [id](const SourceItem &item) { return item.id == id; });
358 
359  if (selected)
360  {
361  if (object == m_objects.cend())
362  {
363  SourceItem item;
364  item.id = id;
365  m_objects.append(item);
366  }
367  }
368  else
369  {
370  if (object != m_objects.end())
371  {
372  m_objects.erase(object);
373  }
374  }
375  }

444 //---------------------------------------------------------------------------------------------------------------------
445 void DialogMove::suffixChanged()
446 {
447  QLineEdit* edit = qobject_cast<QLineEdit*>(sender());
448  if (edit)
449  {
450  const QString suffix = edit->text();
451  if (suffix.isEmpty())
452  {
453  flagName = false;
454  ChangeColor(ui->suffix_Label, Qt::red);
455  CheckState();
456  return;
457  }
458  else
459  {
460  if (m_suffix != suffix)
461  {
462  QRegularExpression rx(NameRegExp());
463  const QStringList uniqueNames = VContainer::AllUniqueNames();
464  for (int i=0; i < uniqueNames.size(); ++i)
465  {
466  const QString name = uniqueNames.at(i) + suffix;
467  if (not rx.match(name).hasMatch() || not data->IsUnique(name))
468  {
469  flagName = false;
470  ChangeColor(ui->suffix_Label, Qt::red);
471  CheckState();
472  return;
473  }
474  }
475  }
476  }
477 
478  flagName = true;
479  ChangeColor(ui->suffix_Label, okColor);
480  }
481  CheckState();
482 }
