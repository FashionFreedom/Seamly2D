33 //
34 // MixedContentModel is a derivative of the abstract content model base
35 // class that handles the special case of mixed model elements. If an element
36 // is mixed model, it has PCDATA as its first possible content, followed
37 // by an alternation of the possible children. The children cannot have any
38 // numeration or order, so it must look like this:
39 //
40 // <!ELEMENT Foo ((#PCDATA|a|b|c|)*)>
41 //
42 // So, all we have to do is to keep an array of the possible children and
43 // validate by just looking up each child being validated by looking it up
44 // in the list.
45 //
46 class MixedContentModel : public XMLContentModel

186 inline void MixedContentModel::checkUniqueParticleAttribution
187  (
188  SchemaGrammar* const
189  , GrammarResolver* const
190  , XMLStringPool* const
191  , XMLValidator* const
192  , unsigned int* const pContentSpecOrgURI
193  , const XMLCh* /*pComplexTypeName*/ /*= 0*/
194  )
195 {
196  // rename back
197  unsigned int i = 0;
198  for (i = 0; i < fCount; i++) {
199  unsigned int orgURIIndex = fChildren[i]->getURI();
200  if ((orgURIIndex != XMLContentModel::gEOCFakeId) &&
201  (orgURIIndex != XMLElementDecl::fgInvalidElemId) &&
202  (orgURIIndex != XMLElementDecl::fgPCDataElemId))
203  fChildren[i]->setURI(pContentSpecOrgURI[orgURIIndex]);
204  }
205 
206  // for mixed content model, it's only a sequence
207  // UPA checking is not necessary
208 }
