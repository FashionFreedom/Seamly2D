103  VAbstractPattern *doc, VContainer *data)

110  const QString suffix = dialogTool->getSuffix();

122 //---------------------------------------------------------------------------------------------------------------------
123 VToolMirrorByLine *VToolMirrorByLine::Create(const quint32 _id, quint32 firstLinePointId, quint32 secondLinePointId,
124  const QString &suffix, const QVector<SourceItem> &source,
125  const QVector<DestinationItem> &destination, VMainGraphicsScene *scene,
126  VAbstractPattern *doc, VContainer *data, const Document &parse,
127  const Source &typeCreation)
128 {
129  const auto firstPoint = *data->GeometricObject<VPointF>(firstLinePointId);
130  const QPointF fPoint = static_cast<QPointF>(firstPoint);
131 
132  const auto secondPoint = *data->GeometricObject<VPointF>(secondLinePointId);
133  const QPointF sPoint = static_cast<QPointF>(secondPoint);
134 
135  QVector<DestinationItem> dest = destination;
136 
137  quint32 id = _id;
138  createDestination(typeCreation, id, dest, source, fPoint, sPoint, suffix, doc, data, parse);
139 
140  if (parse == Document::FullParse)
141  {
142  VDrawTool::AddRecord(id, Tool::MirrorByLine, doc);
143  VToolMirrorByLine *tool = new VToolMirrorByLine(doc, data, id, firstLinePointId, secondLinePointId, suffix,
144  source, dest, typeCreation);
145  scene->addItem(tool);
146  initOperationToolConnections(scene, tool);
147  VAbstractPattern::AddTool(id, tool);
148  doc->IncrementReferens(firstPoint.getIdTool());
149  doc->IncrementReferens(secondPoint.getIdTool());
150  for (int i = 0; i < source.size(); ++i)
151  {
152  const SourceItem item = source.at(i);
153  doc->IncrementReferens(data->GetGObject(item.id)->getIdTool());
154  }
155  return tool;
156  }
157  return nullptr;
158 }
