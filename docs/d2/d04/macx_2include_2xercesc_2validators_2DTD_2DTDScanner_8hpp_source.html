176  // -----------------------------------------------------------------------
177  // Private data members
178  //
179  // fDocTypeHandler
180  // This holds the optional doc type handler that can be installed
181  // and used to call back for all markup events. It is DTD specific.
182  //
183  // fDumAttDef
184  // fDumElemDecl
185  // fDumEntityDecl
186  // These are dummy objects into which mark decls are parsed when
187  // they are just overrides of previously declared markup decls. In
188  // such situations, the first one wins but we need to have somewhere
189  // to parse them into. So these are lazily created and used as needed
190  // when such markup decls are seen.
191  //
192  // fInternalSubset
193  // This is used to track whether we are in the internal subset or not,
194  // in which case we are in the external subset.
195  //
196  // fNextAttrId
197  // Since att defs are per-element, we don't have a validator wide
198  // attribute def pool. So we use a simpler data structure in each
199  // element decl to store its att defs, and we use this simple counter
200  // to apply a unique id to each new attribute.
201  //
202  // fDTDGrammar
203  // The DTD information we scanned like element decl, attribute decl
204  // are stored in this Grammar.
205  //
206  // fBufMgr
207  // This is the buffer manager of the scanner. This is provided as a
208  // convenience so that the DTDScanner doesn't have to create its own
209  // buffer manager during the parse process.
210  //
211  // fReaderMgr
212  // This is a pointer to the reader manager that is being used by the scanner.
213  //
214  // fScanner
215  // The pointer to the scanner to which this DTDScanner belongs
216  //
217  // fPEntityDeclPool
218  // This is a pool of EntityDecl objects, which contains all of the
219  // parameter entities that are declared in the DTD subsets.
220  //
221  // fEmptyNamespaceId
222  // The uri for all DTD decls
223  //
224  // fDocTypeReaderId
225  // The original reader in the fReaderMgr - to be compared against the
226  // current reader to decide whether we are processing an external/internal
227  // declaration
228  // -----------------------------------------------------------------------
229  MemoryManager* fMemoryManager;
230  MemoryManager* fGrammarPoolMemoryManager;
231  DocTypeHandler* fDocTypeHandler;
