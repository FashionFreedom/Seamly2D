476 void ImageDialog::widthChanged(qreal width)
477 {
478  blockSignals(true);
479  qreal oldWidth = m_image.width;
480  qreal height = m_image.height;
481 
482  if (m_image.units != Unit::Px)
483  {
484  width = qApp->toPixel(width);
485  }
486 
487  width = width < m_minDimension ? m_minDimension : width;
488  width = width > m_maxDimension ? m_maxDimension : width;
489 
490  if (m_image.aspectLocked)
491  {
492  height = m_image.height * width / oldWidth;
493 
494  if (height > m_maxDimension)
495  {
496  width = m_maxDimension * oldWidth / m_image.height;
497  height = m_maxDimension;
498  }
499  else if (height < m_minDimension)
500  {
501  width = m_minDimension * oldWidth / m_image.height;
502  height = m_minDimension;
503  }
504  }
505 
506  setWidth(width);
507  setXScale(m_image.width / m_pixmapWidth * 100);
508  setHeight(height);
509  setYScale(m_image.height / m_pixmapHeight * 100);
510 
511  blockSignals(false);
