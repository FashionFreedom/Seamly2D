265  // -----------------------------------------------------------------------
266  // Private data members
267  //
268  // fElemDeclPool
269  // This is the element decl pool. It contains all of the elements
270  // declared in the Schema (and their associated attributes.)
271  //
272  // fElemNonDeclPool
273  // This is the element decl pool that is is populated as new elements
274  // are seen in the XML document (not declared in the Schema), and they
275  // are given default characteristics.
276  //
277  // fGroupElemDeclPool
278  // This is the element decl pool for elements in a group that are
279  // referenced in different scope. It contains all of the elements
280  // declared in the Schema (and their associated attributes.)
281  //
282  // fNotationDeclPool
283  // This is a pool of NotationDecl objects, which contains all of the
284  // notations declared in the Schema.
285  //
286  // fTargetNamespace
287  // Target name space for this grammar.
288  //
289  // fAttributeDeclRegistry
290  // Global attribute declarations
291  //
292  // fComplexTypeRegistry
293  // Stores complexType declaration info
294  //
295  // fGroupInfoRegistry
296  // Stores global <group> declaration info
297  //
298  // fAttGroupInfoRegistry
299  // Stores global <attributeGroup> declaration info
300  //
301  // fDatatypeRegistry
302  // Datatype validator factory
303  //
304  // fValidSubstitutionGroups
305  // Valid list of elements that can substitute a given element
306  //
307  // fIDRefList
308  // List of ids of schema declarations extracted during schema grammar
309  // traversal
310  //
311  // fValidated
312  // Indicates if the content of the Grammar has been pre-validated

476 // Element Decl
477 inline XMLSize_t SchemaGrammar::getElemId (const unsigned int uriId
478  , const XMLCh* const baseName
479  , const XMLCh* const
480  , unsigned int scope ) const
481 {
482  //
483  // In this case, we don't return zero to mean 'not found', so we have to
484  // map it to the official not found value if we don't find it.
485  //
486  const SchemaElementDecl* decl = fElemDeclPool->getByKey(baseName, uriId, scope);
487  if (!decl) {
488 
489  decl = fGroupElemDeclPool->getByKey(baseName, uriId, scope);
490 
491  if (!decl)
492  return XMLElementDecl::fgInvalidElemId;
493  }
494  return decl->getId();
495 }

500  , unsigned int scope ) const

503 
504  if (!decl) {
505 
506  decl = fGroupElemDeclPool->getByKey(baseName, uriId, scope);
507 
508  if (!decl && fElemNonDeclPool)

510  }
511 
512  return decl;

555 inline XMLSize_t
556 SchemaGrammar::putElemDecl(XMLElementDecl* const elemDecl,
557  const bool notDeclared)
558 {
559  if (notDeclared)
560  {
561  if(!fElemNonDeclPool)
562  fElemNonDeclPool = new (fMemoryManager) RefHash3KeysIdPool<SchemaElementDecl>(29, true, 128, fMemoryManager);
563  return fElemNonDeclPool->put(elemDecl->getBaseName(), elemDecl->getURI(), ((SchemaElementDecl* )elemDecl)->getEnclosingScope(), (SchemaElementDecl*) elemDecl);
564  }
565 
566  return fElemDeclPool->put(elemDecl->getBaseName(), elemDecl->getURI(), ((SchemaElementDecl* )elemDecl)->getEnclosingScope(), (SchemaElementDecl*) elemDecl);
567 }
