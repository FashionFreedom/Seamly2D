155 bool TST_QmuTokenParser::IsSingleFromUser(const QString &formula)
156 {
157  if (formula.isEmpty())
158  {
159  return false;// if don't know say no
160  }
161 
162  QMap<int, QString> tokens;
163  QMap<int, QString> numbers;
164 
165  try
166  {
167  QScopedPointer<qmu::QmuTokenParser> cal(new qmu::QmuTokenParser(formula, true, true));
168  tokens = cal->GetTokens();// Tokens (variables, measurements)
169  numbers = cal->GetNumbers();// All numbers in expression
170  }
171  catch (const qmu::QmuParserError &error)
172  {
173  Q_UNUSED(error)
174  return false;// something wrong with formula, say no
175  }
176 
177  // Remove "-" from tokens list if exist. If don't do that unary minus operation will broken.
178  qmu::QmuFormulaBase::RemoveAll(tokens, QLocale().negativeSign());
179 
180  if (tokens.isEmpty() && numbers.size() == 1)
181  {
182  return true;
183  }
184  else
185  {
186  return false;
187  }
188 }
