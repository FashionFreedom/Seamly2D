528  // -----------------------------------------------------------------------
529  // Data members
530  //
531  // fBufferSize
532  // Maximum input buffer size
533  //
534  // fLowWaterMark
535  // The low water mark for the raw byte buffer.
536  //
537  // fAttrList
538  // Every time we get a new element start tag, we have to pass to
539  // the document handler the attributes found. To make it more
540  // efficient we keep this ref vector of XMLAttr objects around. We
541  // just reuse it over and over, allowing it to grow to meet the
542  // peak need.
543  //
544  // fBufMgr
545  // This is a manager for temporary buffers used during scanning.
546  // For efficiency we must use a set of static buffers, but we have
547  // to insure that they are not incorrectly reused. So this manager
548  // provides the smarts to hand out buffers as required.
549  //
550  // fDocHandler
551  // The client code's document handler. If zero, then no document
552  // handler callouts are done. We don't adopt it.
553  //
554  // fDocTypeHandler
555  // The client code's document type handler (used by DTD Validator).
556  //
557  // fDoNamespaces
558  // This flag indicates whether the client code wants us to do
559  // namespaces or not. If the installed validator indicates that it
560  // has to do namespaces, then this is ignored.
561  //
562  // fEntityHandler
563  // The client code's entity handler. If zero, then no entity handler
564  // callouts are done. We don't adopt it.
565  //
566  // fErrorReporter
567  // The client code's error reporter. If zero, then no error reporter
568  // callouts are done. We don't adopt it.
569  //
570  // fErrorHandler
571  // The client code's error handler. Need to store this info for
572  // Schema parse error handling.
573  //
574  // fPSVIHandler
575  // The client code's PSVI handler.
576  //
577  // fExitOnFirstFatal
578  // This indicates whether we bail out on the first fatal XML error
579  // or not. It defaults to true, which is the strict XML way, but it
580  // can be changed.
581  //
582  // fValidationConstraintFatal
583  // This indicates whether we treat validation constraint errors as
584  // fatal errors or not. It defaults to false, but it can be changed.
585  //
586  // fIDRefList
587  // This is a list of XMLRefInfo objects. This member lets us do all
588  // needed ID-IDREF balancing checks.
589  //
590  // fInException
591  // To avoid a circular freakout when we catch an exception and emit
592  // it, which would normally throw again if the 'fail on first error'
593  // flag is one.
594  //
595  // fReaderMgr
596  // This is the reader manager, from which we get characters. It
597  // manages the reader stack for us, and provides a lot of convenience
598  // methods to do specialized checking for chars, sequences of chars,
599  // skipping chars, etc...
600  //
601  // fScannerId
602  // fSequenceId
603  // These are used for progressive parsing, to make sure that the
604  // client code does the right thing at the right time.
605  //
606  // fStandalone
607  // Indicates whether the document is standalone or not. Defaults to
608  // no, but can be overridden in the XMLDecl.
609  //
610  // fHasNoDTD
611  // Indicates the document has no DTD or has only an internal DTD subset
612  // which contains no parameter entity references.
613  //
614  // fValidate
615  // Indicates whether any validation should be done. This is defined
616  // by the existence of a Grammar together with fValScheme.
617  //
618  // fValidator
619  // The installed validator. We look at them via the abstract
620  // validator interface, and don't know what it actual is.
621  // Either point to user's installed validator, or fDTDValidator
622  // or fSchemaValidator.
623  //
624  // fValidatorFromUser
625  // This flag indicates whether the validator was installed from
626  // user. If false, then the validator was created by the Scanner.
627  //
628  // fValScheme
629  // This is the currently set validation scheme. It defaults to
630  // 'never', but can be set by the client.
631  //
632  // fErrorCount
633  // The number of errors we've encountered.
634  //
635  // fDoSchema
636  // This flag indicates whether the client code wants Schema to
637  // be processed or not.
638  //
639  // fSchemaFullChecking
640  // This flag indicates whether the client code wants full Schema
641  // constraint checking.
642  //
643  // fIdentityConstraintChecking
644  // This flag indicates whether the client code wants Identity
645  // Constraint checking, defaulted to true to maintain backward
646  // compatibility (to minimize supprise)
647  //
648  // fAttName
649  // fAttValue
650  // fCDataBuf
651  // fNameBuf
652  // fQNameBuf
653  // fPrefixBuf
654  // For the most part, buffers are obtained from the fBufMgr object
655  // on the fly. However, for the start tag scan, we have a set of
656  // fixed buffers for performance reasons. These are used a lot and
657  // there are a number of them, so asking the buffer manager each
658  // time for new buffers is a bit too much overhead.
659  //
660  // fEmptyNamespaceId
661  // This is the id of the empty namespace URI. This is a special one
662  // because of the xmlns="" type of deal. We have to quickly sense
663  // that its the empty namespace.
664  //
665  // fUnknownNamespaceId
666  // This is the id of the namespace URI which is assigned to the
667  // global namespace. Its for debug purposes only, since there is no
668  // real global namespace URI. Its set by the derived class.
669  //
670  // fXMLNamespaceId
671  // fXMLNSNamespaceId
672  // These are the ids of the namespace URIs which are assigned to the
673  // 'xml' and 'xmlns' special prefixes. The former is officially
674  // defined but the latter is not, so we just provide one for debug
675  // purposes.
676  //
677  // fSchemaNamespaceId
678  // This is the id of the schema namespace URI.
679  //
680  // fGrammarResolver
681  // Grammar Pool that stores all the grammars. Key is namespace for
682  // schema and system id for external DTD. When caching a grammar, if
683  // a grammar is already in the pool, it will be replaced with the
684  // new parsed one.
685  //
686  // fGrammar
687  // Current Grammar used by the Scanner and Validator
688  //
689  // fRootGrammar
690  // The grammar where the root element is declared.
691  //
692  // fGrammarType
693  // Current Grammar Type. Store this value instead of calling getGrammarType
694  // all the time for faster performance.
695  //
696  // fURIStringPool
697  // This is a pool for URIs with unique ids assigned. We use a standard
698  // string pool class. This pool is going to be shared by all Grammar.
699  // Use only if namespace is turned on.
700  //
701  // fRootElemName
702  // No matter we are using DTD or Schema Grammar, if a DOCTYPE exists,
703  // we need to verify the root element name. So store the rootElement
704  // that is used in the DOCTYPE in the Scanner instead of in the DTDGrammar
705  // where it used to
706  //
707  // fExternalSchemaLocation
708  // The list of Namespace/SchemaLocation that was specified externally
709  // using setExternalSchemaLocation.
710  //
711  // fExternalNoNamespaceSchemaLocation
712  // The no target namespace XML Schema Location that was specified
713  // externally using setExternalNoNamespaceSchemaLocation.
714  //
715  // fSecurityManager
716  // The SecurityManager instance; as and when set by the application.
717  //
718  // fEntityExpansionLimit
