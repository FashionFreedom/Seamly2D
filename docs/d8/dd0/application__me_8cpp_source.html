92 inline void noisyFailureMsgHandler(QtMsgType type, const QMessageLogContext &context, const QString &msg)
93 {
94  Q_UNUSED(context)
95 
96  // Why on earth didn't Qt want to make failed signal/slot connections qWarning?
97  if ((type == QtDebugMsg) && msg.contains(QStringLiteral("::connect")))
98  {
99  type = QtWarningMsg;
100  }
101 
102 #if defined(V_NO_ASSERT)
103  // I have decided to hide this annoing message for release builds.
104  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QSslSocket: cannot resolve")))
105  {
106  type = QtDebugMsg;
107  }
108 
109  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("setGeometry: Unable to set geometry")))
110  {
111  type = QtDebugMsg;
112  }
113 #endif //defined(V_NO_ASSERT)
114 
115 #if defined(Q_OS_MAC)
116  // Hide Qt bug 'Assertion when reading an icns file'
117  // https://bugreports.qt.io/browse/QTBUG-45537
118  // Remove after Qt fix will be released
119  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QICNSHandler::read()")))
120  {
121  type = QtDebugMsg;
122  }
123 
124  // See issue #568
125  if (msg.contains(QStringLiteral("Error receiving trust for a CA certificate")))
126  {
127  type = QtDebugMsg;
128  }
129 #endif
130 
131  // this is another one that doesn't make sense as just a debug message. pretty serious
132  // sign of a problem
133  // http://www.developer.nokia.com/Community/Wiki/QPainter::begin:Paint_device_returned_engine_%3D%3D_0_(Known_Issue)
134  if ((type == QtDebugMsg) && msg.contains(QStringLiteral("QPainter::begin"))
135  && msg.contains(QStringLiteral("Paint device returned engine")))
136  {
137  type = QtWarningMsg;
138  }
139 
140  // This qWarning about "Cowardly refusing to send clipboard message to hung application..."
141  // is something that can easily happen if you are debugging and the application is paused.
142  // As it is so common, not worth popping up a dialog.
143  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QClipboard::event"))
144  && msg.contains(QStringLiteral("Cowardly refusing")))
145  {
146  type = QtDebugMsg;
147  }
148 
149  // only the GUI thread should display message boxes. If you are
150  // writing a multithreaded application and the error happens on
151  // a non-GUI thread, you'll have to queue the message to the GUI
152  QCoreApplication *instance = QCoreApplication::instance();
153  const bool isGuiThread = instance && (QThread::currentThread() == instance->thread());
154 
155  switch (type)
156  {
157  case QtDebugMsg:
158  vStdOut() << QApplication::translate("mNoisyHandler", "DEBUG:") << msg << "\n";
159  return;
160  case QtWarningMsg:
161  vStdErr() << QApplication::translate("mNoisyHandler", "WARNING:") << msg << "\n";
162  break;
163  case QtCriticalMsg:
164  vStdErr() << QApplication::translate("mNoisyHandler", "CRITICAL:") << msg << "\n";
165  break;
166  case QtFatalMsg:
167  vStdErr() << QApplication::translate("mNoisyHandler", "FATAL:") << msg << "\n";
168  break;
169  #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
170  case QtInfoMsg:
171  vStdOut() << QApplication::translate("mNoisyHandler", "INFO:") << msg << "\n";
172  break;
173  #endif
174  default:
175  break;
176  }
177 
178  if (isGuiThread)
179  {
180  //fixme: trying to make sure there are no save/load dialogs are opened, because error message during them will
181  //lead to crash
182  const bool topWinAllowsPop = (QApplication::activeModalWidget() == nullptr) ||
183  !QApplication::activeModalWidget()->inherits("QFileDialog");
184  QMessageBox messageBox;
185 
186  switch (type)
187  {
188  case QtWarningMsg:
189  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Warning"));
190  messageBox.setIcon(QMessageBox::Warning);
191  break;
192  case QtCriticalMsg:
193  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Critical Error"));
194  messageBox.setIcon(QMessageBox::Critical);
195  break;
196  case QtFatalMsg:
197  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Fatal Error"));
198  messageBox.setIcon(QMessageBox::Critical);
199  break;
200  #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
201  case QtInfoMsg:
202  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Information"));
203  messageBox.setIcon(QMessageBox::Information);
204  break;
205  #endif
206  case QtDebugMsg:
207  Q_UNREACHABLE(); //-V501
208  break;
209  default:
210  break;
211  }
212 
213  if (type == QtWarningMsg || type == QtCriticalMsg || type == QtFatalMsg)
214  {
215  if (!qApp->isTestMode())
216  {
217  if (topWinAllowsPop)
218  {
219  messageBox.setText(msg);
220  messageBox.setStandardButtons(QMessageBox::Ok);
221  messageBox.setWindowModality(Qt::ApplicationModal);
222  messageBox.setModal(true);
223  #ifndef QT_NO_CURSOR
224  QGuiApplication::setOverrideCursor(Qt::ArrowCursor);
225  #endif
226  messageBox.setWindowFlags(messageBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
227  messageBox.exec();
228  #ifndef QT_NO_CURSOR
229  QGuiApplication::restoreOverrideCursor();
230  #endif
231  }
232  }
233  }
234 
235  if (QtFatalMsg == type)
236  {
237  abort();
238  }
239  }
240  else
241  {
242  if (type != QtDebugMsg && type != QtWarningMsg)
243  {
244  abort(); // be NOISY unless overridden!
245  }
246  }
247 }

292 /**
293  * @brief notify Reimplemented from QApplication::notify().
294  * @param receiver receiver.
295  * @param event event.
296  * @return value that is returned from the receiver's event handler.
297  */
298 // reimplemented from QApplication so we can throw exceptions in slots
299 bool ApplicationME::notify(QObject *receiver, QEvent *event)
300 {
301  try
302  {
303  return QApplication::notify(receiver, event);
304  }
305  catch (const VExceptionObjectError &error)
306  {
307  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error parsing file. Program will be terminated.")), //-V807
308  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
309  exit(V_EX_DATAERR);
310  }
311  catch (const VExceptionBadId &error)
312  {
313  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error bad id. Program will be terminated.")),
314  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
315  exit(V_EX_DATAERR);
316  }
317  catch (const VExceptionConversionError &error)
318  {
319  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error can't convert value. Program will be terminated.")),
320  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
321  exit(V_EX_DATAERR);
322  }
323  catch (const VExceptionEmptyParameter &error)
324  {
325  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error empty parameter. Program will be terminated.")),
326  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
327  exit(V_EX_DATAERR);
328  }
329  catch (const VExceptionWrongId &error)
330  {
331  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error wrong id. Program will be terminated.")),
332  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
333  exit(V_EX_DATAERR);
334  }
335  catch (const VExceptionToolWasDeleted &error)
336  {
337  qCCritical(mApp, "%s\n\n%s\n\n%s",
338  qUtf8Printable("Unhadled deleting tool. Continue use object after deleting!"),
339  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
340  exit(V_EX_DATAERR);
341  }
342  catch (const VException &error)
343  {
344  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Something's wrong!!")),
345  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
346  return true;
347  }
348  // These last two cases special. I found that we can't show here modal dialog with error message.
349  // Somehow program doesn't waite untile an error dialog will be closed. But if ignore this program will hang.
350  catch (const qmu::QmuParserError &error)
351  {
352  qCCritical(mApp, "%s", qUtf8Printable(tr("Parser error: %1. Program will be terminated.").arg(error.GetMsg())));
353  exit(V_EX_DATAERR);
354  }
355  catch (std::exception &error)
356  {
357  qCCritical(mApp, "%s", qUtf8Printable(tr("Exception thrown: %1. Program will be terminated.").arg(error.what())));
358  exit(V_EX_SOFTWARE);
359  }
360  return false;
361 }

453 bool ApplicationME::event(QEvent *event)
454 {
455  switch(event->type())
456  {
457  // In Mac OS X the QFileOpenEvent event is generated when user perform "Open With" from Finder (this event is
458  // Mac specific).
459  case QEvent::FileOpen:
460  {
461  QFileOpenEvent *fileOpenEvent = static_cast<QFileOpenEvent *>(event);
462  const QString macFileOpen = fileOpenEvent->file();
463  if(!macFileOpen.isEmpty())
464  {
465  TMainWindow *mw = mainWindow();
466  if (mw)
467  {
468  mw->LoadFile(macFileOpen); // open file in existing window
469  }
470  return true;
471  }
472  break;
473  }
474 #if defined(Q_OS_MAC)
475  case QEvent::ApplicationActivate:
476  {
477  clean();
478  TMainWindow *mw = mainWindow();
479  if (mw && not mw->isMinimized())
480  {
481  mw->show();
482  }
483  return true;
484  }
485 #endif //defined(Q_OS_MAC)
486  default:
487  return VAbstractApplication::event(event);
488  }
489  return VAbstractApplication::event(event);
490 }

541 //---------------------------------------------------------------------------------------------------------------------
542 void ApplicationME::parseCommandLine(const SocketConnection &connection, const QStringList &arguments)
543 {
544  QCommandLineParser parser;
545  parser.setApplicationDescription(tr("Seamly2D's measurements editor."));
546  parser.addHelpOption();
547  parser.addVersionOption();
548  parser.addPositionalArgument("filename", tr("The measurement file."));
549  //-----
550  QCommandLineOption heightOption(QStringList() << "e" << "height",
551  tr("Open with the base height. Valid values: %1cm.")
552  .arg(MeasurementVariable::WholeListHeights(Unit::Cm).join(", ")),
553  tr("The base height"));
554  parser.addOption(heightOption);
555  //-----
556  QCommandLineOption sizeOption(QStringList() << "s" << "size",
557  tr("Open with the base size. Valid values: %1cm.").arg(MeasurementVariable::WholeListSizes(Unit::Cm).join(", ")),
558  tr("The base size"));
559  parser.addOption(sizeOption);
560  //-----
561  QCommandLineOption unitOption(QStringList() << "u" << "unit",
562  tr("Set pattern file unit: cm, mm, inch."),
563  tr("The pattern unit"));
564  parser.addOption(unitOption);
565  //-----
566  QCommandLineOption testOption(QStringList() << "test",
567  tr("Use for unit testing. Run the program and open a file without showing the main window."));
568  parser.addOption(testOption);
569  //-----
570  QCommandLineOption scalingOption(QStringList() << LONG_OPTION_NO_HDPI_SCALING,
571  tr("Disable high dpi scaling. Call this option if has problem with scaling (by default scaling enabled). "
572  "Alternatively you can use the %1 environment variable.").arg("QT_AUTO_SCREEN_SCALE_FACTOR=0"));
573  parser.addOption(scalingOption);
574  //-----
575  parser.process(arguments);
576 
577  bool flagHeight = false;
578  bool flagSize = false;
579  bool flagUnit = false;
580 
581  int size = 0;
582  int height = 0;
583  Unit unit = Unit::Cm;
584 
585  if (parser.isSet(heightOption))
586  {
587  const QString heightValue = parser.value(heightOption);
588  if (MeasurementVariable::IsGradationHeightValid(heightValue))
589  {
590  flagHeight = true;
591  height = heightValue.toInt();
592  }
593  else
594  {
595  qCCritical(mApp, "%s\n",
596  qPrintable(tr("Invalid base height argument. Must be %1cm.")
597  .arg(MeasurementVariable::WholeListHeights(Unit::Cm).join(", "))));
598  parser.showHelp(V_EX_USAGE);
599  }
600  }
601 
602  if (parser.isSet(sizeOption))
603  {
604  const QString sizeValue = parser.value(sizeOption);
605  if (MeasurementVariable::IsGradationSizeValid(sizeValue))
606  {
607  flagSize = true;
608  size = sizeValue.toInt();
609  }
610  else
611  {
612  qCCritical(mApp, "%s\n",
613  qPrintable(tr("Invalid base size argument. Must be %1cm.")
614  .arg(MeasurementVariable::WholeListSizes(Unit::Cm).join(", "))));
615  parser.showHelp(V_EX_USAGE);
616  }
617  }
618 
619  {
620  const QString unitValue = parser.value(unitOption);
621  if (!unitValue.isEmpty())
622  {
623 
624  const QStringList units = QStringList() << unitMM << unitCM << unitINCH;
625  if (units.contains(unitValue))
626  {
627  flagUnit = true;
628  unit = StrToUnits(unitValue);
629  }
630  else
631  {
632  qCCritical(mApp, "%s\n", qPrintable(tr("Invalid base size argument. Must be cm, mm or inch.")));
633  parser.showHelp(V_EX_USAGE);
634  }
635  }
636  }
637 
638  m_testMode = parser.isSet(testOption);
639 
640  if (!m_testMode && connection == SocketConnection::Client)
641  {
642  const QString serverName = QCoreApplication::applicationName();
643  QLocalSocket socket;
644  socket.connectToServer(serverName);
645  if (socket.waitForConnected(1000))
646  {
647  qCInfo(mApp, "Connected to the server '%s'", qUtf8Printable(serverName));
648  QTextStream stream(&socket);
649  stream << QCoreApplication::arguments().join(";;");
650  stream.flush();
651  socket.waitForBytesWritten();
652  qApp->exit(V_EX_OK);
653  return;
654  }
655 
656  qCDebug(mApp, "Can't establish connection to the server '%s'", qUtf8Printable(serverName));
657 
658  m_localServer = new QLocalServer(this);
659  connect(m_localServer, &QLocalServer::newConnection, this, &ApplicationME::newLocalSocketConnection);
660  if (!m_localServer->listen(serverName))
661  {
662  qCWarning(mApp, "Can't begin to listen for incoming connections on name '%s'",
663  qUtf8Printable(serverName));
664  if (m_localServer->serverError() == QAbstractSocket::AddressInUseError)
665  {
666  QLocalServer::removeServer(serverName);
667  if (!m_localServer->listen(serverName))
