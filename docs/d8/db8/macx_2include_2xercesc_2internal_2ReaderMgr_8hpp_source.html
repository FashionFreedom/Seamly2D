39 // ---------------------------------------------------------------------------
40 // This class is used by the scanner. The scanner must deal with expansion
41 // of entities, some of which are totally different files (external parsed
42 // entities.) It does so by pushing readers onto a stack. The top reader is
43 // the one it wants to read out of, but that one must be popped when it is
44 // empty. To keep that logic from being all over the place, the scanner
45 // talks to the reader manager, which handles the stack and popping off
46 // used up readers.
47 // ---------------------------------------------------------------------------

211  // -----------------------------------------------------------------------
212  // Private data members
213  //
214  // fCurEntity
215  // This is the current top of stack entity. We pull it off the stack
216  // and store it here for efficiency.
217  //
218  // fCurReader
219  // This is the current top of stack reader. We pull it off the
220  // stack and store it here for efficiency.
221  //
222  // fEntityHandler
223  // This is the installed entity handler. Its installed via the
224  // scanner but he passes it on to us since we need it the most, in
225  // process of creating external entity readers.
226  //
227  // fEntityStack
228  // We need to keep up with which of the pushed readers are pushed
229  // entity values that are being spooled. This is done to avoid the
230  // problem of recursive definitions. This stack consists of refs to
231  // EntityDecl objects for the pushed entities.
232  //
233  // fNextReaderNum
234  // This is the reader serial number value. Each new reader that is
235  // created from this reader is given a successive number. This lets
236  // us catch things like partial markup errors and such.
237  //
238  // fReaderStack
239  // This is the stack of reader references. We own all the readers
240  // and destroy them when they are used up.
241  //
242  // fThrowEOE
243  // This flag controls whether we throw an exception when we hit an
244  // end of entity. The scanner doesn't really need to know about ends
245  // of entities in the int/ext subsets, so it will turn this flag off
246  // until it gets into the content usually.
247  //
248  // fXMLVersion
249  // Enum to indicate if each Reader should be created as XML 1.1 or
250  // XML 1.0 conformant
251  //
252  // fStandardUriConformant
253  // This flag controls whether we force conformant URI
254  // -----------------------------------------------------------------------
255  XMLEntityDecl* fCurEntity;
256  XMLReader* fCurReader;
257  XMLEntityHandler* fEntityHandler;
258  RefStackOf<XMLEntityDecl>* fEntityStack;
259  unsigned int fNextReaderNum;
260  RefStackOf<XMLReader>* fReaderStack;
261  bool fThrowEOE;
262  XMLReader::XMLVersion fXMLVersion;
263  bool fStandardUriConformant;
264  MemoryManager* fMemoryManager;
265 };

361 {
362  XMLCh nextCh = 0;
363  do

365  // Get chars until we find the one to skip
366  nextCh = getNextChar();

369  while((nextCh != toSkipTo) && nextCh!=0);

400 // This is a simple class to temporarily change the 'throw at end of entity'

402 // turn this on and off on a scoped basis.

404 class XMLPARSER_EXPORT ThrowEOEJanitor
405 {
406 public :

408  // Constructors and destructor

411 
412  fOld(mgrTarget->getThrowEOE())
413  , fMgr(mgrTarget)
414  {
415  mgrTarget->setThrowEOE(newValue);
416  }
417 
418  ~ThrowEOEJanitor()

420  fMgr->setThrowEOE(fOld);
421  };

424  // -----------------------------------------------------------------------

438  // A pointer to the reader manager we are going to set/reset the
