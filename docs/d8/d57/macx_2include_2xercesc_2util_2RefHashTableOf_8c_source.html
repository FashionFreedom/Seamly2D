146 template <class TVal, class THasher>
147 void RefHashTableOf<TVal, THasher>::
148 removeKey(const void* const key)
149 {
150  // Hash the key
151  XMLSize_t hashVal = fHasher.getHashVal(key, fHashModulus);
152 
153  //
154  // Search the given bucket for this key. Keep up with the previous
155  // element so we can patch around it.
156  //
157  RefHashTableBucketElem<TVal>* curElem = fBucketList[hashVal];
158  RefHashTableBucketElem<TVal>* lastElem = 0;
159 
160  while (curElem)
161  {
162  if (fHasher.equals(key, curElem->fKey))
163  {
164  if (!lastElem)
165  {
166  // It was the first in the bucket
167  fBucketList[hashVal] = curElem->fNext;
168  }
169  else
170  {
171  // Patch around the current element
172  lastElem->fNext = curElem->fNext;
173  }
174 
175  // If we adopted the data, then delete it too
176  // (Note: the userdata hash table instance has data type of void *.
177  // This will generate compiler warnings here on some platforms, but they
178  // can be ignored since fAdoptedElements is false.
179  if (fAdoptedElems)
180  delete curElem->fData;
181 
182  // Then delete the current element and move forward
183  // delete curElem;
184  // destructor doesn't do anything...
185  fMemoryManager->deallocate(curElem);
186 
187  fCount--;
188 
189  return;
190  }
191 
192  // Move both pointers upwards
193  lastElem = curElem;
194  curElem = curElem->fNext;
195  }
196 
197  // We never found that key
198  ThrowXMLwithMemMgr(NoSuchElementException, XMLExcepts::HshTbl_NoSuchKeyExists, fMemoryManager);
199 }

201 template <class TVal, class THasher>
202 void RefHashTableOf<TVal, THasher>::removeAll()
203 {
204  if(isEmpty())
205  return;
206 
207  // Clean up the buckets first
208  for (XMLSize_t buckInd = 0; buckInd < fHashModulus; buckInd++)
209  {
210  // Get the bucket list head for this entry
211  RefHashTableBucketElem<TVal>* curElem = fBucketList[buckInd];
212  RefHashTableBucketElem<TVal>* nextElem;
213  while (curElem)
214  {
215  // Save the next element before we hose this one
216  nextElem = curElem->fNext;
217 
218  // If we adopted the data, then delete it too
219  // (Note: the userdata hash table instance has data type of void *.
220  // This will generate compiler warnings here on some platforms, but they
221  // can be ignored since fAdoptedElements is false.
222  if (fAdoptedElems)
223  delete curElem->fData;
224 
225  // Then delete the current element and move forward
226  // delete curElem;
227  // destructor doesn't do anything...
228  // curElem->~RefHashTableBucketElem();
229  fMemoryManager->deallocate(curElem);
230  curElem = nextElem;
231  }
232 
233  // Clean out this entry
234  fBucketList[buckInd] = 0;
235  }
236 
237  fCount = 0;
238 }

252  // Search the given bucket for this key. Keep up with the previous

254  //
255  RefHashTableBucketElem<TVal>* curElem = fBucketList[hashVal];
256  RefHashTableBucketElem<TVal>* lastElem = 0;
257 
258  while (curElem)

261  {
262  if (!lastElem)

264  // It was the first in the bucket
265  fBucketList[hashVal] = curElem->fNext;
266  }

270  lastElem->fNext = curElem->fNext;
271  }

276  // delete curElem;

394 template <class TVal, class THasher>
395 void RefHashTableOf<TVal, THasher>::put(void* key, TVal* const valueToAdopt)
396 {
397  // Apply 0.75 load factor to find threshold.
398  XMLSize_t threshold = fHashModulus * 3 / 4;
399 
400  // If we've grown too big, expand the table and rehash.
401  if (fCount >= threshold)
402  rehash();
403 
404  // First see if the key exists already
405  XMLSize_t hashVal;
406  RefHashTableBucketElem<TVal>* newBucket = findBucketElem(key, hashVal);
407 
408  //
409  // If so,then update its value. If not, then we need to add it to
410  // the right bucket
411  //
412  if (newBucket)
413  {
414  if (fAdoptedElems)
415  delete newBucket->fData;
416  newBucket->fData = valueToAdopt;
417  newBucket->fKey = key;
418  }
419  else
420  {
421  newBucket =
422  new (fMemoryManager->allocate(sizeof(RefHashTableBucketElem<TVal>)))
423  RefHashTableBucketElem<TVal>(key, valueToAdopt, fBucketList[hashVal]);
424  fBucketList[hashVal] = newBucket;
425  fCount++;
426  }
427 }

434 template <class TVal, class THasher>
435 void RefHashTableOf<TVal, THasher>::rehash()
436 {
437  const XMLSize_t newMod = (fHashModulus * 2) + 1;
438 
439  RefHashTableBucketElem<TVal>** newBucketList =
440  (RefHashTableBucketElem<TVal>**) fMemoryManager->allocate
441  (
442  newMod * sizeof(RefHashTableBucketElem<TVal>*)
443  );
444 
445  // Make sure the new bucket list is destroyed if an
446  // exception is thrown.
447  ArrayJanitor<RefHashTableBucketElem<TVal>*> guard(newBucketList, fMemoryManager);
448 
449  memset(newBucketList, 0, newMod * sizeof(newBucketList[0]));
450 
451 
452  // Rehash all existing entries.
453  for (XMLSize_t index = 0; index < fHashModulus; index++)
454  {
455  // Get the bucket list head for this entry
456  RefHashTableBucketElem<TVal>* curElem = fBucketList[index];
457 
458  while (curElem)
459  {
460  // Save the next element before we detach this one
461  RefHashTableBucketElem<TVal>* const nextElem = curElem->fNext;
462 
463  const XMLSize_t hashVal = fHasher.getHashVal(curElem->fKey, newMod);
464 
465  RefHashTableBucketElem<TVal>* const newHeadElem = newBucketList[hashVal];
466 
467  // Insert at the start of this bucket's list.
468  curElem->fNext = newHeadElem;
469  newBucketList[hashVal] = curElem;
470 
471  curElem = nextElem;
472  }
473  }
474 
475  RefHashTableBucketElem<TVal>** const oldBucketList = fBucketList;
476 
477  // Everything is OK at this point, so update the
478  // member variables.
479  fBucketList = guard.release();
480  fHashModulus = newMod;
481 
482  // Delete the old bucket list.
483  fMemoryManager->deallocate(oldBucketList);//delete[] oldBucketList;
484 
485 }

630 template <class TVal, class THasher>
631 void RefHashTableOfEnumerator<TVal, THasher>::findNext()
632 {
633  //
634  // If there is a current element, move to its next element. If this
635  // hits the end of the bucket, the next block will handle the rest.
636  //
637  if (fCurElem)
638  fCurElem = fCurElem->fNext;
639 
640  //
641  // If the current element is null, then we have to move up to the
642  // next hash value. If that is the hash modulus, then we cannot
643  // go further.
644  //
645  if (!fCurElem)
646  {
647  fCurHash++;
648  if (fCurHash == fToEnum->fHashModulus)
649  return;
650 
651  // Else find the next non-empty bucket
652  while (fToEnum->fBucketList[fCurHash]==0)
653  {
654  // Bump to the next hash value. If we max out return
655  fCurHash++;
656  if (fCurHash == fToEnum->fHashModulus)
657  return;
658  }
659  fCurElem = fToEnum->fBucketList[fCurHash];
660  }
