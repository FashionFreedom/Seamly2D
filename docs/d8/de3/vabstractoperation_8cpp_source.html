237 void VAbstractOperation::ExtractData(const QDomElement &domElement, QVector<SourceItem> &source,
238  QVector<DestinationItem> &destination)
239 {
240  const QDomNodeList nodeList = domElement.childNodes();
241  for (qint32 i = 0; i < nodeList.size(); ++i)
242  {
243  const QDomElement dataElement = nodeList.at(i).toElement();
244  if (not dataElement.isNull() && dataElement.tagName() == TagSource)
245  {
246  source.clear();
247  const QDomNodeList srcList = dataElement.childNodes();
248  for (qint32 j = 0; j < srcList.size(); ++j)
249  {
250  const QDomElement element = srcList.at(j).toElement();
251  if (not element.isNull())
252  {
253  SourceItem item;
254  item.id = VDomDocument::GetParametrUInt(element, AttrIdObject, NULL_ID_STR);
255  item.alias = VDomDocument::GetParametrEmptyString(element, AttrAlias);
256  item.lineType = VDomDocument::GetParametrString(element, AttrLineType, LineTypeSolidLine);
257  item.color = VDomDocument::GetParametrString(element, AttrColor, "black");
258  source.append(item);
259  }
260  }
261  }
262 
263  if (not dataElement.isNull() && dataElement.tagName() == TagDestination)
264  {
265  destination.clear();
266  const QDomNodeList srcList = dataElement.childNodes();
267  for (qint32 j = 0; j < srcList.size(); ++j)
268  {
269  const QDomElement element = srcList.at(j).toElement();
270  if (not element.isNull())
271  {
272  DestinationItem d;
273  d.id = VDomDocument::GetParametrUInt(element, AttrIdObject, NULL_ID_STR);
274  d.mx = qApp->toPixel(VDomDocument::GetParametrDouble(element, AttrMx, QString::number(INT_MAX)));
275  d.my = qApp->toPixel(VDomDocument::GetParametrDouble(element, AttrMy, QString::number(INT_MAX)));
276  d.showPointName = VDomDocument::getParameterBool(element, AttrShowPointName, trueStr);
277  destination.append(d);
278  }
279  }
280  }
281  }
282 }

588 void VAbstractOperation::SaveSourceDestination(QDomElement &tag)
589 {
590  doc->RemoveAllChildren(tag);
591 
592  QDomElement tagObjects = doc->createElement(TagSource);
593  for (auto sourceItem : qAsConst(source))
594  {
595  QDomElement item = doc->createElement(TagItem);
596  doc->SetAttribute(item, AttrIdObject, sourceItem.id);
597  doc->SetAttribute(item, AttrAlias, sourceItem.alias);
598  doc->SetAttribute(item, AttrLineType, sourceItem.lineType);
599  doc->SetAttribute(item, AttrColor, sourceItem.color);
600  tagObjects.appendChild(item);
601  }
602  tag.appendChild(tagObjects);
603 
604  tagObjects = doc->createElement(TagDestination);
605  for (auto destinationItem : qAsConst(destination))
606  {
607  QDomElement item = doc->createElement(TagItem);
608  doc->SetAttribute(item, AttrIdObject, destinationItem.id);
609 
610  if (not VFuzzyComparePossibleNulls(destinationItem.mx, INT_MAX) &&
611  not VFuzzyComparePossibleNulls(destinationItem.my, INT_MAX))
612  {
613  doc->SetAttribute(item, AttrMx, qApp->fromPixel(destinationItem.mx));
614  doc->SetAttribute(item, AttrMy, qApp->fromPixel(destinationItem.my));
615  doc->SetAttribute<bool>(item, AttrShowPointName, destinationItem.showPointName);
616  }
617 
618  tagObjects.appendChild(item);
619  }
620  tag.appendChild(tagObjects);
621 }

683 //---------------------------------------------------------------------------------------------------------------------
684 void VAbstractOperation::InitOperatedObjects()
685 {
686  for (int i = 0; i < destination.size(); ++i)
687  {
688  const DestinationItem item = destination.at(i);
689  const QSharedPointer<VGObject> object = VAbstractTool::data.GetGObject(item.id);
690 
691  // This check helps to find missed objects in the switch
692  Q_STATIC_ASSERT_X(static_cast<int>(GOType::Unknown) == 7, "Not all objects were handled.");
693 
694 QT_WARNING_PUSH
695 QT_WARNING_DISABLE_GCC("-Wswitch-default")
696  switch(static_cast<GOType>(object->getType()))
697  {
698  case GOType::Point:
699  {
700  VSimplePoint *point = new VSimplePoint(item.id, QColor(Qt::black));
701  point->setParentItem(this);
702  point->SetType(GOType::Point);
703  point->setToolTip(complexPointToolTip(item.id));
704  connect(point, &VSimplePoint::Choosed, this, [this](quint32 id)
705  {
706  emit chosenTool(id, SceneObject::Point);
707  });
