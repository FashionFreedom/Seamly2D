161 /**
162  * @brief Create help create tool.
163  * @param _id tool id, 0 if tool doesn't exist yet.
164  * @param firstPoint id first line point.
165  * @param secondPoint id second line point.
166  * @param lineType line type.
167  * @param lineWeight line weight.
168  * @param lineColor line color.
169  * @param scene pointer to scene.
170  * @param doc dom document container.
171  * @param data container with variables.
172  * @param parse parser file mode.
173  * @param typeCreation way we create this tool.
174  */
175 VToolLine * VToolLine::Create(const quint32 &_id, const quint32 &firstPoint, const quint32 &secondPoint,
176  const QString &lineType, const QString &lineWeight, const QString &lineColor,
177  VMainGraphicsScene *scene, VAbstractPattern *doc, VContainer *data,
178  const Document &parse, const Source &typeCreation)
179 {
180  SCASSERT(scene != nullptr)
181  SCASSERT(doc != nullptr)
182  SCASSERT(data != nullptr)
183  quint32 id = _id;
184  if (typeCreation == Source::FromGui)
185  {
186  id = VContainer::getNextId();
187  data->AddLine(firstPoint, secondPoint);
188  }
189  else
190  {
191  VContainer::UpdateId(id);
192  data->AddLine(firstPoint, secondPoint);
193  if (parse != Document::FullParse)
194  {
195  doc->UpdateToolData(id, data);
196  }
197  }
198 
199  if (parse == Document::FullParse)
200  {
201  VDrawTool::AddRecord(id, Tool::Line, doc);
202  VToolLine *line = new VToolLine(doc, data, id, firstPoint, secondPoint, lineType, lineWeight, lineColor, typeCreation);
203  scene->addItem(line);
204  InitDrawToolConnections(scene, line);
205  connect(scene, &VMainGraphicsScene::EnableLineItemSelection, line, &VToolLine::AllowSelecting);
206  connect(scene, &VMainGraphicsScene::EnableLineItemHover, line, &VToolLine::AllowHover);
207  VAbstractPattern::AddTool(id, line);
208 
209  const QSharedPointer<VPointF> first = data->GeometricObject<VPointF>(firstPoint);
210  const QSharedPointer<VPointF> second = data->GeometricObject<VPointF>(secondPoint);
211 
212  doc->IncrementReferens(first->getIdTool());
213  doc->IncrementReferens(second->getIdTool());
214  return line;
215  }
216  return nullptr;
217 }
