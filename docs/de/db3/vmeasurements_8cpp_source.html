279 void VMeasurements::ReadMeasurements() const
280 {
281  // For conversion values we must first calculate all data in measurement file's unit.
282  // That's why we need two containers: one for converted values, second for real data.
283 
284  // Container for values in measurement file's unit
285  QScopedPointer<VContainer> tempData(new VContainer(data->GetTrVars(), data->GetPatternUnit()));
286 
287  const QDomNodeList list = elementsByTagName(TagMeasurement);
288  for (int i=0; i < list.size(); ++i)
289  {
290  const QDomElement dom = list.at(i).toElement();
291 
292  const QString name = GetParametrString(dom, AttrName);
293 
294  QString description;
295  try
296  {
297  description = GetParametrString(dom, AttrDescription);
298  }
299  catch (VExceptionEmptyParameter &e)
300  {
301  Q_UNUSED(e)
302  }
303 
304  QString fullName;
305  try
306  {
307  fullName = GetParametrString(dom, AttrFullName);
308  }
309  catch (VExceptionEmptyParameter &e)
310  {
311  Q_UNUSED(e)
312  }
313 
314  QSharedPointer<VMeasurement> meash;
315  QSharedPointer<VMeasurement> tempMeash;
316  if (type == MeasurementsType::Multisize)
317  {
318  qreal base = GetParametrDouble(dom, AttrBase, "0");
319  qreal ksize = GetParametrDouble(dom, AttrSizeIncrease, "0");
320  qreal kheight = GetParametrDouble(dom, AttrHeightIncrease, "0");
321 
322  tempMeash = QSharedPointer<VMeasurement>(new VMeasurement(static_cast<quint32>(i), name, BaseSize(),
323  BaseHeight(), base, ksize, kheight));
324  tempMeash->SetSize(m_currentSize);
325  tempMeash->SetHeight(m_currentHeight);
326  tempMeash->SetUnit(data->GetPatternUnit());
327 
328  base = UnitConvertor(base, MUnit(), *data->GetPatternUnit());
329  ksize = UnitConvertor(ksize, MUnit(), *data->GetPatternUnit());
330  kheight = UnitConvertor(kheight, MUnit(), *data->GetPatternUnit());
331 
332  const qreal baseSize = UnitConvertor(BaseSize(), MUnit(), *data->GetPatternUnit());
333  const qreal baseHeight = UnitConvertor(BaseHeight(), MUnit(), *data->GetPatternUnit());
334 
335  meash = QSharedPointer<VMeasurement>(new VMeasurement(static_cast<quint32>(i), name, baseSize, baseHeight,
336  base, ksize, kheight, fullName, description));
337  meash->SetSize(m_currentSize);
338  meash->SetHeight(m_currentHeight);
339  meash->SetUnit(data->GetPatternUnit());
340  }
341  else
342  {
343  const QString formula = GetParametrString(dom, AttrValue, "0");
344  bool ok = false;
345  qreal value = EvalFormula(tempData.data(), formula, &ok);
346 
347  tempMeash = QSharedPointer<VMeasurement>(new VMeasurement(tempData.data(), static_cast<quint32>(i), name,
348  value, formula, ok));
349 
350  value = UnitConvertor(value, MUnit(), *data->GetPatternUnit());
351  meash = QSharedPointer<VMeasurement>(new VMeasurement(data, static_cast<quint32>(i), name, value, formula,
352  ok, fullName, description));
353  }
354  tempData->AddVariable(name, tempMeash);
355  data->AddVariable(name, meash);
356  }
357 }

737 //---------------------------------------------------------------------------------------------------------------------
738 void VMeasurements::CreateEmptyMultisizeFile(Unit unit, int baseSize, int baseHeight)
739 {
740  this->clear();
741  QDomElement mElement = this->createElement(TagVST);
742 
743  mElement.appendChild(createComment(FileComment()));
744 
745  QDomElement version = createElement(TagVersion);
746  const QDomText newNodeText = createTextNode(VVSTConverter::MeasurementMaxVerStr);
747  version.appendChild(newNodeText);
748  mElement.appendChild(version);
749 
750  QDomElement ro = createElement(TagReadOnly);
751  const QDomText roNodeText = createTextNode("false");
752  ro.appendChild(roNodeText);
753  mElement.appendChild(ro);
754 
755  mElement.appendChild(createElement(TagNotes));
756 
757  QDomElement mUnit = createElement(TagUnit);
758  const QDomText unitText = createTextNode(UnitsToStr(unit));
759  mUnit.appendChild(unitText);
760  mElement.appendChild(mUnit);
761 
762  QDomElement system = createElement(TagPMSystem);
763  system.appendChild(createTextNode(ClearPMCode(p998_S)));
764  mElement.appendChild(system);
765 
766  QDomElement size = createElement(TagSize);
767  SetAttribute(size, AttrBase, QString().setNum(baseSize));
768  mElement.appendChild(size);
769 
770  QDomElement height = createElement(TagHeight);
771  SetAttribute(height, AttrBase, QString().setNum(baseHeight));
772  mElement.appendChild(height);
773 
774  mElement.appendChild(createElement(TagBodyMeasurements));
775 
776  this->appendChild(mElement);
777  insertBefore(createProcessingInstruction("xml", "version=\"1.0\" encoding=\"UTF-8\""), this->firstChild());
778 }
