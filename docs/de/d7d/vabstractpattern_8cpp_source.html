301 QStringList VAbstractPattern::ListMeasurements() const
302 {
303  QSet<QString> measurements;
304  QSet<QString> others;
305 
306  const QStringList variables = listVariables();
307  for (int i=0; i < variables.size(); ++i)
308  {
309  others.insert(variables.at(i));
310  }
311 
312  const QVector<VFormulaField> expressions = ListExpressions();
313  for (int i=0; i < expressions.size(); ++i)
314  {
315  // Eval formula
316  QScopedPointer<qmu::QmuTokenParser> cal(new qmu::QmuTokenParser(expressions.at(i).expression, false, false));
317  const QMap<int, QString> tokens = cal->GetTokens();// Tokens (variables, measurements)
318  delete cal.take();
319 
320  const QList<QString> tValues = tokens.values();
321  for (int j = 0; j < tValues.size(); ++j)
322  {
323  if (tValues.at(j) == QChar('-'))
324  {
325  continue;
326  }
327 
328  if (measurements.contains(tValues.at(j)))
329  {
330  continue;
331  }
332 
333  if (others.contains(tValues.at(j)))
334  {
335  continue;
336  }
337 
338  if (IsVariable(tValues.at(j)) || IsPostfixOperator(tValues.at(j)) || IsFunction(tValues.at(j)))
339  {
340  others.insert(tValues.at(j));
341  }
342  else
343  {
344  measurements.insert(tValues.at(j));
345  }
346  }
347  }
348 
349  return QStringList(measurements.values());
350 }

473 void VAbstractPattern::parseGroups(const QDomElement &domElement)
474 {
475  Q_ASSERT_X(not domElement.isNull(), Q_FUNC_INFO, "domElement is null");
476 
477  QMap<quint32, quint32> itemTool;
478  QMap<quint32, bool> itemVisibility;
479 
480  QDomNode domNode = domElement.firstChild();
481  while (domNode.isNull() == false)
482  {
483  if (domNode.isElement())
484  {
485  const QDomElement domElement = domNode.toElement();
486  if (domElement.isNull() == false)
487  {
488  if (domElement.tagName() == TagGroup)
489  {
490  VContainer::UpdateId(GetParametrUInt(domElement, AttrId, NULL_ID_STR));
491 
492  const QPair<bool, QMap<quint32, quint32> > groupData = parseItemElement(domElement);
493  const QMap<quint32, quint32> group = groupData.second;
494  auto i = group.constBegin();
495  while (i != group.constEnd())
496  {
497  if (not itemTool.contains(i.key()))
498  {
499  itemTool.insert(i.key(), i.value());
500  }
501 
502  const bool previous = itemVisibility.value(i.key(), false);
503  itemVisibility.insert(i.key(), previous || groupData.first);
504  ++i;
505  }
506  }
507  }
508  }
509  domNode = domNode.nextSibling();
510  }
511 
512  auto i = itemTool.constBegin();
513  while (i != itemTool.constEnd())
514  {
515  if (tools.contains(i.value()))
516  {
517  VDataTool* tool = tools.value(i.value());
518  tool->GroupVisibility(i.key(), itemVisibility.value(i.key(), true));
519  }
520  ++i;
521  }
522 }
