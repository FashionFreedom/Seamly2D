103 bool VCurveVariable::Filter(quint32 id)
104 {
105  if (id == NULL_ID)
106  {
107  return false;
108  }
109 
110  if (d->parentId != NULL_ID)//Do not check if value zero
111  {// Not all curves have parents. Only those who was created after cutting the parent curve.
112  return d->id == id || d->parentId == id;
113  }
114  else
115  {
116  return d->id == id;
117  }
118 }
