97  m_handleRect = m_boundingRect.adjusted(HANDLE_SIZE/2, HANDLE_SIZE/2, -HANDLE_SIZE/2, -HANDLE_SIZE/2);
98  m_origin = m_boundingRect.topLeft() + QPointF(m_image.xOrigin, m_image.yOrigin);
99 
100 
101  if (m_image.order == 0)
102  {
103  qint32 minZValue = maxImageZvalue+1;
104  foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
105  {
106  minZValue = qMin(minZValue, item->m_image.order);
107  }
108  m_image.order = minZValue-1;
109  moveToTop();
110  }
111 
112  updateImage();
113 
114  m_resizeHandles = new ResizeHandlesItem(this, m_minDimension, m_maxDimension);

208 void ImageItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
209 {
210  Q_UNUSED(option)
211  Q_UNUSED(widget)
212 
213  if (isSelected())
214  {
215  painter->save();
216  painter->setPen(QPen(Qt::black, 1, Qt::DashLine));
217  painter->drawRect(m_boundingRect);
218  painter->restore();
219  }
220 
221  if (!m_image.locked && m_isHovered)
222  {
223  painter->save();
224  QColor color = QColor(qApp->Settings()->getTertiarySupportColor());
225  color.setAlpha(20);
226  painter->setPen(QPen(Qt::black, 2, Qt::DashLine));
227  painter->setBrush(color);
228  painter->drawRect(m_boundingRect);
229  painter->restore();
230  }
231 
232  painter->setRenderHint(QPainter::SmoothPixmapTransform, (m_transformationMode == Qt::SmoothTransformation));
233  painter->drawPixmap(m_boundingRect.x(), m_boundingRect.y(), m_image.width, m_image.height, m_pixmap);
234 
235  if (m_origin != m_boundingRect.topLeft())
236  {
237  painter->save();
238  painter->setPen(QPen(Qt::blue, 2, Qt::SolidLine));
239  qreal x1 = qMax(m_origin.x() - 8, m_boundingRect.x());
240  qreal y1 = qMax(m_origin.y() - 8, m_boundingRect.y());
241  qreal x2 = qMin(m_origin.x() + 8, m_boundingRect.x() + m_boundingRect.width());
242  qreal y2 = qMin(m_origin.y() + 8, m_boundingRect.y() + m_boundingRect.height());
243  painter->drawLine(QLineF(x1, m_origin.y(), x2, m_origin.y()));
244  painter->drawLine(QLineF(m_origin.x(), y1, m_origin.x(), y2));
245  painter->restore();
246  }
247 }

462 void ImageItem::mousePressEvent(QGraphicsSceneMouseEvent *event)
463 {
464  if (!m_selectable)
465  {
466  event->ignore();
467  return;
468  }
469 
470  // Special for not selectable item first need to call standard mousePressEvent then accept event
471  //QGraphicsItem::mousePressEvent(event);
472 
473  // Somehow clicking on non selectable object does not clear previous selections.
474  if (not (flags() & ItemIsSelectable) && scene())
475  {
476  scene()->clearSelection();
477  }
478  m_mousePressed = true;
479 
480  if (flags() & QGraphicsItem::ItemIsMovable)
481  {
482  if (event->button() == Qt::LeftButton && event->type() != QEvent::GraphicsSceneMouseDoubleClick)
483  {
484  SetItemOverrideCursor(this, cursorArrowCloseHand, 1, 1);
485  m_offset = event->pos() - m_origin;
486  }
487  }
488  if (m_selectionType == SelectionType::ByMouseRelease)
489  {
490  event->accept(); // This help for non selectable items still receive mouseReleaseEvent events
491  }
492  else
493  {
494  if (event->button() == Qt::LeftButton && event->type() != QEvent::GraphicsSceneMouseDoubleClick)
495  {
496  emit imageSelected(m_image.id);
497  event->accept();
498  }
499  }
500 
501  if(m_selectNewOrigin)
502  {
503  m_image.xOrigin = event->pos().x() - m_boundingRect.topLeft().x();
504  m_image.yOrigin = event->pos().y() - m_boundingRect.topLeft().y();
505 
506  m_origin = m_boundingRect.topLeft() + QPointF(m_image.xOrigin, m_image.yOrigin);
507 
508  m_image.xPos = mapToScene(event->pos()).x();
509  m_image.yPos = mapToScene(event->pos()).y();
510  }
511 }

598 void ImageItem::updateFromHandles(QRectF rect)
599 {
600  prepareGeometryChange();
601 
602  //The image origin is moved so that it stays at the same place on the image
603  //The image origin is different from the QGraphicsItem origin, see below
604  m_image.xOrigin = m_image.xOrigin / m_image.width * rect.width();
605  m_image.yOrigin = m_image.yOrigin / m_image.height * rect.height();
606 
607  //m_image.xOrigin / m_image.yOrigin is the distance between the top left corner of the image and the image origin
608  //m_origin is the distance between the real origin of the QGraphicsItem and the image origin
609  m_origin = rect.topLeft() + QPointF(m_image.xOrigin, m_image.yOrigin);
610 
611  m_image.xPos = mapToScene(m_origin).x();
612  m_image.yPos = mapToScene(m_origin).y();
613  m_image.width = rect.width();
614  m_image.height = rect.height();
615 
616  m_boundingRect.setTopLeft(rect.topLeft());
617 
618  updateImage();
619  scene()->update();
620 }

642 void ImageItem::moveToBottom()
643 {
644  qint32 minZValue = m_image.order;
645  foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
646  {
647  if (item != this && item->m_image.order < m_image.order)
648  {
649  minZValue = qMin(minZValue, item->m_image.order);
650  item->m_image.order++;
651  item->updateImage();
652  emit item->imageNeedsSave();
653  }
654  }
655  m_image.order = minZValue;
656  updateImage();
657 }

675 void ImageItem::moveUp()

696 void ImageItem::moveDown()
697 {

699  {
700  return;
701  }
702 
703  foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
