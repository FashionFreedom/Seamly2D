148 //---------------------------------------------------------------------------------------------------------------------
149 VToolRotation *VToolRotation::Create(const quint32 _id, const quint32 &origin, QString &angle, const QString &suffix,
150  const QVector<SourceItem> &source,
151  const QVector<DestinationItem> &destination,
152  VMainGraphicsScene *scene, VAbstractPattern *doc, VContainer *data,
153  const Document &parse, const Source &typeCreation)
154 {
155  qreal calcAngle = 0;
156 
157  calcAngle = CheckFormula(_id, angle, data);
158 
159  const auto originPoint = *data->GeometricObject<VPointF>(origin);
160  const QPointF oPoint = static_cast<QPointF>(originPoint);
161 
162  QVector<DestinationItem> dest = destination;
163 
164  quint32 id = _id;
165  if (typeCreation == Source::FromGui)
166  {
167  dest.clear();// Try to avoid mistake, value must be empty
168 
169  id = VContainer::getNextId();//Just reserve id for tool
170 
171  for (int i = 0; i < source.size(); ++i)
172  {
173  const SourceItem item = source.at(i);
174  const quint32 objectId = item.id;
175  const QSharedPointer<VGObject> obj = data->GetGObject(objectId);
176 
177  // This check helps to find missed objects in the switch
178  Q_STATIC_ASSERT_X(static_cast<int>(GOType::Unknown) == 7, "Not all objects were handled.");
179 
180 QT_WARNING_PUSH
181 QT_WARNING_DISABLE_GCC("-Wswitch-default")
182  switch(static_cast<GOType>(obj->getType()))
183  {
184  case GOType::Point:
185  dest.append(createPoint(id, objectId, oPoint, calcAngle, suffix, data));
186  break;
187  case GOType::Arc:
188  dest.append(createArc<VArc>(id, objectId, oPoint, calcAngle, suffix, data));
189  break;
190  case GOType::EllipticalArc:
191  dest.append(createArc<VEllipticalArc>(id, objectId, oPoint, calcAngle, suffix, data));
192  break;
193  case GOType::Spline:
194  dest.append(createCurve<VSpline>(id, objectId, oPoint, calcAngle, suffix, data));
195  break;
196  case GOType::SplinePath:
197  dest.append(createCurveWithSegments<VSplinePath>(id, objectId, oPoint, calcAngle, suffix, data));
198  break;
199  case GOType::CubicBezier:
200  dest.append(createCurve<VCubicBezier>(id, objectId, oPoint, calcAngle, suffix, data));
201  break;
202  case GOType::CubicBezierPath:
203  dest.append(createCurveWithSegments<VCubicBezierPath>(id, objectId, oPoint, calcAngle, suffix,
204  data));
205  break;
206  case GOType::Unknown:
207  case GOType::Curve:
208  case GOType::Path:
209  case GOType::AllCurves:
210  default:
211  break;
212  }
213 QT_WARNING_POP
214  }
215  }
216  else
217  {
218  for (int i = 0; i < source.size(); ++i)
219  {
220  const SourceItem item = source.at(i);
221  const quint32 objectId = item.id;
222  const QSharedPointer<VGObject> obj = data->GetGObject(objectId);
223 
224  // This check helps to find missed objects in the switch
225  Q_STATIC_ASSERT_X(static_cast<int>(GOType::Unknown) == 7, "Not all objects were handled.");
226 
227 QT_WARNING_PUSH
228 QT_WARNING_DISABLE_GCC("-Wswitch-default")
229  switch(static_cast<GOType>(obj->getType()))
230  {
231  case GOType::Point:
232  {
233  const DestinationItem &item = dest.at(i);
234  updatePoint(id, objectId, oPoint, calcAngle, suffix, data, item);
235  break;
236  }
237  case GOType::Arc:
238  updateArc<VArc>(id, objectId, oPoint, calcAngle, suffix, data, dest.at(i).id);
239  break;
240  case GOType::EllipticalArc:
241  updateArc<VEllipticalArc>(id, objectId, oPoint, calcAngle, suffix, data, dest.at(i).id);
242  break;
243  case GOType::Spline:
244  updateCurve<VSpline>(id, objectId, oPoint, calcAngle, suffix, data, dest.at(i).id);
245  break;
246  case GOType::SplinePath:
247  updateCurveWithSegments<VSplinePath>(id, objectId, oPoint, calcAngle, suffix, data, dest.at(i).id);
248  break;
249  case GOType::CubicBezier:
250  updateCurve<VCubicBezier>(id, objectId, oPoint, calcAngle, suffix, data, dest.at(i).id);
251  break;
252  case GOType::CubicBezierPath:
253  updateCurveWithSegments<VCubicBezierPath>(id, objectId, oPoint, calcAngle, suffix, data,
254  dest.at(i).id);
255  break;
256  case GOType::Unknown:
257  case GOType::Curve:
258  case GOType::Path:
259  case GOType::AllCurves:
260  break;
261  }
262 QT_WARNING_POP
263  }
264  if (parse != Document::FullParse)
265  {
266  doc->UpdateToolData(id, data);
267  }
268  }
269 
270  if (parse == Document::FullParse)
271  {
272  VDrawTool::AddRecord(id, Tool::Rotation, doc);
273  VToolRotation *tool = new VToolRotation(doc, data, id, origin, angle, suffix, source, dest, typeCreation);
274  scene->addItem(tool);
275  initOperationToolConnections(scene, tool);
276  VAbstractPattern::AddTool(id, tool);
277  doc->IncrementReferens(originPoint.getIdTool());
278  for (int i = 0; i < source.size(); ++i)
279  {
280  const SourceItem item = source.at(i);
281  doc->IncrementReferens(data->GetGObject(item.id)->getIdTool());
282  }
283  return tool;
284  }
285  return nullptr;
286 }
