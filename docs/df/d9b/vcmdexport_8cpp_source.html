288  {

292  bool a = parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_PAGEH)));

295

301  }
302 
303  if ((a || b || c) && !(a && b && c))
304  {
305  qCritical() << translate("VCommandLine", "Page height, width, units must be used all 3 at once.") << "\n";
306  const_cast<VCommandLine*>(this)->parser.showHelp(V_EX_USAGE);

308 
309  }
310 
311  {

316  if ((a || b) && !(a && b))

319  << "\n";
320  const_cast<VCommandLine*>(this)->parser.showHelp(V_EX_USAGE);
321  }

328

383 
384  const int rotateDegree = OptRotation();
385  diag.SetRotate(rotateDegree != 0 ); // 0 disable rotation
386 
387  if (rotateDegree != VSettings::GetDefLayoutRotationIncrease())//Value by default
388  {
389  if (not diag.SetIncrease(rotateDegree))
390  {

392  << "\n";
393  const_cast<VCommandLine*>(this)->parser.showHelp(V_EX_USAGE);
394  }
395  }
396 
397  // if present units MUST be set before any other to keep conversions correct
398  if (!diag.SelectTemplate(OptPaperSize()))

401  const_cast<VCommandLine*>(this)->parser.showHelp(V_EX_USAGE);
402  }
403 
404  if (parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_PAGEH))))
405  { //at this point we already sure 3 are set or none
406

414  diag.SetPaperWidth (Pg2Px(parser.value(*optionsUsed.value(optionsIndex.value(LONG_OPTION_PAGEW))), diag));
415  }
416 
417  if (parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_SHIFTUNITS))))
418  {

420  {
421  qCritical() << translate("VCommandLine", "Unsupported layout units.") << "\n";
422  const_cast<VCommandLine*>(this)->parser.showHelp(V_EX_USAGE);
423  }
424  }
425 
426  if (parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_SHIFTLENGTH))))
427  {
428 
429  diag.SetShift(Lo2Px(parser.value(*optionsUsed.value(optionsIndex.value(LONG_OPTION_SHIFTLENGTH))), diag));
430  }
431 
432  if (parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_GAPWIDTH))))
433  {
434  diag.SetLayoutWidth(Lo2Px(parser.value(*optionsUsed.value(optionsIndex.value(LONG_OPTION_GAPWIDTH))), diag));
435  }
436 
437  diag.SetAutoCrop(parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_CROP))));
438  diag.SetUnitePages(parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_UNITE))));
439  diag.SetSaveLength(parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_SAVELENGTH))));
440  diag.SetGroup(OptGroup());
441 
442  if (parser.isSet(*optionsUsed.value(optionsIndex.value(LONG_OPTION_IGNORE_MARGINS))))
443  {
444  diag.SetIgnoreAllFields(true);
445  }
446  else
447  {
448  QMarginsF margins = diag.GetFields();
449 
450  {
451  const QCommandLineOption *option = optionsUsed.value(optionsIndex.value(LONG_OPTION_LEFT_MARGIN));
452  if (parser.isSet(*option))
453  {
454  margins.setLeft(Pg2Px(parser.value(*option), diag));
455  }
456  }
457 
458  {
459  const QCommandLineOption *option = optionsUsed.value(optionsIndex.value(LONG_OPTION_RIGHT_MARGIN));
460  if (parser.isSet(*option))
461  {
462  margins.setRight(Pg2Px(parser.value(*option), diag));
463  }
464  }
465 
466  {
467  const QCommandLineOption *option = optionsUsed.value(optionsIndex.value(LONG_OPTION_TOP_MARGIN));
468  if (parser.isSet(*option))
469  {
470  margins.setTop(Pg2Px(parser.value(*option), diag));
471  }
472  }
473 
474  {
475  const QCommandLineOption *option = optionsUsed.value(optionsIndex.value(LONG_OPTION_BOTTOM_MARGIN));
476  if (parser.isSet(*option))
477  {
478  margins.setBottom(Pg2Px(parser.value(*option), diag));
479  }
480  }
481  diag.SetFields(margins);
482  }
483 
484  diag.DialogAccepted(); // filling VLayoutGenerator
485

487 }
