66 QPainterPath VCurvePathItem::shape() const
67 {
68  QPainterPath itemPath;
69 
70  if (not m_points.isEmpty())
71  {
72  for (qint32 i = 0; i < m_points.count()-1; ++i)
73  {
74  itemPath.moveTo(m_points.at(i));
75  itemPath.lineTo(m_points.at(i+1));
76  }
77  }
78  else
79  {
80  itemPath = path();
81  }
82 
83  const QPainterPath arrowsPath = VAbstractCurve::ShowDirection(m_directionArrows,
84  scaleWidth(VAbstractCurve::lengthCurveDirectionArrow,
85  sceneScale(scene())));
86  if (arrowsPath != QPainterPath())
87  {
88  itemPath.addPath(arrowsPath);
89  }
90  itemPath.setFillRule(Qt::WindingFill);
91  return itemShapeFromPath(itemPath, pen());
92 }
