102 void VisToolBisector::RefreshGeometry()
103 {
104  if (object1Id > NULL_ID)
105  {
106  const QSharedPointer<VPointF> first = Visualization::data->GeometricObject<VPointF>(object1Id);
107  DrawPoint(line1P1, static_cast<QPointF>(*first), supportColor);
108 
109  if (object2Id <= NULL_ID)
110  {
111  DrawLine(line1, QLineF(static_cast<QPointF>(*first), Visualization::scenePos), supportColor, lineWeight);
112  }
113  else
114  {

116  DrawPoint(line1P2, static_cast<QPointF>(*second), supportColor);
117 
118  DrawLine(line1, QLineF(static_cast<QPointF>(*first), static_cast<QPointF>(*second)), supportColor, lineWeight);
119 
120  if (object3Id <= NULL_ID)
121  {
122  DrawLine(line2, QLineF(static_cast<QPointF>(*second), Visualization::scenePos), supportColor, lineWeight);
123  }
124  else
125  {
126  const QSharedPointer<VPointF> third = Visualization::data->GeometricObject<VPointF>(object3Id);
127  DrawPoint(line2P2, static_cast<QPointF>(*third), supportColor);
128 
129  DrawLine(line2, QLineF(static_cast<QPointF>(*second), static_cast<QPointF>(*third)), supportColor, lineWeight);
130 
131  if (not qFuzzyIsNull(length))
132  {
133  qreal angle = VToolBisector::BisectorAngle(static_cast<QPointF>(*first),
134  static_cast<QPointF>(*second),
135  static_cast<QPointF>(*third));
136  QLineF mainLine = VGObject::BuildLine(static_cast<QPointF>(*second), length, angle);
137  DrawLine(this, mainLine, mainColor, lineWeight, lineStyle);
138 
139  DrawPoint(point, mainLine.p2(), mainColor);
140  }
141  else
142  {
143  qreal angle = VToolBisector::BisectorAngle(static_cast<QPointF>(*first),
144  static_cast<QPointF>(*second),
145  static_cast<QPointF>(*third));
146  QPointF endRay = Ray(static_cast<QPointF>(*second), angle);
147  QLineF mainLine = VGObject::BuildLine(static_cast<QPointF>(*second),
148  QLineF(static_cast<QPointF>(*second), endRay).length(),
149  angle);
150  DrawLine(this, mainLine, mainColor, lineWeight, lineStyle);
151  }
