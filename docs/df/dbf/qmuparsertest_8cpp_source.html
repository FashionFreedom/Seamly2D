102 int QmuParserTester::IsHexVal ( const QString &a_szExpr, int *a_iPos, qreal *a_fVal, const QLocale &locale,
103  const QChar &decimal, const QChar &thousand )
104 {
105  Q_UNUSED(locale)
106  Q_UNUSED(decimal)
107  Q_UNUSED(thousand)
108  if ( a_szExpr.size() <= 2 || ( a_szExpr.at(0) != '0' || a_szExpr.at(1) != 'x' ) )
109  {
110  return 0;
111  }
112 
113  unsigned iVal ( 0 );
114 
115  std::wstring a_szExprStd = a_szExpr.mid(2).toStdWString();
116 
117  // New code based on streams for UNICODE compliance:
118  stringstream_type::pos_type nPos(0);
119  stringstream_type ss(a_szExprStd);
120  ss >> std::hex >> iVal;
121  nPos = ss.tellg();
122 
123  if (nPos==static_cast<stringstream_type::pos_type>(0))
124  {
125  return 1;
126  }
127 
128  *a_iPos += static_cast<int>(2 + nPos);
129  *a_fVal = static_cast<qreal>(iVal);
130  return 1;
131 }
