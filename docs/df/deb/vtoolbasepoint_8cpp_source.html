130 VToolBasePoint *VToolBasePoint::Create(quint32 _id, const QString &activeDraftBlock, VPointF *point,
131  VMainGraphicsScene *scene, VAbstractPattern *doc, VContainer *data,
132  const Document &parse, const Source &typeCreation)
133 {
134  SCASSERT(point != nullptr)
135 
136  quint32 id = _id;
137  if (typeCreation == Source::FromGui)
138  {
139  id = data->AddGObject(point);
140  }
141  else
142  {
143  data->UpdateGObject(id, point);
144  if (parse != Document::FullParse)
145  {
146  doc->UpdateToolData(id, data);
147  }
148  }
149 
150  if (parse == Document::FullParse)
151  {
152  VDrawTool::AddRecord(id, Tool::BasePoint, doc);
153  VToolBasePoint *spoint = new VToolBasePoint(doc, data, id, typeCreation, activeDraftBlock);
154  scene->addItem(spoint);
155  InitToolConnections(scene, spoint);
156  VAbstractPattern::AddTool(id, spoint);
157  return spoint;
158  }
159  return nullptr;
160 }

172 void VToolBasePoint::AddToFile()
173 {
174  Q_ASSERT_X(not draftBlockName.isEmpty(), Q_FUNC_INFO, "name pattern piece is empty");
175 
176  QDomElement sPoint = doc->createElement(getTagName());
177 
178  // Create SPoint tag
179  QSharedPointer<VGObject> obj = VAbstractTool::data.GetGObject(m_id);
180  SaveOptions(sPoint, obj);
181 
182  //Create pattern piece structure
183  QDomElement patternPiece = doc->createElement(VAbstractPattern::TagDraftBlock);
184  doc->SetAttribute(patternPiece, AttrName, draftBlockName);
185 
186  QDomElement calcElement = doc->createElement(VAbstractPattern::TagCalculation);
187  calcElement.appendChild(sPoint);
188 
189  patternPiece.appendChild(calcElement);
190  patternPiece.appendChild(doc->createElement(VAbstractPattern::TagModeling));
191  patternPiece.appendChild(doc->createElement(VAbstractPattern::TagPieces));
192 
193  AddDraftBlock *addPP = new AddDraftBlock(patternPiece, doc, draftBlockName);
194  connect(addPP, &AddDraftBlock::ClearScene, doc, &VAbstractPattern::ClearScene);
195  connect(addPP, &AddDraftBlock::NeedFullParsing, doc, &VAbstractPattern::NeedFullParsing);
196  qApp->getUndoStack()->push(addPP);
197 }

200 /**
201  * @brief itemChange handle tool change.
202  * @param change change.
203  * @param value value.
204  * @return value.
205  */
206 QVariant VToolBasePoint::itemChange(QGraphicsItem::GraphicsItemChange change, const QVariant &value)
207 {
208  if (change == ItemPositionChange && scene())
209  {
210  // Each time we move something we call recalculation scene rect. In some cases this can cause moving
211  // objects positions. And this cause infinite redrawing. That's why we wait the finish of saving the last move.
212  static bool changeFinished = true;
213  if (changeFinished)
214  {
215  changeFinished = false;
216  // value - this is new position.
217  QPointF newPos = value.toPointF();
218 
219  MoveSPoint *moveSP = new MoveSPoint(doc, newPos.x(), newPos.y(), m_id, this->scene());
220  connect(moveSP, &MoveSPoint::NeedLiteParsing, doc, &VAbstractPattern::LiteParseTree);
221  qApp->getUndoStack()->push(moveSP);
222  const QList<QGraphicsView *> viewList = scene()->views();
223  if (not viewList.isEmpty())
224  {
225  if (QGraphicsView *view = viewList.at(0))
226  {
227  const int xmargin = 50;
228  const int ymargin = 50;
229 
230  const QRectF viewRect = VMainGraphicsView::SceneVisibleArea(view);
231  const QRectF itemRect = mapToScene(boundingRect()).boundingRect();
232 
233  // If item's rect is bigger than view's rect ensureVisible works very unstable.
234  if (itemRect.height() + 2*ymargin < viewRect.height() &&
235  itemRect.width() + 2*xmargin < viewRect.width())
236  {
237  view->ensureVisible(itemRect, xmargin, ymargin);
238  }
239  else
240  {
241  // Ensure visible only small rect around a cursor
242  VMainGraphicsScene *currentScene = qobject_cast<VMainGraphicsScene *>(scene());
243  SCASSERT(currentScene)
244  const QPointF cursorPosition = currentScene->getScenePos();
245  view->ensureVisible(QRectF(cursorPosition.x()-5, cursorPosition.y()-5, 10, 10));
246  }
247  }
248  }
249  changeFinished = true;
250  }
251  }
252  return VToolSinglePoint::itemChange(change, value);
253 }

289 {
290  qCDebug(vTool, "Deleting base point.");
291  qApp->getSceneView()->itemClicked(nullptr);
292  if (ask)
293  {
294  qCDebug(vTool, "Asking.");
295  if (ConfirmDeletion() == QMessageBox::No)
296  {
297  qCDebug(vTool, "User said no.");
298  return;
299  }
300  }
301 
302  qCDebug(vTool, "Begin deleting.");
303  DeleteDraftBlock *deletePP = new DeleteDraftBlock(doc, activeBlockName);
304  connect(deletePP, &DeleteDraftBlock::NeedFullParsing, doc, &VAbstractPattern::NeedFullParsing);
305  qApp->getUndoStack()->push(deletePP);
306 
307  // Throw exception, this will help prevent case when we forget to immediately quit function.
308  VExceptionToolWasDeleted e("Tool was used after deleting.");
309  throw e;
