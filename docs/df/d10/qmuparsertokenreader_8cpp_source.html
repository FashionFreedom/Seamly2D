238  * @brief Read the next token from the string.
239  */
240 QmuParserTokenReader::token_type QmuParserTokenReader::ReadNextToken(const QLocale &locale, const QChar &decimal,
241  const QChar &thousand)
242 {
243  assert ( m_pParser );
244 
245  token_type tok;
246 
247  // Ignore all non printable characters when reading the expression
248  while (m_strFormula.size() > m_iPos && m_strFormula.at(m_iPos) <= QChar(0x20))
249  {
250  ++m_iPos;
251  }
252 
253  if ( IsEOF ( tok ) )
254  {
255  return SaveBeforeReturn ( tok ); // Check for end of formula
256  }
257  if ( IsOprt ( tok ) )
258  {
259  return SaveBeforeReturn ( tok ); // Check for user defined binary operator
260  }
261  if ( IsFunTok ( tok ) )
262  {
263  return SaveBeforeReturn ( tok ); // Check for function token
264  }
265  if ( IsBuiltIn ( tok ) )
266  {
267  return SaveBeforeReturn ( tok ); // Check built in operators / tokens
268  }
269  if ( IsArgSep ( tok ) )
270  {
271  return SaveBeforeReturn ( tok ); // Check for function argument separators
272  }
273  if ( IsValTok ( tok, locale, decimal, thousand ) )
274  {
275  return SaveBeforeReturn ( tok ); // Check for values / constant tokens
276  }
277  if ( IsVarTok ( tok ) )
278  {
279  return SaveBeforeReturn ( tok ); // Check for variable tokens
280  }
281  if ( IsStrVarTok ( tok ) )
282  {
283  return SaveBeforeReturn ( tok ); // Check for string variables
284  }
285  if ( IsString ( tok ) )
286  {
287  return SaveBeforeReturn ( tok ); // Check for String tokens
288  }
289  if ( IsInfixOpTok ( tok ) )
290  {
291  return SaveBeforeReturn ( tok ); // Check for unary operators
292  }
293  if ( IsPostOpTok ( tok ) )
294  {
295  return SaveBeforeReturn ( tok ); // Check for unary operators
296  }
297 
298  // Check String for undefined variable token. Done only if a
299  // flag is set indicating to ignore undefined variables.
300  // This is a way to conditionally avoid an error if
301  // undefined variables occur.
302  // (The GetUsedVar function must suppress the error for
303  // undefined variables in order to collect all variable
304  // names including the undefined ones.)
305  if ( ( m_bIgnoreUndefVar || m_pFactory ) && IsUndefVarTok ( tok ) )
306  {
307  return SaveBeforeReturn ( tok );
308  }
309 
310  // Check for unknown token
311  //
312  // !!! From this point on there is no exit without an exception possible...
313  //
314  QString strTok;
315  int iEnd = ExtractToken ( m_pParser->ValidNameChars(), strTok, m_iPos );
316  if ( iEnd != m_iPos )
317  {
318  Error ( ecUNASSIGNABLE_TOKEN, m_iPos, strTok );
319  }
320 
321  Error ( ecUNASSIGNABLE_TOKEN, m_iPos, m_strFormula.mid ( m_iPos ) );
322  return token_type(); // never reached
323 }

339 /**
340  * @brief Extract all characters that belong to a certain charset.
341  *
342  * @param a_szCharSet [in] Const char array of the characters allowed in the token.
343  * @param a_sTok [out] The string that consists entirely of characters listed in a_szCharSet.
344  * @param a_iPos [in] Position in the string from where to start reading.
345  * @return The Position of the first character not listed in a_szCharSet.
346  * @throw nothrow
347  */
348 QT_WARNING_PUSH
349 QT_WARNING_DISABLE_MSVC(4309)
350 int QmuParserTokenReader::ExtractToken ( const QString &a_szCharSet, QString &a_sTok, int a_iPos ) const
351 {
352  const std::wstring m_strFormulaStd = m_strFormula.toStdWString();
353  const std::wstring a_szCharSetStd = a_szCharSet.toStdWString();
354 
355  int iEnd = static_cast<int>(m_strFormulaStd.find_first_not_of ( a_szCharSetStd, static_cast<std::size_t>(a_iPos) ));
356 
357  if ( iEnd == static_cast<int>(string_type::npos) )
358  {
359  iEnd = static_cast<int>(m_strFormulaStd.length());
360  }
361 
362  // Assign token string if there was something found
363  if ( a_iPos != iEnd )
364  {
365  a_sTok = QString().fromStdWString ( std::wstring ( m_strFormulaStd.begin() + a_iPos,
366  m_strFormulaStd.begin() + iEnd ) );
367  }
368 
369  return iEnd;
370 }

373 /**
374  * @brief Check Expression for the presence of a binary operator token.
375  *
376  * Userdefined binary operator "++" gives inconsistent parsing result for the equations "a++b" and "a ++ b" if
377  * alphabetic characters are allowed in operator tokens. To avoid this this function checks specifically
378  * for operator tokens.
379  */
380 int QmuParserTokenReader::ExtractOperatorToken ( QString &a_sTok, int a_iPos ) const
381 {
382  const std::wstring m_strFormulaStd = m_strFormula.toStdWString();
383  // Changed as per Issue 6: https://code.google.com/p/muparser/issues/detail?id=6
384  const std::wstring oprtCharsStd = m_pParser->ValidOprtChars().toStdWString();
385 
386  int iEnd = static_cast<int>( m_strFormulaStd.find_first_not_of ( oprtCharsStd, static_cast<std::size_t>(a_iPos) ) );
387  if ( iEnd == static_cast<int>( string_type::npos ) )
388  {
389  iEnd = static_cast<int>( m_strFormulaStd.length() );
390  }
391 
392  // Assign token string if there was something found
393  if ( a_iPos != iEnd )
394  {
395  a_sTok = QString().fromStdWString ( string_type ( m_strFormulaStd.begin() + a_iPos,
396  m_strFormulaStd.begin() + iEnd ) );
397  return iEnd;
398  }
399  else
400  {
401  // There is still the chance of having to deal with an operator consisting exclusively
402  // of alphabetic characters.
403  return ExtractToken ( QMUP_CHARS, a_sTok, a_iPos );
404  }
405 }

409 /**
410  * @brief Check if a built in operator or other token can be found
411  * @param a_Tok [out] Operator token if one is found. This can either be a binary operator or an infix operator token.
412  * @return true if an operator token has been found.
413  */
414 bool QmuParserTokenReader::IsBuiltIn ( token_type &a_Tok )
415 {
416  const QStringList pOprtDef = QmuParserBase::GetOprtDef();
417 
418  // Compare token with function and operator strings
419  // check string for operator/function
420  for ( int i = 0; i < pOprtDef.size(); ++i )
421  {
422  int len = pOprtDef.at ( i ).length();
423  if ( pOprtDef.at ( i ) == m_strFormula.mid ( m_iPos, len ) )
424  {
425  if (i >= cmLE && i <= cmASSIGN)
426  {
427  //if (len!=sTok.length())
428  // continue;
429 
430  // The assignement operator need special treatment
431  if ( i == cmASSIGN && m_iSynFlags & noASSIGN )
432  {
433  Error ( ecUNEXPECTED_OPERATOR, m_iPos, pOprtDef.at ( i ) );
434  }
435 
436  if ( m_pParser->HasBuiltInOprt() == false)
437  {
438  continue;
439  }
440  if ( m_iSynFlags & noOPT )
441  {
442  // Maybe its an infix operator not an operator
443  // Both operator types can share characters in
444  // their identifiers
445  if ( IsInfixOpTok ( a_Tok ) )
446  {
447  return true;
448  }
449 
450  Error ( ecUNEXPECTED_OPERATOR, m_iPos, pOprtDef.at ( i ) );
451  }
452 
453  m_iSynFlags = noBC | noOPT | noARG_SEP | noPOSTOP | noASSIGN | noIF | noELSE;
454  m_iSynFlags |= noEND;
455  }
456  else if (i == cmBO)
457  {
458  if ( m_iSynFlags & noBO )
459  {
460  Error ( ecUNEXPECTED_PARENS, m_iPos, pOprtDef.at ( i ) );
461  }
462 
463  if ( m_lastTok.GetCode() == cmFUNC )
464  {
465  m_iSynFlags = noOPT | noEND | noARG_SEP | noPOSTOP | noASSIGN | noIF | noELSE;
466  }
467  else
468  {
469  m_iSynFlags = noBC | noOPT | noEND | noARG_SEP | noPOSTOP | noASSIGN | noIF | noELSE;
470  }
471 
472  ++m_iBrackets;
473  }
474  else if (i == cmBC)
475  {
476  if ( m_iSynFlags & noBC )
477  {
478  Error ( ecUNEXPECTED_PARENS, m_iPos, pOprtDef.at ( i ) );
479  }
480 
481  m_iSynFlags = noBO | noVAR | noVAL | noFUN | noINFIXOP | noSTR | noASSIGN;
482 
483  if ( --m_iBrackets < 0 )
484  {
485  Error ( ecUNEXPECTED_PARENS, m_iPos, pOprtDef.at ( i ) );
486  }
487  }
488  else if (i == cmELSE)
489  {
490  if ( m_iSynFlags & noELSE )
491  {
492  Error ( ecUNEXPECTED_CONDITIONAL, m_iPos, pOprtDef.at ( i ) );
493  }
494 
495  m_iSynFlags = noBC | noPOSTOP | noEND | noOPT | noIF | noELSE;
496  }
497  else if (i == cmIF)
498  {
499  if ( m_iSynFlags & noIF )
500  {
501  Error ( ecUNEXPECTED_CONDITIONAL, m_iPos, pOprtDef.at ( i ) );
502  }
503 
504  m_iSynFlags = noBC | noPOSTOP | noEND | noOPT | noIF | noELSE;
505  }
506  else // The operator is listed in c_DefaultOprt, but not here. This is a bad thing...
507  {
508  Error ( ecINTERNAL_ERROR );
509  }
510 
511  m_iPos += len;
512  a_Tok.Set ( static_cast<ECmdCode>(i), pOprtDef.at ( i ) );
513  return true;
514  } // if operator string found
515  } // end of for all operator strings
516 
517  return false;
518 }

545 /**
546  * @brief Check for End of Formula.
547  *
548  * @return true if an end of formula is found false otherwise.
549  * @param a_Tok [out] If an eof is found the corresponding token will be stored there.
550  * @sa IsOprt, IsFunTok, IsStrFunTok, IsValTok, IsVarTok, IsString, IsInfixOpTok, IsPostOpTok
551  */
552 bool QmuParserTokenReader::IsEOF ( token_type &a_Tok )
553 {
554  // check for EOF
555  if ( m_iPos >= m_strFormula.size())
556  {
557  if ( m_iSynFlags & noEND )
558  {
559  try
560  {
561  Error ( ecUNEXPECTED_EOF, m_iPos );
562  }
563  catch (qmu::QmuParserError &error)
564  {
565  qDebug() << " Code:" << error.GetCode() << "(" << error.GetMsg() << ")";
566  throw;
567  }
568  }
569 
570  if ( m_iBrackets > 0 )
571  {
572  Error ( ecMISSING_PARENS, m_iPos, ")" );
573  }
574 
575  m_iSynFlags = 0;
576  a_Tok.Set ( cmEND );
577  return true;
578  }
579 
580  return false;
581 }

584 /**
585  * @brief Check if a string position contains a unary infix operator.
586  * @return true if a function token has been found false otherwise.
587  */
588 bool QmuParserTokenReader::IsInfixOpTok ( token_type &a_Tok )
589 {
590  QString sTok;
591  int iEnd = ExtractToken ( m_pParser->ValidInfixOprtChars(), sTok, m_iPos );
592  if ( iEnd == m_iPos )
593  {
594  return false;
595  }
596 
597  // iteraterate over all postfix operator strings
598  auto it = m_pInfixOprtDef->rbegin();
599  for ( ; it != m_pInfixOprtDef->rend(); ++it )
600  {
601  if ( sTok.indexOf ( it->first ) == 0 )
602  {
603  a_Tok.Set ( it->second, it->first );
604  m_iPos += static_cast<int>(it->first.length());
605 
606  if ( m_iSynFlags & noINFIXOP )
607  {
608  Error ( ecUNEXPECTED_OPERATOR, m_iPos, a_Tok.GetAsString() );
609  }
610 
611  m_iSynFlags = noPOSTOP | noINFIXOP | noOPT | noBC | noSTR | noASSIGN;
612  return true;
613  }
614  }
615 
616  return false;
617 }

620 /**
621  * @brief Check whether the token at a given position is a function token.
622  * @param a_Tok [out] If a value token is found it will be placed here.
623  * @throw ParserException if Syntaxflags do not allow a function at a_iPos
624  * @return true if a function token has been found false otherwise.
625  * @pre [assert] m_pParser!=0
626  */
627 bool QmuParserTokenReader::IsFunTok ( token_type &a_Tok )
628 {
629  QString strTok;
630  int iEnd = ExtractToken ( m_pParser->ValidNameChars(), strTok, m_iPos );
631  if ( iEnd == m_iPos )
632  {
633  return false;
634  }
635 
636  funmap_type::const_iterator item = m_pFunDef->find ( strTok );
637  if ( item == m_pFunDef->end() )
638  {
639  return false;
640  }
641

661 /**
662  * @brief Check if a string position contains a binary operator.
663  * @param a_Tok [out] Operator token if one is found. This can either be a binary operator or an infix operator token.
664  * @return true if an operator token has been found.
665  */
666 bool QmuParserTokenReader::IsOprt ( token_type &a_Tok )
667 {
668  QString strTok;
669 
670  int iEnd = ExtractOperatorToken ( strTok, m_iPos );
671  if ( iEnd == m_iPos )
672  {
673  return false;
674  }
675 
676  // Check if the operator is a built in operator, if so ignore it here
677  const QStringList &pOprtDef = QmuParserBase::GetOprtDef();
678  QStringList::const_iterator constIterator;
679  for ( constIterator = pOprtDef.constBegin(); m_pParser->HasBuiltInOprt() && constIterator != pOprtDef.constEnd();
680  ++constIterator )
681  {
682  if ( ( *constIterator ) == strTok )
683  {
684  return false;
685  }
686  }
687 
688  // Note:
689  // All tokens in oprt_bin_maptype are have been sorted by their length
690  // Long operators must come first! Otherwise short names (like: "add") that
691  // are part of long token names (like: "add123") will be found instead
692  // of the long ones.
693  // Length sorting is done with ascending length so we use a reverse iterator here.
694  auto it = m_pOprtDef->rbegin();
695  for ( ; it != m_pOprtDef->rend(); ++it )
696  {
697  const QString &sID = it->first;
698  if ( sID == m_strFormula.mid ( m_iPos, sID.length() ) )
699  {
700  a_Tok.Set ( it->second, strTok );
701 
702  // operator was found
703  if ( m_iSynFlags & noOPT )
704  {
705  // An operator was found but is not expected to occur at
706  // this position of the formula, maybe it is an infix
707  // operator, not a binary operator. Both operator types
708  // can share characters in their identifiers.
709  if ( IsInfixOpTok ( a_Tok ) )
710  {
711  return true;
712  }
713  else
714  {
715  // nope, no infix operator
716  return false;
717  //Error(ecUNEXPECTED_OPERATOR, m_iPos, a_Tok.GetAsString());
718  }
719 
720  }
721 
722  m_iPos += sID.length();
723  m_iSynFlags = noBC | noOPT | noARG_SEP | noPOSTOP | noEND | noASSIGN;
724  return true;
725  }
726  }
727 
728  return false;
729 }

733  * @brief Check if a string position contains a unary post value operator.
734  */
735 bool QmuParserTokenReader::IsPostOpTok ( token_type &a_Tok )
736 {
737  // <ibg 20110629> Do not check for postfix operators if they are not allowed at
738  // the current expression index.
739  //
740  // This will fix the bug reported here:
741  //
742  // http://sourceforge.net/tracker/index.php?func=detail&aid=3343891&group_id=137191&atid=737979
743  //
744  if ( m_iSynFlags & noPOSTOP )
745  {
746  return false;
747  }
748  // </ibg>
749 
750  // Tricky problem with equations like "3m+5":
751  // m is a postfix operator, + is a valid sign for postfix operators and
752  // for binary operators parser detects "m+" as operator string and
753  // finds no matching postfix operator.
754  //
755  // This is a special case so this routine slightly differs from the other
756  // token readers.
757 
758  // Test if there could be a postfix operator
759  QString sTok;
760  int iEnd = ExtractToken ( m_pParser->ValidOprtChars(), sTok, m_iPos );
761  if ( iEnd == m_iPos )
762  {
763  return false;
764  }
765 
766  // iteraterate over all postfix operator strings
767  auto it = m_pPostOprtDef->rbegin();
768  for ( ; it != m_pPostOprtDef->rend(); ++it )
769  {
770  if ( sTok.indexOf ( it->first ) == 0 )
771  {
772  a_Tok.Set ( it->second, sTok );
773  m_iPos += it->first.length();
774 
775  m_iSynFlags = noVAL | noVAR | noFUN | noBO | noPOSTOP | noSTR | noASSIGN;
776  return true;
777  }
778  }
779 
780  return false;
781 }

986 /**
987  * @brief Check wheter a token at a given position is a string.
988  * @param a_Tok [out] If a variable token has been found it will be placed here.
989  * @return true if a string token has been found.
990  * @sa IsOprt, IsFunTok, IsStrFunTok, IsValTok, IsVarTok, IsEOF, IsInfixOpTok, IsPostOpTok
991  */
992 bool QmuParserTokenReader::IsString ( token_type &a_Tok )
993 {
994  if ( m_strFormula[m_iPos] != '"' )
995  {
996  return false;
997  }
998 
999  QString strBuf (m_strFormula.mid(m_iPos + 1));
1000  int iEnd ( 0 ), iSkip ( 0 );
1001 
1002  // parser over escaped '\"' end replace them with '"'
1003  for ( iEnd = strBuf.indexOf ( "\"" ); iEnd != 0 && iEnd != -1; iEnd = strBuf.indexOf ( "\"", iEnd ) )
1004  {
1005  if ( strBuf[iEnd - 1] != '\\' )
1006  {
1007  break;
1008  }
1009  strBuf.replace ( iEnd - 1, 2, "\"" );
1010  iSkip++;
1011  }
1012 
1013  if ( iEnd == -1 )
1014  {
1015  Error ( ecUNTERMINATED_STRING, m_iPos, "\"" );
1016  }
1017 
1018  QString strTok = strBuf.mid ( 0, iEnd );
1019 
1020  if ( m_iSynFlags & noSTR )
1021  {
1022  Error ( ecUNEXPECTED_STR, m_iPos, strTok );
1023  }
1024 
1025  m_pParser->m_vStringBuf.push_back ( strTok ); // Store string in internal buffer
1026  a_Tok.SetString ( strTok, m_pParser->m_vStringBuf.size() );
1027 
1028  m_iPos += strTok.length() + 2 + iSkip; // +2 wg Anfhrungszeichen; +iSkip fr entfernte escape zeichen
1029  m_iSynFlags = noANY ^ ( noARG_SEP | noBC | noOPT | noEND );
1030 
1031  return true;
1032 }
