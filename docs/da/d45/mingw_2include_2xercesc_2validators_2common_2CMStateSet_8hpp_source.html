28 // the validator. It assumes that its never called with two objects of
29 // different bit counts, and that bit sets smaller than a threshold are far

31 // purpose utility bitset class

76  // -----------------------------------------------------------------------
77  // Constructors and Destructor
78  // -----------------------------------------------------------------------
79  CMStateSet( const XMLSize_t bitCount
80  , MemoryManager* const manager = XMLPlatformUtils::fgMemoryManager) :
81 
82  fBitCount(bitCount)
83  , fDynamicBuffer(0)
84  {
85  //
86  // See if we need to allocate the byte array or whether we can live
87  // within the cached bit high performance scheme.
88  //
89  if (fBitCount > (CMSTATE_CACHED_INT32_SIZE * 32))
90  {
91  fDynamicBuffer = (CMDynamicBuffer*)manager->allocate(sizeof(CMDynamicBuffer));
92  fDynamicBuffer->fMemoryManager = manager;
93  // allocate an array of vectors, each one containing CMSTATE_BITFIELD_CHUNK bits
94  fDynamicBuffer->fArraySize = fBitCount / CMSTATE_BITFIELD_CHUNK;
95  if (fBitCount % CMSTATE_BITFIELD_CHUNK)
96  fDynamicBuffer->fArraySize++;
97  try
98  {
99  fDynamicBuffer->fBitArray = (XMLInt32**) fDynamicBuffer->fMemoryManager->allocate(fDynamicBuffer->fArraySize*sizeof(XMLInt32*));
100  }
101  catch( const OutOfMemoryException& )
102  {
103  fDynamicBuffer->fMemoryManager->deallocate(fDynamicBuffer);
104  throw;
105  }
106  for(XMLSize_t index = 0; index < fDynamicBuffer->fArraySize; index++)
107  fDynamicBuffer->fBitArray[index]=NULL;
108  }
109  else
110  {
111  for (XMLSize_t index = 0; index < CMSTATE_CACHED_INT32_SIZE; index++)
112  fBits[index] = 0;
113  }
114  }

127  fDynamicBuffer = (CMDynamicBuffer*) toCopy.fDynamicBuffer->fMemoryManager->allocate(sizeof(CMDynamicBuffer));
128  fDynamicBuffer->fMemoryManager = toCopy.fDynamicBuffer->fMemoryManager;

130  if (fBitCount % CMSTATE_BITFIELD_CHUNK)

170  void operator|=(const CMStateSet& setToOr)
171  {
172  if(fDynamicBuffer==0)
173  {
174 #ifdef XERCES_HAVE_SSE2_INTRINSIC
175  if(XMLPlatformUtils::fgSSE2ok)
176  {
177  __m128i xmm1 = _mm_loadu_si128(reinterpret_cast<const __m128i*>(fBits));
178  __m128i xmm2 = _mm_loadu_si128(reinterpret_cast<const __m128i*>(setToOr.fBits));
179  __m128i xmm3 = _mm_or_si128(xmm1, xmm2); // OR 4 32-bit words
180  _mm_storeu_si128(reinterpret_cast<__m128i*>(fBits), xmm3);
181  }
182  else
183 #endif
184  {
185  for (XMLSize_t index = 0; index < CMSTATE_CACHED_INT32_SIZE; index++)
186  if(setToOr.fBits[index])
187  {
188  if(fBits[index])
189  fBits[index] |= setToOr.fBits[index];
190  else
191  fBits[index] = setToOr.fBits[index];
192  }
193  }
194  }
195  else
196  {
197  for (XMLSize_t index = 0; index < fDynamicBuffer->fArraySize; index++)
198  {
199  XMLInt32 *& other = setToOr.fDynamicBuffer->fBitArray[index];
200  if(other!=NULL)
201  {
202  // if we haven't allocated the subvector yet, allocate it and copy
203  if(fDynamicBuffer->fBitArray[index]==NULL)
204  {
205  allocateChunk(index);
206  memcpy((void *) fDynamicBuffer->fBitArray[index],
207  (const void *) other,
208  CMSTATE_BITFIELD_INT32_SIZE * sizeof(XMLInt32));
209  }
210  else
211  {
212  // otherwise, merge them
213  XMLInt32*& mine = fDynamicBuffer->fBitArray[index];
214 #ifdef XERCES_HAVE_SSE2_INTRINSIC
215  if(XMLPlatformUtils::fgSSE2ok)
216  {
217  for(XMLSize_t subIndex = 0; subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex+=4)
218  {
219  __m128i xmm1 = _mm_load_si128(reinterpret_cast<const __m128i*>(&other[subIndex]));
220  __m128i xmm2 = _mm_load_si128(reinterpret_cast<const __m128i*>(&mine[subIndex]));
221  __m128i xmm3 = _mm_or_si128(xmm1, xmm2); // OR 4 32-bit words
222  _mm_store_si128(reinterpret_cast<__m128i*>(&mine[subIndex]), xmm3);
223  }
224  }
225  else
226 #endif
227  {
228  for(XMLSize_t subIndex = 0; subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex++)
229  if(setToOr.fDynamicBuffer->fBitArray[index][subIndex])
230  {
231  if(fDynamicBuffer->fBitArray[index][subIndex])
232  fDynamicBuffer->fBitArray[index][subIndex] |= setToOr.fDynamicBuffer->fBitArray[index][subIndex];
233  else
234  fDynamicBuffer->fBitArray[index][subIndex] = setToOr.fDynamicBuffer->fBitArray[index][subIndex];
235  }
236  }
237  }
238  }
239  }
240  }

243  bool operator==(const CMStateSet& setToCompare) const
244  {
245  if (fBitCount != setToCompare.fBitCount)
246  return false;
247 
248  if(fDynamicBuffer==0)
249  {
250  for (XMLSize_t index = 0; index < CMSTATE_CACHED_INT32_SIZE; index++)
251  {
252  if (fBits[index] != setToCompare.fBits[index])
253  return false;
254  }
255  }
256  else
257  {
258  for (XMLSize_t index = 0; index < fDynamicBuffer->fArraySize; index++)
259  {
260  XMLInt32 *& other = setToCompare.fDynamicBuffer->fBitArray[index],
261  *& mine = fDynamicBuffer->fBitArray[index];
262  if(mine==NULL && other==NULL)
263  continue;
264  else if(mine==NULL || other==NULL) // the other should have been empty too
265  return false;
266  else
267  {
268  for(XMLSize_t subIndex = 0; subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex++)
269  if(mine[subIndex]!=other[subIndex])
270  return false;
271  }
272  }
273  }
274  return true;
275  }

277  CMStateSet& operator=(const CMStateSet& srcSet)
278  {
279  if (this == &srcSet)
280  return *this;
281 
282  // They have to be the same size
283  if (fBitCount != srcSet.fBitCount)
284  {
285  if(fDynamicBuffer)
286  ThrowXMLwithMemMgr(RuntimeException, XMLExcepts::Bitset_NotEqualSize, fDynamicBuffer->fMemoryManager);
287  else
288  ThrowXML(RuntimeException, XMLExcepts::Bitset_NotEqualSize);
289  }
290 
291  if(fDynamicBuffer==0)
292  {
293  for (XMLSize_t index = 0; index < CMSTATE_CACHED_INT32_SIZE; index++)
294  fBits[index] = srcSet.fBits[index];
295  }
296  else
297  {
298  for (XMLSize_t index = 0; index < fDynamicBuffer->fArraySize; index++)
299  if(srcSet.fDynamicBuffer->fBitArray[index]==NULL)
300  {
301  // delete this subentry
302  if(fDynamicBuffer->fBitArray[index]!=NULL)
303  deallocateChunk(index);
304  }
305  else
306  {
307  // if we haven't allocated the subvector yet, allocate it and copy
308  if(fDynamicBuffer->fBitArray[index]==NULL)
309  allocateChunk(index);
310  memcpy((void *) fDynamicBuffer->fBitArray[index],
311  (const void *) srcSet.fDynamicBuffer->fBitArray[index],
312  CMSTATE_BITFIELD_INT32_SIZE * sizeof(XMLInt32));
313  }
314  }
315  return *this;

318  XMLSize_t getBitCountInRange(XMLSize_t start, XMLSize_t end) const
319  {
320  XMLSize_t count = 0;
321  end /= 32;
322  if(fDynamicBuffer==0)
323  {
324  if(end > CMSTATE_CACHED_INT32_SIZE)
325  end = CMSTATE_CACHED_INT32_SIZE;
326  for (XMLSize_t index = start / 32; index < end; index++)
327  {
328  if (fBits[index] != 0)
329  for(int i=0;i<32;i++)
330  {
331  const XMLInt32 mask = 1UL << i;
332  if(fBits[index] & mask)
333  count++;
334  }
335  }
336  }
337  else
338  {
339  if(end > fDynamicBuffer->fArraySize)
340  end = fDynamicBuffer->fArraySize;
341  for (XMLSize_t index = start / 32; index < end; index++)
342  {
343  if(fDynamicBuffer->fBitArray[index]==NULL)
344  continue;
345  for(XMLSize_t subIndex=0;subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex++)
346  {
347  if (fDynamicBuffer->fBitArray[index][subIndex] != 0)
348  for(int i=0;i<32;i++)
349  {
350  const XMLInt32 mask = 1UL << i;
351  if(fDynamicBuffer->fBitArray[index][subIndex] & mask)
352  count++;
353  }
354  }
355  }
356  }
357  return count;

360  bool getBit(const XMLSize_t bitToGet) const
361  {
362  if (bitToGet >= fBitCount)
363  {
364  if(fDynamicBuffer)
365  ThrowXMLwithMemMgr(ArrayIndexOutOfBoundsException, XMLExcepts::Bitset_BadIndex, fDynamicBuffer->fMemoryManager);
366  else
367  ThrowXML(ArrayIndexOutOfBoundsException, XMLExcepts::Bitset_BadIndex);
368  }
369 
370  // And access the right bit and byte
371  if(fDynamicBuffer==0)
372  {
373  const XMLInt32 mask = 1UL << (bitToGet % 32);
374  const XMLSize_t byteOfs = bitToGet / 32;
375  return (fBits[byteOfs]!=0 && (fBits[byteOfs] & mask) != 0);
376  }
377  else
378  {
379  const XMLSize_t vectorOfs = bitToGet / CMSTATE_BITFIELD_CHUNK;
380  if(fDynamicBuffer->fBitArray[vectorOfs]==NULL)
381  return false;
382  const XMLInt32 mask = 1UL << (bitToGet % 32);
383  const XMLSize_t byteOfs = (bitToGet % CMSTATE_BITFIELD_CHUNK) / 32;
384  return (fDynamicBuffer->fBitArray[vectorOfs][byteOfs]!=0 && (fDynamicBuffer->fBitArray[vectorOfs][byteOfs] & mask) != 0);
385  }
386  }

414  void setBit(const XMLSize_t bitToSet)
415  {
416  if (bitToSet >= fBitCount)
417  {
418  if(fDynamicBuffer)
419  ThrowXMLwithMemMgr(ArrayIndexOutOfBoundsException, XMLExcepts::Bitset_BadIndex, fDynamicBuffer->fMemoryManager);
420  else
421  ThrowXML(ArrayIndexOutOfBoundsException, XMLExcepts::Bitset_BadIndex);
422  }
423 
424  const XMLInt32 mask = 1UL << (bitToSet % 32);
425 
426  // And access the right bit and byte
427  if(fDynamicBuffer==0)
428  {
429  const XMLSize_t byteOfs = bitToSet / 32;
430  fBits[byteOfs] &= ~mask;
431  fBits[byteOfs] |= mask;
432  }
433  else
434  {
435  const XMLSize_t vectorOfs = bitToSet / CMSTATE_BITFIELD_CHUNK;
436  if(fDynamicBuffer->fBitArray[vectorOfs]==NULL)
437  {
438  allocateChunk(vectorOfs);
439  for(XMLSize_t index=0;index < CMSTATE_BITFIELD_INT32_SIZE; index++)
440  fDynamicBuffer->fBitArray[vectorOfs][index]=0;
441  }
442  const XMLSize_t byteOfs = (bitToSet % CMSTATE_BITFIELD_CHUNK) / 32;
443  fDynamicBuffer->fBitArray[vectorOfs][byteOfs] &= ~mask;
444  fDynamicBuffer->fBitArray[vectorOfs][byteOfs] |= mask;
445  }

464  XMLSize_t hashCode() const
465  {
466  XMLSize_t hash = 0;
467  if(fDynamicBuffer==0)
468  {
469  for (XMLSize_t index = 0; index<CMSTATE_CACHED_INT32_SIZE; index++)
470  hash = fBits[index] + hash * 31;
471  }
472  else
473  {
474  for (XMLSize_t index = 0; index<fDynamicBuffer->fArraySize; index++)
475  {
476  if(fDynamicBuffer->fBitArray[index]==NULL)
477  // simulates the iteration on the missing bits
478  for(XMLSize_t subIndex=0;subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex++)
479  hash *= 31;
480  else
481  for(XMLSize_t subIndex=0;subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex++)
482  hash = fDynamicBuffer->fBitArray[index][subIndex] + hash * 31;
483  }
484  }
485  return hash;

542 class CMStateSetEnumerator : public XMemory
543 {
544 public:
545  CMStateSetEnumerator(const CMStateSet* toEnum, XMLSize_t start = 0) :
546  fToEnum(toEnum),
547  fIndexCount((XMLSize_t)-1),
548  fLastValue(0)
549  {
550  // if a starting bit is specified, place fIndexCount at the beginning of the previous 32 bit area
551  // so the findNext moves to the one where 'start' is located
552  if(start > 32)
553  fIndexCount = (start/32 - 1) * 32;
554  findNext();
555  // if we found data, and fIndexCount is still pointing to the area where 'start' is located, erase the bits before 'start'
556  if(hasMoreElements() && fIndexCount < start)
557  {
558  for(XMLSize_t i=0;i< (start - fIndexCount);i++)
559  {
560  XMLInt32 mask=1UL << i;
561  if(fLastValue & mask)
562  fLastValue &= ~mask;
563  }
564  // in case the 32 bit area contained only bits before 'start', advance
565  if(fLastValue==0)
566  findNext();
567  }
568  }
569 
570  bool hasMoreElements()
571  {
572  return fLastValue!=0;
573  }
574 
575  unsigned int nextElement()
576  {
577  for(int i=0;i<32;i++)
578  {
579  XMLInt32 mask=1UL << i;
580  if(fLastValue & mask)
581  {
582  fLastValue &= ~mask;
583  unsigned int retVal=(unsigned int)fIndexCount+i;
584  if(fLastValue==0)
585  findNext();
586  return retVal;
587  }
588  }
589  return 0;
590  }
591 
592 private:
593  void findNext()
594  {
595  if(fToEnum->fDynamicBuffer==0)
596  {
597  XMLSize_t nOffset=((fIndexCount==(XMLSize_t)-1)?0:(fIndexCount/32)+1);
598  for(XMLSize_t index=nOffset;index<CMSTATE_CACHED_INT32_SIZE;index++)
599  {
600  if(fToEnum->fBits[index]!=0)
601  {
602  fIndexCount=index*32;
603  fLastValue=fToEnum->fBits[index];
604  return;
605  }
606  }
607  }
608  else
609  {
610  XMLSize_t nOffset=((fIndexCount==(XMLSize_t)-1)?0:(fIndexCount/CMSTATE_BITFIELD_CHUNK));
611  XMLSize_t nSubOffset=((fIndexCount==(XMLSize_t)-1)?0:((fIndexCount % CMSTATE_BITFIELD_CHUNK) /32)+1);
612  for (XMLSize_t index = nOffset; index<fToEnum->fDynamicBuffer->fArraySize; index++)
613  {
614  if(fToEnum->fDynamicBuffer->fBitArray[index]!=NULL)
615  {
616  for(XMLSize_t subIndex=nSubOffset;subIndex < CMSTATE_BITFIELD_INT32_SIZE; subIndex++)
617  if(fToEnum->fDynamicBuffer->fBitArray[index][subIndex]!=0)
618  {
619  fIndexCount=index*CMSTATE_BITFIELD_CHUNK + subIndex*32;
620  fLastValue=fToEnum->fDynamicBuffer->fBitArray[index][subIndex];
621  return;
622  }
623  }
624  nSubOffset = 0; // next chunks will be processed from the beginning
625  }
626  }
627  }
628 
629  const CMStateSet* fToEnum;
630  XMLSize_t fIndexCount;
631  XMLInt32 fLastValue;
632 };
