92 inline void noisyFailureMsgHandler(QtMsgType type, const QMessageLogContext &context, const QString &msg)
93 {
94  Q_UNUSED(context)
95 
96  // Why on earth didn't Qt want to make failed signal/slot connections qWarning?
97  if ((type == QtDebugMsg) && msg.contains(QStringLiteral("::connect")))
98  {
99  type = QtWarningMsg;
100  }
101 
102 #if defined(V_NO_ASSERT)
103  // I have decided to hide this annoing message for release builds.
104  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QSslSocket: cannot resolve")))
105  {
106  type = QtDebugMsg;
107  }
108 
109  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("setGeometry: Unable to set geometry")))
110  {
111  type = QtDebugMsg;
112  }
113 #endif //defined(V_NO_ASSERT)
114 
115 #if defined(Q_OS_MAC)
116  // Hide Qt bug 'Assertion when reading an icns file'
117  // https://bugreports.qt.io/browse/QTBUG-45537
118  // Remove after Qt fix will be released
119  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QICNSHandler::read()")))
120  {
121  type = QtDebugMsg;
122  }
123 
124  // See issue #568
125  if (msg.contains(QStringLiteral("Error receiving trust for a CA certificate")))
126  {
127  type = QtDebugMsg;
128  }
129 #endif
130 
131  // this is another one that doesn't make sense as just a debug message. pretty serious
132  // sign of a problem
133  // http://www.developer.nokia.com/Community/Wiki/QPainter::begin:Paint_device_returned_engine_%3D%3D_0_(Known_Issue)
134  if ((type == QtDebugMsg) && msg.contains(QStringLiteral("QPainter::begin"))
135  && msg.contains(QStringLiteral("Paint device returned engine")))
136  {
137  type = QtWarningMsg;
138  }
139 
140  // This qWarning about "Cowardly refusing to send clipboard message to hung application..."
141  // is something that can easily happen if you are debugging and the application is paused.
142  // As it is so common, not worth popping up a dialog.
143  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QClipboard::event"))
144  && msg.contains(QStringLiteral("Cowardly refusing")))
145  {
146  type = QtDebugMsg;
147  }
148 
149  // only the GUI thread should display message boxes. If you are
150  // writing a multithreaded application and the error happens on
151  // a non-GUI thread, you'll have to queue the message to the GUI
152  QCoreApplication *instance = QCoreApplication::instance();
153  const bool isGuiThread = instance && (QThread::currentThread() == instance->thread());
154 
155  switch (type)
156  {
157  case QtDebugMsg:
158  vStdOut() << QApplication::translate("mNoisyHandler", "DEBUG:") << msg << "\n";
159  return;
160  case QtWarningMsg:
161  vStdErr() << QApplication::translate("mNoisyHandler", "WARNING:") << msg << "\n";
162  break;
163  case QtCriticalMsg:
164  vStdErr() << QApplication::translate("mNoisyHandler", "CRITICAL:") << msg << "\n";
165  break;
166  case QtFatalMsg:
167  vStdErr() << QApplication::translate("mNoisyHandler", "FATAL:") << msg << "\n";
168  break;
169  #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
170  case QtInfoMsg:
171  vStdOut() << QApplication::translate("mNoisyHandler", "INFO:") << msg << "\n";
172  break;
173  #endif
174  default:
175  break;
176  }
177 
178  if (isGuiThread)
179  {
180  //fixme: trying to make sure there are no save/load dialogs are opened, because error message during them will
181  //lead to crash
182  const bool topWinAllowsPop = (QApplication::activeModalWidget() == nullptr) ||
183  !QApplication::activeModalWidget()->inherits("QFileDialog");
184  QMessageBox messageBox;
185 
186  switch (type)
187  {
188  case QtWarningMsg:
189  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Warning"));
190  messageBox.setIcon(QMessageBox::Warning);
191  break;
192  case QtCriticalMsg:
193  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Critical Error"));
194  messageBox.setIcon(QMessageBox::Critical);
195  break;
196  case QtFatalMsg:
197  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Fatal Error"));
198  messageBox.setIcon(QMessageBox::Critical);
199  break;
200  #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
201  case QtInfoMsg:
202  messageBox.setWindowTitle(QApplication::translate("mNoisyHandler", "Information"));
203  messageBox.setIcon(QMessageBox::Information);
204  break;
205  #endif
206  case QtDebugMsg:
207  Q_UNREACHABLE(); //-V501
208  break;
209  default:
210  break;
211  }
212 
213  if (type == QtWarningMsg || type == QtCriticalMsg || type == QtFatalMsg)
214  {
215  if (not qApp->IsTestMode())
216  {
217  if (topWinAllowsPop)
218  {
219  messageBox.setText(msg);
220  messageBox.setStandardButtons(QMessageBox::Ok);
221  messageBox.setWindowModality(Qt::ApplicationModal);
222  messageBox.setModal(true);
223  #ifndef QT_NO_CURSOR
224  QGuiApplication::setOverrideCursor(Qt::ArrowCursor);
225  #endif
226  messageBox.setWindowFlags(messageBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
227  messageBox.exec();
228  #ifndef QT_NO_CURSOR
229  QGuiApplication::restoreOverrideCursor();
230  #endif
231  }
232  }
233  }
234 
235  if (QtFatalMsg == type)
236  {
237  abort();
238  }
239  }
240  else
241  {
242  if (type != QtDebugMsg && type != QtWarningMsg)
243  {
244  abort(); // be NOISY unless overridden!
245  }
246  }
247 }

287 /**
288  * @brief notify Reimplemented from QApplication::notify().
289  * @param receiver receiver.
290  * @param event event.
291  * @return value that is returned from the receiver's event handler.
292  */
293 // reimplemented from QApplication so we can throw exceptions in slots
294 bool MApplication::notify(QObject *receiver, QEvent *event)
295 {
296  try
297  {
298  return QApplication::notify(receiver, event);
299  }
300  catch (const VExceptionObjectError &error)
301  {
302  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error parsing file. Program will be terminated.")), //-V807
303  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
304  exit(V_EX_DATAERR);
305  }
306  catch (const VExceptionBadId &error)
307  {
308  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error bad id. Program will be terminated.")),
309  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
310  exit(V_EX_DATAERR);
311  }
312  catch (const VExceptionConversionError &error)
313  {
314  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error can't convert value. Program will be terminated.")),
315  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
316  exit(V_EX_DATAERR);
317  }
318  catch (const VExceptionEmptyParameter &error)
319  {
320  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error empty parameter. Program will be terminated.")),
321  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
322  exit(V_EX_DATAERR);
323  }
324  catch (const VExceptionWrongId &error)
325  {
326  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error wrong id. Program will be terminated.")),
327  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
328  exit(V_EX_DATAERR);
329  }
330  catch (const VExceptionToolWasDeleted &error)
331  {
332  qCCritical(mApp, "%s\n\n%s\n\n%s",
333  qUtf8Printable("Unhadled deleting tool. Continue use object after deleting!"),
334  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
335  exit(V_EX_DATAERR);
336  }
337  catch (const VException &error)
338  {
339  qCCritical(mApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Something's wrong!!")),
340  qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
341  return true;
342  }
343  // These last two cases special. I found that we can't show here modal dialog with error message.
344  // Somehow program doesn't waite untile an error dialog will be closed. But if ignore this program will hang.
345  catch (const qmu::QmuParserError &error)
346  {
347  qCCritical(mApp, "%s", qUtf8Printable(tr("Parser error: %1. Program will be terminated.").arg(error.GetMsg())));
348  exit(V_EX_DATAERR);
349  }
350  catch (std::exception &error)
351  {
352  qCCritical(mApp, "%s", qUtf8Printable(tr("Exception thrown: %1. Program will be terminated.").arg(error.what())));
353  exit(V_EX_SOFTWARE);
354  }
355  return false;
356 }

448 bool MApplication::event(QEvent *event)
449 {
450  switch(event->type())
451  {
452  // In Mac OS X the QFileOpenEvent event is generated when user perform "Open With" from Finder (this event is
453  // Mac specific).
454  case QEvent::FileOpen:
455  {
456  QFileOpenEvent *fileOpenEvent = static_cast<QFileOpenEvent *>(event);
457  const QString macFileOpen = fileOpenEvent->file();
458  if(not macFileOpen.isEmpty())
459  {
460  TMainWindow *mw = MainWindow();
461  if (mw)
462  {
463  mw->LoadFile(macFileOpen); // open file in existing window
464  }
465  return true;
466  }
467  break;
468  }
469 #if defined(Q_OS_MAC)
470  case QEvent::ApplicationActivate:
471  {
472  Clean();
473  TMainWindow *mw = MainWindow();
474  if (mw && not mw->isMinimized())
475  {
476  mw->show();
477  }
478  return true;
479  }
480 #endif //defined(Q_OS_MAC)
481  default:
482  return VAbstractApplication::event(event);
483  }
484  return VAbstractApplication::event(event);
485 }

536 //---------------------------------------------------------------------------------------------------------------------
537 void MApplication::ParseCommandLine(const SocketConnection &connection, const QStringList &arguments)
538 {
539  QCommandLineParser parser;
540  parser.setApplicationDescription(tr("Seamly2D's measurements editor."));
541  parser.addHelpOption();
542  parser.addVersionOption();
543  parser.addPositionalArgument("filename", tr("The measurement file."));
544  //-----
545  QCommandLineOption heightOption(QStringList() << "e" << "height",
546  tr("Open with the base height. Valid values: %1cm.")
547  .arg(MeasurementVariable::WholeListHeights(Unit::Cm).join(", ")),
548  tr("The base height"));
549  parser.addOption(heightOption);
550  //-----
551  QCommandLineOption sizeOption(QStringList() << "s" << "size",
552  tr("Open with the base size. Valid values: %1cm.").arg(MeasurementVariable::WholeListSizes(Unit::Cm).join(", ")),
553  tr("The base size"));
554  parser.addOption(sizeOption);
555  //-----
556  QCommandLineOption unitOption(QStringList() << "u" << "unit",
557  tr("Set pattern file unit: cm, mm, inch."),
558  tr("The pattern unit"));
559  parser.addOption(unitOption);
560  //-----
561  QCommandLineOption testOption(QStringList() << "test",
562  tr("Use for unit testing. Run the program and open a file without showing the main window."));
563  parser.addOption(testOption);
564  //-----
565  QCommandLineOption scalingOption(QStringList() << LONG_OPTION_NO_HDPI_SCALING,
566  tr("Disable high dpi scaling. Call this option if has problem with scaling (by default scaling enabled). "
567  "Alternatively you can use the %1 environment variable.").arg("QT_AUTO_SCREEN_SCALE_FACTOR=0"));
568  parser.addOption(scalingOption);
569  //-----
570  parser.process(arguments);
571 
572  bool flagHeight = false;
573  bool flagSize = false;
574  bool flagUnit = false;
575 
576  int size = 0;
577  int height = 0;
578  Unit unit = Unit::Cm;
579 
580  if (parser.isSet(heightOption))
581  {
582  const QString heightValue = parser.value(heightOption);
583  if (MeasurementVariable::IsGradationHeightValid(heightValue))
584  {
585  flagHeight = true;
586  height = heightValue.toInt();
587  }
588  else
589  {
590  qCCritical(mApp, "%s\n",
591  qPrintable(tr("Invalid base height argument. Must be %1cm.")
592  .arg(MeasurementVariable::WholeListHeights(Unit::Cm).join(", "))));
593  parser.showHelp(V_EX_USAGE);
594  }
595  }
596 
597  if (parser.isSet(sizeOption))
598  {
599  const QString sizeValue = parser.value(sizeOption);
600  if (MeasurementVariable::IsGradationSizeValid(sizeValue))
601  {
602  flagSize = true;
603  size = sizeValue.toInt();
604  }
605  else
606  {
607  qCCritical(mApp, "%s\n",
608  qPrintable(tr("Invalid base size argument. Must be %1cm.")
609  .arg(MeasurementVariable::WholeListSizes(Unit::Cm).join(", "))));
610  parser.showHelp(V_EX_USAGE);
611  }
612  }
613 
614  {
615  const QString unitValue = parser.value(unitOption);
616  if (not unitValue.isEmpty())
617  {
618 
619  const QStringList units = QStringList() << unitMM << unitCM << unitINCH;
620  if (units.contains(unitValue))
621  {
622  flagUnit = true;
623  unit = StrToUnits(unitValue);
624  }
625  else
626  {
627  qCCritical(mApp, "%s\n", qPrintable(tr("Invalid base size argument. Must be cm, mm or inch.")));
628  parser.showHelp(V_EX_USAGE);
629  }
630  }
631  }
632 
633  testMode = parser.isSet(testOption);
634 
635  if (not testMode && connection == SocketConnection::Client)
636  {
637  const QString serverName = QCoreApplication::applicationName();
638  QLocalSocket socket;
639  socket.connectToServer(serverName);
640  if (socket.waitForConnected(1000))
641  {
642  qCInfo(mApp, "Connected to the server '%s'", qUtf8Printable(serverName));
643  QTextStream stream(&socket);
644  stream << QCoreApplication::arguments().join(";;");
645  stream.flush();
646  socket.waitForBytesWritten();
647  qApp->exit(V_EX_OK);
648  return;
649  }
650 
651  qCDebug(mApp, "Can't establish connection to the server '%s'", qUtf8Printable(serverName));
652 
653  localServer = new QLocalServer(this);
654  connect(localServer, &QLocalServer::newConnection, this, &MApplication::NewLocalSocketConnection);
655  if (not localServer->listen(serverName))
656  {
657  qCWarning(mApp, "Can't begin to listen for incoming connections on name '%s'",
658  qUtf8Printable(serverName));
659  if (localServer->serverError() == QAbstractSocket::AddressInUseError)
660  {
661  QLocalServer::removeServer(serverName);
662  if (not localServer->listen(serverName))
663  {
664  qCWarning(mApp, "%s",
665  qUtf8Printable(tr("Can't begin to listen for incoming connections on name '%1'").arg(serverName)));
666  }
667  }
668  }
669 
670  //loadTranslations(SeamlyMeSettings()->GetLocale());
671  }
672 
673  const QStringList args = parser.positionalArguments();
674  if (args.count() > 0)
675  {
676  if (testMode && args.count() > 1)
677  {
