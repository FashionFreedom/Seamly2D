197 template <typename T>
198 void VAbstractOperation::ShowToolVisualization(bool show)
199 {
200  if (show)
201  {
202  if (vis.isNull())
203  {
204  AddVisualization<T>();
205  SetVisualization();
206  }
207  else
208  {
209  if (T *visual = qobject_cast<T *>(vis))
210  {
211  visual->show();
212  }
213  }
214  }
215  else
216  {
217  delete vis;
218  }
219 }
