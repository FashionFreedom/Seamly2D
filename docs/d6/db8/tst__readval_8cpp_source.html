156 void TST_ReadVal::TestVal()
157 {
158  QFETCH(QString, formula);
159  QFETCH(int, expCount);
160  QFETCH(qreal, expVal);
161  QFETCH(QLocale, locale);
162 
163  qreal resVal = 0;
164  QLocale::setDefault(locale);
165 
166  const int resCount = ReadVal(formula, resVal, locale, locale.decimalPoint(), locale.groupSeparator());
167 
168  QString errorMsg = QString("Conversion failed. Locale: '%1'.").arg(locale.name());
169  QVERIFY2(resCount == expCount, qUtf8Printable(errorMsg));
170 
171  if (resCount != -1)
172  {
173  QString errorMsg = QString("Unexpected result. Locale: '%1'.").arg(locale.name());
174  QVERIFY2(QmuFuzzyComparePossibleNulls(resVal, expVal), qUtf8Printable(errorMsg));
175  }
