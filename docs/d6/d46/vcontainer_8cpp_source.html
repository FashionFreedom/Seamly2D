517 /**
518  * @brief Clears all geometric objects marked for calculation from the container.
519  *
520  * This method removes all geometric objects that are marked with the mode `Draw::Calculation` from the container.
521  *
522  * @details
523  * - The method first checks if the `gObjects` hash map is not empty.
524  * - It iterates over the `gObjects` hash map to identify objects marked with `Draw::Calculation` mode.
525  * - Identified objects are cleared and their keys are collected in a vector.
526  * - After the iteration, the method removes the objects with the collected keys from the hash map.
527  * - This two-step process ensures that the iterator is not invalidated during the removal of objects.
528  */
529 void VContainer::ClearCalculationGObjects()
530 {
531  if (not d->gObjects.isEmpty()) //-V807
532  {
533  QVector<quint32> keys;
534  QHash<quint32, QSharedPointer<VGObject> >::iterator i;
535  for (i = d->gObjects.begin(); i != d->gObjects.end(); ++i)
536  {
537  if (i.value()->getMode() == Draw::Calculation)
538  {
539  i.value().clear();
540  keys.append(i.key());
541  }
542  }
543  // We can't delete objects in previous loop it will destroy the iterator.
544  if (not keys.isEmpty())
545  {
546  for (int i = 0; i < keys.size(); ++i)
547  {
548  d->gObjects.remove(keys.at(i));
549  }
550  }
551  }
552 }

554 /**
555  * @brief Clears variables of the specified type from the container.
556  *
557  * This method removes variables of the specified type from the container. If the type is `VarType::Unknown`, all variables are cleared.
558  *
559  * @param type The type of variables to clear.
560  *
561  * @details
562  * - The method first checks if the `variables` hash map is not empty.
563  * - If the specified type is `VarType::Unknown`, it clears the entire `variables` hash map.
564  * - Otherwise, it iterates over the `variables` hash map to identify variables of the specified type.
565  * - Identified variables' keys are collected in a vector.
566  * - After the iteration, the method removes the variables with the collected keys from the hash map.
567  * - This two-step process ensures that the iterator is not invalidated during the removal of variables.
568  */
569 
570 void VContainer::ClearVariables(const VarType &type)
571 {
572  if (d->variables.size()>0) //-V807
573  {
574  if (type == VarType::Unknown)
575  {
576  d->variables.clear();
577  }
578  else
579  {
580  QVector<QString> keys;
581  QHash<QString, QSharedPointer<VInternalVariable> >::iterator i;
582  for (i = d->variables.begin(); i != d->variables.end(); ++i)
583  {
584  if (i.value()->GetType() == type)
585  {
586  keys.append(i.key());
587  }
588  }
589 
590  for (int i = 0; i < keys.size(); ++i)
591  {
592  d->variables.remove(keys.at(i));
593  }
594  }
595  }
596 }

631 void VContainer::AddArc(const QSharedPointer<VAbstractCurve> &arc, const quint32 &id, const quint32 &parentId)
632 {
633  AddCurve(arc, id, parentId);
634 
635  if (arc->getType() == GOType::Arc)
636  {
637  const QSharedPointer<VArc> casted = arc.staticCast<VArc>();
638 
639  VArcRadius *radius = new VArcRadius(id, parentId, casted.data(), *GetPatternUnit());
640  AddVariable(radius->GetName(), radius);
641  }
642  else if (arc->getType() == GOType::EllipticalArc)
643  {
644  const QSharedPointer<VEllipticalArc> casted = arc.staticCast<VEllipticalArc>();
645 
646  VArcRadius *radius1 = new VArcRadius(id, parentId, casted.data(), 1, *GetPatternUnit());
647  AddVariable(radius1->GetName(), radius1);
648
