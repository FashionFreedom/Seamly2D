394 /**
395  * @brief Generates and returns the next unique ID.
396  *
397  * This method increments the current unique ID and returns the new value.
398  *
399  * @return quint32 The next unique ID.
400  *
401  * @details
402  * - The method first checks if the current ID has reached the maximum value (`UINT_MAX`).
403  * - If the maximum value is reached, it logs a critical message indicating that the number of free IDs is exhausted.
404  * - It increments the static member variable `_id` to generate the next unique ID.
405  * - The method returns the new value of `_id`.
406  *
407  * @note Currently, the method simply increments the ID without reusing free IDs.
408  */
409 quint32 VContainer::getNextId()
410 {
411  //Because UINT_MAX is so big, it's impossible to run out of ids
412  //We don't try to reuse free ids to avoid the complexity of tracking them
413  if (_id == UINT_MAX)
414  {
415  qCritical() << (tr("Number of free id exhausted."));
416  }
417  _id++;
418  return _id;
419 }

515 /**
516  * @brief Clears all geometric objects marked for calculation from the container.
517  *
518  * This method removes all geometric objects that are marked with the mode `Draw::Calculation` from the container.
519  *
520  * @details
521  * - The method first checks if the `gObjects` hash map is not empty.
522  * - It iterates over the `gObjects` hash map to identify objects marked with `Draw::Calculation` mode.
523  * - Identified objects are cleared and their keys are collected in a vector.
524  * - After the iteration, the method removes the objects with the collected keys from the hash map.
525  * - This two-step process ensures that the iterator is not invalidated during the removal of objects.
526  */
527 void VContainer::ClearCalculationGObjects()
528 {
529  if (not d->gObjects.isEmpty()) //-V807
530  {
531  QVector<quint32> keys;
532  QHash<quint32, QSharedPointer<VGObject> >::iterator i;
533  for (i = d->gObjects.begin(); i != d->gObjects.end(); ++i)
534  {
535  if (i.value()->getMode() == Draw::Calculation)
536  {
537  i.value().clear();
538  keys.append(i.key());
539  }
540  }
541  // We can't delete objects in previous loop it will destroy the iterator.
542  if (not keys.isEmpty())
543  {
544  for (int i = 0; i < keys.size(); ++i)
545  {
546  d->gObjects.remove(keys.at(i));
547  }
548  }
549  }
550 }

552 /**
553  * @brief Clears variables of the specified type from the container.
554  *
555  * This method removes variables of the specified type from the container. If the type is `VarType::Unknown`, all variables are cleared.
556  *
557  * @param type The type of variables to clear.
558  *
559  * @details
560  * - The method first checks if the `variables` hash map is not empty.
561  * - If the specified type is `VarType::Unknown`, it clears the entire `variables` hash map.
562  * - Otherwise, it iterates over the `variables` hash map to identify variables of the specified type.
563  * - Identified variables' keys are collected in a vector.
564  * - After the iteration, the method removes the variables with the collected keys from the hash map.
565  * - This two-step process ensures that the iterator is not invalidated during the removal of variables.
566  */
567 void VContainer::ClearVariables(const VarType &type)
568 {
569  if (d->variables.size()>0) //-V807
570  {
571  if (type == VarType::Unknown)
572  {
573  d->variables.clear();
574  }
575  else
576  {
577  QVector<QString> keys;
578  QHash<QString, QSharedPointer<VInternalVariable> >::iterator i;
579  for (i = d->variables.begin(); i != d->variables.end(); ++i)
580  {
581  if (i.value()->GetType() == type)
582  {
583  keys.append(i.key());
584  }
585  }
586 
587  for (int i = 0; i < keys.size(); ++i)
588  {
589  d->variables.remove(keys.at(i));
590  }
591  }
592  }
593 }
