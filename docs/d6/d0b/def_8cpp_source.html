240 {
241  switch (unit)
242  {
243  case Unit::Mm:
244  return (val / 25.4) * PrintDPI;
245  case Unit::Cm:
246  return ((val * 10.0) / 25.4) * PrintDPI;
247  case Unit::Inch:
248  return val * PrintDPI;
249  case Unit::Px:

251  default:
252  break;
253  }
254  return 0;

277 qreal UnitConvertor(qreal value, const Unit &from, const Unit &to)
278 {
279  switch (from)
280  {
281  case Unit::Mm:
282  switch (to)
283  {
284  case Unit::Mm:
285  return value;
286  case Unit::Cm:
287  return value / 10.0;
288  case Unit::Inch:
289  return value / 25.4;
290  case Unit::Px:
291  return (value / 25.4) * PrintDPI;
292  default:
293  break;
294  }
295  break;
296  case Unit::Cm:
297  switch (to)
298  {
299  case Unit::Mm:
300  return value * 10.0;
301  case Unit::Cm:
302  return value;
303  case Unit::Inch:
304  return value / 2.54;
305  case Unit::Px:
306  return ((value * 10.0) / 25.4) * PrintDPI;
307  default:
308  break;
309  }
310  break;
311  case Unit::Inch:
312  switch (to)
313  {
314  case Unit::Mm:
315  return value * 25.4;
316  case Unit::Cm:
317  return value * 2.54;
318  case Unit::Inch:
319  return value;
320  case Unit::Px:
321  return value * PrintDPI;
322  default:
323  break;
324  }
325  break;
326  case Unit::Px:
327  switch (to)
328  {
329  case Unit::Mm:
330  return (value / PrintDPI) * 25.4;
331  case Unit::Cm:
332  return ((value / PrintDPI) * 25.4) / 10.0;
333  case Unit::Inch:
334  return value / PrintDPI;
335  case Unit::Px:
336  return value;
337  default:
338  break;
339  }
340  break;
341  default:
342  break;
343  }
344  return 0;
345 }

536 QPixmap darkenPixmap(const QPixmap &pixmap)
537 {
538  QImage img = pixmap.toImage().convertToFormat(QImage::Format_ARGB32);
539  const int imgh = img.height();
540  const int imgw = img.width();
541  for (int y = 0; y < imgh; ++y)
542  {
543  for (int x = 0; x < imgw; ++x)
544  {
545  int h, s, v;
546  QRgb pixel = img.pixel(x, y);
547  const int a = qAlpha(pixel);
548  QColor hsvColor(pixel);
549  hsvColor.getHsv(&h, &s, &v);
550  s = qMin(100, s * 2);
551  v = v / 2;
552  hsvColor.setHsv(h, s, v);
553  pixel = hsvColor.rgb();
554  img.setPixel(x, y, qRgba(qRed(pixel), qGreen(pixel), qBlue(pixel), a));
555  }
556  }
557  return QPixmap::fromImage(img);
558 }
