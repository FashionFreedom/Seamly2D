173 int TST_QmuParserErrorMsg::loadTranslations(const QString &checkedLocale)
174 {
175  const QString path = TranslationsPath();
176  const QString file = QString("seamly2d_%1.qm").arg(checkedLocale);
177 
178  if (QFileInfo(path+QLatin1String("/")+file).size() <= 34)
179  {
180  const QString message = QString("Translation for locale = %1 is empty. \nFull path: %2/%3")
181  .arg(checkedLocale)
182  .arg(path)
183  .arg(file);
184  QWARN(qUtf8Printable(message));
185 
186  return ErrorSize;
187  }
188 
189  appTranslator = new QTranslator(this);
190 
191  if (not appTranslator->load(file, path))
192  {
193  const QString message = QString("Can't load translation for locale = %1. \nFull path: %2/%3")
194  .arg(checkedLocale)
195  .arg(path)
196  .arg(file);
197  QWARN(qUtf8Printable(message));
198 
199  delete appTranslator;
200 
201  return ErrorLoad;
202  }
203 
204  if (not QCoreApplication::installTranslator(appTranslator))
205  {
206  const QString message = QString("Can't install translation for locale = %1. \nFull path: %2/%3")
207  .arg(checkedLocale)
208  .arg(path)
209  .arg(file);
210  QWARN(qUtf8Printable(message));
211 
212  delete appTranslator;
213 
214  return ErrorInstall;
215  }
216 
217  delete msg;
218  msg = new qmu::QmuParserErrorMsg();//Very important do it after load QM file.
219 
220  return NoError;
221 }
