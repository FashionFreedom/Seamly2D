109 qreal VToolBisector::BisectorAngle(const QPointF &firstPoint, const QPointF &secondPoint, const QPointF &thirdPoint)
110 {
111  QLineF line1(secondPoint, firstPoint);
112  QLineF line2(secondPoint, thirdPoint);
113  qreal angle = line1.angleTo(line2);
114  if (angle>180)
115  {
116  angle = 360 - angle;
117  return line1.angle()-angle/2;
118  }
119  else
120  {
121  return line1.angle()+angle/2;
122  }
123 }

127  * @brief FindPoint find bisector point.
128  * @param firstPoint first point of angle.
129  * @param secondPoint second point of angle.
130  * @param thirdPoint third point of angle.
131  * @param length bisector length.
132  * @return bisector point.
133  */
134 QPointF VToolBisector::FindPoint(const QPointF &firstPoint, const QPointF &secondPoint,
135  const QPointF &thirdPoint, const qreal &length)
136 {
137  QLineF line1(secondPoint, firstPoint);
138  line1.setAngle(BisectorAngle(firstPoint, secondPoint, thirdPoint));
139  line1.setLength(length);
140  return line1.p2();
141 }

195 /**
196  * @brief Create help create tool.
197  * @param _id tool id, 0 if tool doesn't exist yet.
198  * @param formula string with formula.
199  * @param firstPointId id first point of angle.
200  * @param secondPointId id second point of angle.
201  * @param thirdPointId id third point of angle.
202  * @param lineType line type.
203  * @param lineWeight line weight.
204  * @param lineColor line color.
205  * @param pointName point name.
206  * @param mx label bias x axis.
207  * @param my label bias y axis.
208  * @param showPointName show/hide point name text.
209  * @param scene pointer to scene.
210  * @param doc dom document container.
211  * @param data container with variables.
212  * @param parse parser file mode.
213  * @param typeCreation way we create this tool.
214  */
215 VToolBisector* VToolBisector::Create(const quint32 _id, QString &formula, quint32 firstPointId, quint32 secondPointId,
216  quint32 thirdPointId, const QString &lineType,
217  const QString &lineWeight, const QString &lineColor,
218  const QString &pointName, qreal mx, qreal my, bool showPointName,
219  VMainGraphicsScene *scene, VAbstractPattern *doc,
220  VContainer *data, const Document &parse, const Source &typeCreation)
221 {
222  const QSharedPointer<VPointF> firstPoint = data->GeometricObject<VPointF>(firstPointId);
223  const QSharedPointer<VPointF> secondPoint = data->GeometricObject<VPointF>(secondPointId);
224  const QSharedPointer<VPointF> thirdPoint = data->GeometricObject<VPointF>(thirdPointId);
225 
226  const qreal result = CheckFormula(_id, formula, data);
227 
228  QPointF fPoint = VToolBisector::FindPoint(static_cast<QPointF>(*firstPoint), static_cast<QPointF>(*secondPoint),
229  static_cast<QPointF>(*thirdPoint), qApp->toPixel(result));
230  quint32 id = _id;
231  VPointF *p = new VPointF(fPoint, pointName, mx, my);
232  p->setShowPointName(showPointName);
233 
234  if (typeCreation == Source::FromGui)
235  {
236  id = data->AddGObject(p);
237  data->AddLine(secondPointId, id);
238  }
239  else
240  {
241  data->UpdateGObject(id, p);
242  data->AddLine(secondPointId, id);
243  if (parse != Document::FullParse)
244  {
245  doc->UpdateToolData(id, data);
246  }
247  }
248 
249  if (parse == Document::FullParse)
250  {
251  VDrawTool::AddRecord(id, Tool::Bisector, doc);
252  VToolBisector *point = new VToolBisector(doc, data, id, lineType, lineWeight, lineColor, formula, firstPointId,
253  secondPointId, thirdPointId, typeCreation);
254  scene->addItem(point);
255  InitToolConnections(scene, point);
256  VAbstractPattern::AddTool(id, point);
257  doc->IncrementReferens(firstPoint->getIdTool());
258  doc->IncrementReferens(secondPoint->getIdTool());
259  doc->IncrementReferens(thirdPoint->getIdTool());
260  return point;
261  }
262  return nullptr;
263 }
