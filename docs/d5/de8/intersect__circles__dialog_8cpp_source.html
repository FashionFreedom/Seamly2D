227 //---------------------------------------------------------------------------------------------------------------------
228 void IntersectCirclesDialog::ChosenObject(quint32 id, const SceneObject &type)
229 {
230  if (prepare == false)// After first choose we ignore all objects
231  {
232  if (type == SceneObject::Point)
233  {
234  IntersectCirclesVisual *point = qobject_cast<IntersectCirclesVisual *>(vis);
235  SCASSERT(point != nullptr)
236 
237  switch (number)
238  {
239  case 0:
240  if (SetObject(id, ui->comboBoxCircle1Center, tr("Select second circle center")))
241  {
242  number++;
243  point->VisualMode(id);
244  }
245  break;
246  case 1:
247  if (getCurrentObjectId(ui->comboBoxCircle1Center) != id)
248  {
249  if (SetObject(id, ui->comboBoxCircle2Center, ""))
250  {
251  number = 0;
252  point->setObject2Id(id);
253  point->RefreshGeometry();
254  prepare = true;
255  this->setModal(true);
256  this->show();
257  }
258  }
259  break;
260  default:
261  break;
262  }
263  }
264  }
265 }
