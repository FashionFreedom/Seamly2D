138 VToolPointFromArcAndTangent *VToolPointFromArcAndTangent::Create(const quint32 _id,
139  const QString &pointName,
140  quint32 arcId,
141  quint32 tangentPointId,
142  CrossCirclesPoint crossPoint,
143  qreal mx, qreal my,
144  bool showPointName,
145  VMainGraphicsScene *scene,
146  VAbstractPattern *doc,
147  VContainer *data,
148  const Document &parse,
149  const Source &typeCreation)
150 {
151  const VArc arc = *data->GeometricObject<VArc>(arcId);
152  const VPointF tPoint = *data->GeometricObject<VPointF>(tangentPointId);
153 
154  const QPointF point = VToolPointFromArcAndTangent::FindPoint(static_cast<QPointF>(tPoint), &arc, crossPoint);
155 
156  if (point == QPointF())
157  {
158  const QString msg = tr("<b><big>Can't find intersection point %1 of</big></b><br>"
159  "<b><big>%2 and Tangent</big></b><br><br>"
160  "Using origin point as a place holder until pattern is corrected.")
161  .arg(pointName)
162  .arg(arc.name());
163 
164  QMessageBox msgBox(qApp->getMainWindow());
165  msgBox.setWindowTitle(tr("Intersect Arc and Tangent"));
166  msgBox.setWindowFlags(msgBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
167  msgBox.setWindowIcon(QIcon(":/toolicon/32x32/point_from_arc_and_tangent.png"));
168  msgBox.setIcon(QMessageBox::Warning);
169  msgBox.setText(msg);
170  msgBox.setStandardButtons(QMessageBox::Ok);
171  msgBox.exec();
172  }
173 
174  quint32 id = _id;
175 
176  VPointF *p = new VPointF(point, pointName, mx, my);
177  p->setShowPointName(showPointName);
178 
179  if (typeCreation == Source::FromGui)
180  {
181  id = data->AddGObject(p);
182  }
183  else
184  {
185  data->UpdateGObject(id, p);
186  if (parse != Document::FullParse)
187  {
188  doc->UpdateToolData(id, data);
189  }
190  }
191 
192  if (parse == Document::FullParse)
193  {
194  VDrawTool::AddRecord(id, Tool::PointFromArcAndTangent, doc);
195  VToolPointFromArcAndTangent *point = new VToolPointFromArcAndTangent(doc, data, id, arcId, tangentPointId,
196  crossPoint, typeCreation);
197 
198  scene->addItem(point);
199  InitToolConnections(scene, point);
200  VAbstractPattern::AddTool(id, point);
201  doc->IncrementReferens(arc.getIdTool());
202  doc->IncrementReferens(tPoint.getIdTool());
203  return point;
204  }
205  return nullptr;
206 }

209 QPointF VToolPointFromArcAndTangent::FindPoint(const QPointF &p, const VArc *arc, const CrossCirclesPoint pType)
210 {
211  QPointF p1, p2;
212  const QPointF center = static_cast<QPointF>(arc->GetCenter());
213  const qreal radius = arc->GetRadius();
214  const int res = VGObject::ContactPoints (p, center, radius, p1, p2);
215 
216  QLineF r1Arc(center, p1);
217  r1Arc.setLength(radius+10);
218 
219  QLineF r2Arc(center, p2);
220  r2Arc.setLength(radius+10);
221 
222  switch(res)
223  {
224  case 2:
225  {
226  int localRes = 0;
227  bool flagP1 = false;
228 
229  if (arc->IsIntersectLine(r1Arc))
230  {
231  ++localRes;
232  flagP1 = true;
233  }
234 
235  if (arc->IsIntersectLine(r2Arc))
236  {
237  ++localRes;
238  }
239 
240  switch(localRes)
241  {
242  case 2:
243  if (pType == CrossCirclesPoint::FirstPoint)
244  {
245  return p1;
246  }
247  else
248  {
249  return p2;
250  }
251  case 1:
252  if (flagP1)
253  {
254  return p1;
255  }
256  else
257  {
258  return p2;
259  }
260  case 0:
261  default:
262  return QPointF();
263  }
264 
265  break;
266  }
267  case 1:
268  if (arc->IsIntersectLine(r1Arc))
269  {
270  return p1;
271  }
272  else
273  {
274  return QPointF();
275  }
276  case 3:
277  case 0:
278  default:
279  break;
280  }
281  return QPointF();
282 }
