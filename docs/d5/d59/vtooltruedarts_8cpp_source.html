102 void VToolTrueDarts::FindPoint(const QPointF &baseLineP1, const QPointF &baseLineP2, const QPointF &dartP1,
103  const QPointF &dartP2, const QPointF &dartP3, QPointF &p1, QPointF &p2)
104 {
105  const QLineF d2d1(dartP2, dartP1);
106  const QLineF d2d3(dartP2, dartP3);
107 
108  const qreal degrees = d2d3.angleTo(d2d1);
109 
110  QLineF d2blP2(dartP2, baseLineP2);
111  d2blP2.setAngle(d2blP2.angle()+degrees);
112 
113  if (QLineF(baseLineP1, d2blP2.p2()).intersects(d2d1, &p1) == QLineF::NoIntersection)
114  {
115  p1 = QPointF(0, 0);
116  p2 = QPointF(0, 0);
117  return;
118  }
119 
120  QLineF d2p1(dartP2, p1);
121  d2p1.setAngle(d2p1.angle()-degrees);
122  p2 = d2p1.p2();
123 }

171 VToolTrueDarts *VToolTrueDarts::Create(quint32 _id,
172  const quint32 &_p1id, const quint32 &_p2id,
173  const quint32 &baseLineP1Id,
174  const quint32 &baseLineP2Id,
175  const quint32 &dartP1Id,
176  const quint32 &dartP2Id,
177  const quint32 &dartP3Id,
178  const QString &point1Name,
179  qreal mx1, qreal my1, bool showPointName1,
180  const QString &point2Name,
181  qreal mx2, qreal my2, bool showPointName2,
182  VMainGraphicsScene *scene, VAbstractPattern *doc, VContainer *data,
183  const Document &parse, const Source &typeCreation)
184 {
185  const QSharedPointer<VPointF> baseLineP1 = data->GeometricObject<VPointF>(baseLineP1Id);
186  const QSharedPointer<VPointF> baseLineP2 = data->GeometricObject<VPointF>(baseLineP2Id);
187  const QSharedPointer<VPointF> dartP1 = data->GeometricObject<VPointF>(dartP1Id);
188  const QSharedPointer<VPointF> dartP2 = data->GeometricObject<VPointF>(dartP2Id);
189  const QSharedPointer<VPointF> dartP3 = data->GeometricObject<VPointF>(dartP3Id);
190 
191  QPointF fPoint1;
192  QPointF fPoint2;
193  VToolTrueDarts::FindPoint(static_cast<QPointF>(*baseLineP1), static_cast<QPointF>(*baseLineP2),
194  static_cast<QPointF>(*dartP1), static_cast<QPointF>(*dartP2),
195  static_cast<QPointF>(*dartP3), fPoint1, fPoint2);
196 
197  quint32 id = _id;
198  quint32 p1id = _p1id;
199  quint32 p2id = _p2id;
200 
201  VPointF *p1 = new VPointF(fPoint1, point1Name, mx1, my1, id);
202  p1->setShowPointName(showPointName1);
203 
204  VPointF *p2 = new VPointF(fPoint2, point2Name, mx2, my2, id);
205  p2->setShowPointName(showPointName2);
206 
207 
208  if (typeCreation == Source::FromGui)
209  {
210  id = VContainer::getNextId(); //Just reserve id for tool
211  p1id = data->AddGObject(p1);
212  p2id = data->AddGObject(p2);
213  }
214  else
215  {
216  data->UpdateGObject(p1id, p1);
217  data->UpdateGObject(p2id, p2);
218  if (parse != Document::FullParse)
219  {
220  doc->UpdateToolData(id, data);
221  }
222  }
223 
224  if (parse == Document::FullParse)
225  {
226  VDrawTool::AddRecord(id, Tool::TrueDarts, doc);
227  VToolTrueDarts *points = new VToolTrueDarts(doc, data, id, p1id, p2id, baseLineP1Id, baseLineP2Id,
228  dartP1Id, dartP2Id, dartP3Id, typeCreation);
229  scene->addItem(points);
230  InitToolConnections(scene, points);
231  VAbstractPattern::AddTool(id, points);
232  doc->IncrementReferens(baseLineP1->getIdTool());
233  doc->IncrementReferens(baseLineP2->getIdTool());
234  doc->IncrementReferens(dartP1->getIdTool());
235  doc->IncrementReferens(dartP2->getIdTool());
236  doc->IncrementReferens(dartP3->getIdTool());
237  return points;
238  }
239  return nullptr;
240 }
