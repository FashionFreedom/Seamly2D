101 void VAbstractNode::incrementReferens()
102 {
103  VAbstractTool::incrementReferens();
104  if (_referens == 1)
105  {
106  if (idTool != NULL_ID)
107  {
108  doc->IncrementReferens(idTool);
109  }
110  else
111  {
112  const QSharedPointer<VGObject> node = VAbstractTool::data.GetGObject(idNode);
113  doc->IncrementReferens(node->getIdTool());
114  }
115  ShowNode();
116  QDomElement domElement = doc->elementById(m_id, getTagName());
117  if (domElement.isElement())
118  {
119  doc->SetParametrUsage(domElement, AttrInUse, NodeUsage::InUse);
120  }
121  }
122 }
