99 QString IndividualSizeConverter::getSchema(int ver) const
100 {
101  switch (ver)
102  {
103  case (0x000200):
104  return QStringLiteral("://schema/individual_size_measurements/v0.2.0.xsd");
105  case (0x000300):
106  return QStringLiteral("://schema/individual_size_measurements/v0.3.0.xsd");
107  case (0x000301):
108  return QStringLiteral("://schema/individual_size_measurements/v0.3.1.xsd");
109  case (0x000302):
110  return QStringLiteral("://schema/individual_size_measurements/v0.3.2.xsd");
111  case (0x000303):
112  return QStringLiteral("://schema/individual_size_measurements/v0.3.3.xsd");
113  case (0x000304):
114  return CurrentSchema;
115  default:
116  InvalidVersion(ver);

118  }

218 
219  const QMultiMap<QString, QString> names = OldNamesToNewNames_InV0_3_0();
220  const QList<QString> keys = names.uniqueKeys();
221  for (int i = 0; i < keys.size(); ++i)
222  {
223  qreal resValue = 0;
224 
225  // This has the same effect as a .values(), just isn't as elegant
226  const QList<QString> list = names.values( keys.at(i) );
227  foreach(const QString &val, list )
228  {
229  const QDomNodeList nodeList = this->elementsByTagName(val);
230  if (nodeList.isEmpty())
231  {
232  continue;

234 
235  const qreal value = GetParametrDouble(nodeList.at(0).toElement(), QStringLiteral("value"), "0.0");
236 
237  if (not qFuzzyIsNull(value))
238  {
239  resValue = value;
240  }
241  }
242 
243  bm.appendChild(addMeasurementsVer0_3_0(keys.at(i), resValue));
244  }
245 
246  QDomElement rootElement = this->documentElement();

248  rootElement.replaceChild(bm, listBM.at(0));
