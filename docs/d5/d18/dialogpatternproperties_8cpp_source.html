320 //---------------------------------------------------------------------------------------------------------------------
321 void DialogPatternProperties::CheckStateHeight(int state)
322 {
323  QCheckBox* box = qobject_cast<QCheckBox*>(sender());
324  if (box)
325  {
326  if (state == Qt::Checked)
327  {
328  ++heightsChecked;
329  if (heightsChecked == heightsCount)
330  {
331  ui->checkBoxAllHeights->blockSignals(true);//don't touch anothers checkboxes
332  ui->checkBoxAllHeights->setCheckState(Qt::Checked);
333  heights.insert(GHeights::ALL, true);
334  ui->checkBoxAllHeights->blockSignals(false);
335  }
336  }
337  else if (state == Qt::Unchecked)
338  {
339  if (heightsChecked == heightsCount)
340  {
341  ui->checkBoxAllHeights->blockSignals(true);//don't touch anothers checkboxes
342  ui->checkBoxAllHeights->setCheckState(Qt::Unchecked);
343  heights.insert(GHeights::ALL, false);
344  ui->checkBoxAllHeights->blockSignals(false);
345  }
346  --heightsChecked;
347  }
348 
349  if (data.contains(box))
350  {
351  heights.insert(static_cast<GHeights>(data.value(box)), box->isChecked());
352  }
353 
354  UpdateDefHeight();
355 
356  CheckApplyOk();
357  gradationChanged = true;
358  }
359 }
