251 void SymlinkCopyDirRecursive(const QString &fromDir, const QString &toDir, bool replaceOnConflit)
252 {
253  QDir dir;
254  dir.setPath(fromDir);
255 
256  foreach (QString copyFile, dir.entryList(QDir::Files))
257  {
258  const QString from = fromDir + QDir::separator() + copyFile;
259  QString to = toDir + QDir::separator() + copyFile;
260 
261 #ifdef Q_OS_WIN
262  {
263  // To fix issue #702 check each not symlink if it is actually broken symlink.
264  // Also trying to mimic Unix symlink. If a file eaxists do not create a symlink and remove it if exists.
265  QFile fileTo(to);
266  if (fileTo.exists())
267  {
268  if (not fileTo.rename(to + QLatin1String(".lnk")))
269  {
270  QFile::remove(to + QLatin1String(".lnk"));
271  fileTo.rename(to + QLatin1String(".lnk"));
272  }
273 
274  QFileInfo info(to + QLatin1String(".lnk"));
275  if (info.symLinkTarget().isEmpty())
276  {
277  fileTo.copy(to);
278  fileTo.remove();
279  continue; // The file already exists, skip creating shortcut
280  }
281  }
282  }
283 
284  to = to + QLatin1String(".lnk");
285 #endif
286 
287  if (QFile::exists(to))
288  {
289  if (replaceOnConflit)
290  {
291  QFile::remove(to);
292  }
293  else
294  {
295  continue;
296  }
297  }
298 
299  QFile::link(from, to);
300  }
301 
302  foreach (QString copyDir, dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot))
303  {
304  const QString from = fromDir + QDir::separator() + copyDir;
305  const QString to = toDir + QDir::separator() + copyDir;
306 
307  if (dir.mkpath(to) == false)
308  {
309  return;
310  }
311 
312  SymlinkCopyDirRecursive(from, to, replaceOnConflit);
313  }
314 }

317 QString PrepareStandardFiles(const QString &currentPath, const QString &standardPath, const QString &defPath)
318 {
319  QDir standardPathDir(standardPath);
320  QDir currentPathDir(currentPath);
321  if ((currentPath == defPath || not currentPathDir.exists()) && standardPathDir.exists())
322  {
323  const QDir localdata (defPath);
324  if (localdata.mkpath("."))
325  {
326  SymlinkCopyDirRecursive(standardPath, defPath, false);
327  }
328  return defPath;
329  }
330  return currentPath;
