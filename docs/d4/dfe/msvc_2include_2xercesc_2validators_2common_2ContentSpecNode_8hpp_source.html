160  // -----------------------------------------------------------------------
161  // Private Data Members
162  //
163  // fElement
164  // If the type is Leaf/Any*, then this is the qName of the element. If the URI
165  // is fgPCDataElemId, then its a PCData node. Else, it is zero.
166  //
167  // fFirst
168  // fSecond
169  // The optional first and second nodes. The fType field indicates
170  // which of these are valid. The validity constraints are:
171  //
172  // Leaf = Neither valid
173  // ZeroOrOne, ZeroOrMore = First
174  // Choice, Sequence, All = First and Second
175  // Any* = Neither valid
176  //
177  // fType
178  // The type of node. This controls how many of the child node fields
179  // are used.
180  //
181  // fAdoptFirst
182  // Indicate if this ContentSpecNode adopts the fFirst, and is responsible
183  // for deleting it.
184  //
185  // fAdoptSecond
186  // Indicate if this ContentSpecNode adopts the fSecond, and is responsible
187  // for deleting it.
188  //
189  // fMinOccurs
190  // Indicate the minimum times that this node can occur
191  //
192  // fMaxOccurs
193  // Indicate the maximum times that this node can occur
194  // -1 (Unbounded), default (1)
195  // -----------------------------------------------------------------------

200  ContentSpecNode* fSecond;
