122 /**
123  * @brief Create help create tool.
124  * @param doc dom document container.
125  * @param data container with variables.
126  * @param id object id in container.
127  * @param idPoint object id in containerPoint.
128  * @param parse parser file mode.
129  * @param typeCreation way we create this tool.
130  * @param idTool tool id.
131  */
132 void VNodePoint::Create(VAbstractPattern *doc, VContainer *data, VMainGraphicsScene *scene,
133  quint32 id, quint32 idPoint, const Document &parse,
134  const Source &typeCreation, const QString &blockName, const quint32 &idTool)
135 {
136  if (parse == Document::FullParse)
137  {
138  VAbstractTool::AddRecord(id, Tool::NodePoint, doc);
139  //TODO Need create garbage collector and remove all nodes that we don't use.
140  //Better check garbage before each saving file. Check only modeling tags.
141  VNodePoint *point = new VNodePoint(doc, data, id, idPoint, typeCreation, blockName, idTool, doc);
142 
143  connect(scene, &VMainGraphicsScene::EnableToolMove, point, &VNodePoint::EnableToolMove);
144  connect(scene, &VMainGraphicsScene::EnablePointItemHover, point, &VNodePoint::AllowHover);
145  connect(scene, &VMainGraphicsScene::EnablePointItemSelection, point, &VNodePoint::AllowSelecting);
146  connect(scene, &VMainGraphicsScene::enableTextItemHover, point, &VNodePoint::allowTextHover);
147  connect(scene, &VMainGraphicsScene::enableTextItemSelection, point, &VNodePoint::allowTextSelectable);
148  VAbstractPattern::AddTool(id, point);
149  if (idTool != NULL_ID)
150  {
151  //Some nodes we don't show on scene. Tool that create this node must free memory.
152  VDataTool *tool = VAbstractPattern::getTool(idTool);
153  SCASSERT(tool != nullptr)
154  point->setParent(tool); // Adopted by a tool
155  }
156  else
157  {
158  // Try to prevent memory leak
159  scene->addItem(point); // First adopted by scene
160  point->hide(); // If no one will use node, it will stay hidden
161  point->SetParentType(ParentType::Scene);
162  }
163  }
164  else
165  {
166  doc->UpdateToolData(id, data);
167  }
168 }

200  * @brief AddToFile add tag with Information about tool into file.
201  */
202 void VNodePoint::AddToFile()
203 {
204  const QSharedPointer<VPointF> point = VAbstractTool::data.GeometricObject<VPointF>(m_id);
205  QDomElement domElement = doc->createElement(getTagName());
206 
207  doc->SetAttribute(domElement, VDomDocument::AttrId, m_id);
208  doc->SetAttribute(domElement, AttrType, ToolType);
209  doc->SetAttribute(domElement, AttrIdObject, idNode);
210  doc->SetAttribute(domElement, AttrMx, qApp->fromPixel(point->mx()));
211  doc->SetAttribute(domElement, AttrMy, qApp->fromPixel(point->my()));
212  doc->SetAttribute<bool>(domElement, AttrShowPointName, point->isShowPointName());
213 
214  if (idTool != NULL_ID)
215  {
216  doc->SetAttribute(domElement, AttrIdTool, idTool);
217  }
218 
219  AddToModeling(domElement);
220 }
