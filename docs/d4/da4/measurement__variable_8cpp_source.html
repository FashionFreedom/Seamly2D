137 QStringList MeasurementVariable::ListHeights(QMap<GHeights, bool> heights, Unit patternUnit)
138 {
139  QStringList list;
140 
141  QMap<GHeights, bool>::const_iterator i = heights.constBegin();
142  while (i != heights.constEnd())
143  {
144  if (i.value() && i.key() != GHeights::ALL)
145  {
146  ListValue(list, static_cast<int>(i.key()), patternUnit);
147  }
148  ++i;
149  }
150 
151  if (list.isEmpty())
152  {
153  list = MeasurementVariable::WholeListHeights(patternUnit);
154  }
155  return list;
156 }

235 qreal MeasurementVariable::CalcValue() const
236 {
237  if (d->currentUnit == nullptr || d->currentSize == nullptr || d->currentHeight == nullptr)
238  {
239  return VInternalVariable::GetValue();
240  }
241 
242  if (*d->currentUnit == Unit::Inch)
243  {
244  qWarning("Gradation doesn't support inches");
245  return 0;
246  }
247 
248  const qreal sizeIncrement = UnitConvertor(2.0, Unit::Cm, *d->currentUnit);
249  const qreal heightIncrement = UnitConvertor(6.0, Unit::Cm, *d->currentUnit);
250 
251  // Formula for calculation gradation
252  const qreal k_size = ( *d->currentSize - d->baseSize ) / sizeIncrement;
253  const qreal k_height = ( *d->currentHeight - d->baseHeight ) / heightIncrement;
254  return d->base + k_size * d->ksize + k_height * d->kheight;
255 }
