191 void TST_VArc::TestGetPoints()
192 {
193  QFETCH(qreal, radius);
194  QFETCH(qreal, startAngle);
195  QFETCH(qreal, endAngle);
196 
197  const VPointF center;
198  VArc arc(center, radius, startAngle, endAngle);
199 
200  QVector<QPointF> points = arc.getPoints();
201 
202  {
203  const qreal epsRadius = 1.5; // computing error
204 
205  for (int i=0; i < points.size(); ++i)
206  {
207  QLineF rLine(static_cast<QPointF>(center), points.at(i));
208  const qreal value = qAbs(rLine.length() - radius);
209  const QString errorMsg = QString("Broken the first rule. All points should be on the same distance from "
210  "the center. Error ='%1'.").arg(value);
211  QVERIFY2(value <= epsRadius, qUtf8Printable(errorMsg));
212  }
213  }
214 
215  {
216  qreal gSquere = 0.0;// geometry square
217 
218  if (VFuzzyComparePossibleNulls(arc.AngleArc(), 360.0))
219  {// circle square
220  gSquere = M_PI * radius * radius;
221  }
222  else
223  {// sector square
224  gSquere = (M_PI * radius * radius) / 360.0 * arc.AngleArc();
225  points.append(static_cast<QPointF>(center));
226  }
227 
228  // calculated square
229  const qreal cSquare = qAbs(VAbstractPiece::sumTrapezoids(points)/2.0);
230  const qreal value = qAbs(gSquere - cSquare);
231  const QString errorMsg =
232  QString("Broken the second rule. Interpolation has too big computing error. Error ='%1'.").arg(value);
233  const qreal epsSquere = gSquere * 0.24 / 100; // computing error 0.24 % from origin squere
234  QVERIFY2(value <= epsSquere, qUtf8Printable(errorMsg));
235  }
236 }
