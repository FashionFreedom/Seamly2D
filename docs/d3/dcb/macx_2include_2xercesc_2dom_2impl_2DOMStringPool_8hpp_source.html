155 inline void DOMBuffer::
156 append (const XMLCh* const chars)
157 {
158  XMLSize_t count = XMLString::stringLen(chars);
159  if (fIndex + count >= fCapacity)
160  expandCapacity(count);
161 
162  memcpy(&fBuffer[fIndex], chars, count * sizeof(XMLCh));
163  fIndex += count;
164 
165  // Keep it null terminated
166  fBuffer[fIndex] = 0;
167 }
