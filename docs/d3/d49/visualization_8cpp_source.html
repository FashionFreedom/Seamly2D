183 //---------------------------------------------------------------------------------------------------------------------
184 qreal Visualization::FindVal(const QString &expression,
185  const QHash<QString, QSharedPointer<VInternalVariable> > *vars)
186 {
187  qreal val = 0;
188  if (expression.isEmpty())
189  {
190  val = 0;
191  }
192  else
193  {
194  try
195  {
196  // Replace line return with spaces for calc if exist
197  QString formula = expression;
198  formula.replace("\n", " ");
199  formula = qApp->TrVars()->FormulaFromUser(formula, qApp->Settings()->GetOsSeparator());
200  QScopedPointer<Calculator> cal(new Calculator());
201  val = cal->EvalFormula(vars, formula);
202 
203  if (qIsInf(val) || qIsNaN(val))
204  {
205  val = 0;
206  }
207  }
208  catch (qmu::QmuParserError &error)
209  {
210  val = 0;
211  qDebug() << "\nMath parser error:\n"
212  << "--------------------------------------\n"
213  << "Message: " << error.GetMsg() << "\n"
214  << "Expression: " << error.GetExpr() << "\n"
215  << "--------------------------------------";
216  }
217  }
218  return val;
219 }
