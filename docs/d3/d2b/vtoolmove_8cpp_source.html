104 QPointF findRotationOrigin(const QVector<SourceItem> objects, const VContainer *data, qreal calcLength, qreal calcAngle)
105 {
106  QPolygonF originObjects;
107  qCDebug(vTool, "Find center of objects: ");
108  for (auto item : objects)
109  {
110  qCDebug(vTool, "Object: %d", item.id);
111  const QSharedPointer<VGObject> object = data->GetGObject(item.id);
112 
113  Q_STATIC_ASSERT_X(static_cast<int>(GOType::AllCurves) == 10, "Not all objects were handled.");
114 
115  switch(static_cast<GOType>(object->getType()))
116  {
117  case GOType::Point:
118  originObjects.append(data->GeometricObject<VPointF>(item.id)->toQPointF());
119  break;
120  case GOType::Arc:
121  case GOType::EllipticalArc:
122  case GOType::Spline:
123  case GOType::SplinePath:
124  case GOType::CubicBezier:
125  case GOType::CubicBezierPath:
126  originObjects.append(data->GeometricObject<VAbstractCurve>(item.id)->getPoints());
127  break;
128  case GOType::Unknown:
129  case GOType::Curve:
130  case GOType::Path:
131  case GOType::AllCurves:
132  default:
133  Q_UNREACHABLE();
134  break;
135  }
136  }
137 
138  QPointF rotationOrigin = originObjects.boundingRect().center();
139  QLineF move(rotationOrigin, QPointF(rotationOrigin.x() + calcLength, rotationOrigin.y()));
140  move.setAngle(calcAngle);
141  return move.p2();}
142 QT_WARNING_POP
143 }

201 VToolMove *VToolMove::Create(quint32 _id, QString &formulaAngle, QString &formulaLength, QString &formulaRotation,
202  quint32 &originPointId, const QString &suffix, const QVector<SourceItem> &source,
203  const QVector<DestinationItem> &destination, VMainGraphicsScene *scene,
204  VAbstractPattern *doc, VContainer *data, const Document &parse, const Source &typeCreation)
205 {
206  qreal calcAngle = 0;
207  qreal calcLength = 0;
208  qreal calcRotation = 0;
209 
210  calcAngle = CheckFormula(_id, formulaAngle, data);
211  calcLength = qApp->toPixel(CheckFormula(_id, formulaLength, data));
212  calcRotation = CheckFormula(_id, formulaRotation, data);
213 
214  QPointF rotationOrigin;
215  QSharedPointer<VPointF> originPoint;
216 
217  if (originPointId == NULL_ID)
218  {
219  rotationOrigin = findRotationOrigin(source, data, calcLength, calcAngle);
220  }
221  else
222  {
223  originPoint = data->GeometricObject<VPointF>(originPointId);
224  rotationOrigin = static_cast<QPointF>(*originPoint);
225  QLineF moveLine(rotationOrigin, QPointF(rotationOrigin.x() + calcLength, rotationOrigin.y()));
226  moveLine.setAngle(calcAngle);
227  rotationOrigin = moveLine.p2();
228  }
229 
230  QVector<DestinationItem> dest = destination;
231 
232  quint32 id = _id;
233  if (typeCreation == Source::FromGui)
234  {
235  dest.clear();// Try to avoid mistake, value must be empty
236 
237  id = VContainer::getNextId();//Just reserve id for tool
238 
239  qCDebug(vTool, "Create SourceItem GUI");
240  for (int i = 0; i < source.size(); ++i)
241  {
242  const SourceItem item = source.at(i);
243  qCDebug(vTool, "Add Object: %d", item.id);
244  const QSharedPointer<VGObject> object = data->GetGObject(item.id);
245 
246  // This check helps to find missed objects in the switch
247  Q_STATIC_ASSERT_X(static_cast<int>(GOType::AllCurves) == 10, "Not all objects were handled.");
248 
249 QT_WARNING_PUSH
250 QT_WARNING_DISABLE_GCC("-Wswitch-default")
251  qCDebug(vTool, "VToolMove - Rotation Origin");
252  qCDebug(vTool, "X: %f", rotationOrigin.x());
253  qCDebug(vTool, "Y: %f", rotationOrigin.y());
254  switch(static_cast<GOType>(object->getType()))
255  {
256  case GOType::Point:
257  qCDebug(vTool, "VToolMove - Point");
258  qCDebug(vTool, "length: %f", calcLength);
259  qCDebug(vTool, "angle: %f", calcAngle);
260  qCDebug(vTool, "rotation: %f\n", calcRotation);
261  dest.append(createPoint(id, item.id, calcAngle, calcLength, calcRotation,
262  rotationOrigin, suffix, data));
263  break;
264  case GOType::Arc:
265  dest.append(createArc<VArc>(id, item.id, calcAngle, calcLength, calcRotation,
266  rotationOrigin, suffix, data));
267  break;
268  case GOType::EllipticalArc:
269  dest.append(createArc<VEllipticalArc>(id, item.id, calcAngle, calcLength, calcRotation,
270  rotationOrigin, suffix, data));
271  break;
272  case GOType::Spline:
273  qCDebug(vTool, "VToolMove - Spline");
274  qCDebug(vTool, "length: %f", calcLength);
275  qCDebug(vTool, "angle: %f", calcAngle);
276  qCDebug(vTool, "rotation: %f\n", calcRotation);
277  dest.append(createCurve<VSpline>(id, item.id, calcAngle, calcLength, calcRotation,
278  rotationOrigin, suffix, data));
279  break;
280  case GOType::SplinePath:
281  dest.append(createCurveWithSegments<VSplinePath>(id, item.id, calcAngle, calcLength,
282  calcRotation, rotationOrigin, suffix,
283  data));
284  break;
285  case GOType::CubicBezier:
286  dest.append(createCurve<VCubicBezier>(id, item.id, calcAngle, calcLength, calcRotation,
287  rotationOrigin, suffix, data));
288  break;
289  case GOType::CubicBezierPath:
290  qCDebug(vTool, "VToolMove - VCubicBezier");
291  qCDebug(vTool, "length: %f", calcLength);
292  qCDebug(vTool, "angle: %f", calcAngle);
293  qCDebug(vTool, "rotation: %f\n", calcRotation);
294  dest.append(createCurveWithSegments<VCubicBezierPath>(id, item.id, calcAngle, calcLength,
295  calcRotation, rotationOrigin, suffix,
296  data));
297  break;
298  case GOType::Unknown:
299  case GOType::Curve:
300  case GOType::Path:
301  case GOType::AllCurves:
302  default:
303  break;
304  }
305 QT_WARNING_POP
306  }
307  }
308  else
309  {
310 
311  qCDebug(vTool, "Create sourceItem\n");
312  for (int i = 0; i < source.size(); ++i)
313  {
314  const SourceItem item = source.at(i);
315  qCDebug(vTool, "Add Object: %d\n", item.id);
316  const QSharedPointer<VGObject> object = data->GetGObject(item.id);
317 
318  // This check helps to find missed objects in the switch
319  Q_STATIC_ASSERT_X(static_cast<int>(GOType::AllCurves) == 10, "Not all objects were handled.");
320 
321 QT_WARNING_PUSH
322 QT_WARNING_DISABLE_GCC("-Wswitch-default")
323  switch(static_cast<GOType>(object->getType()))
324  {
325  case GOType::Point:
326  {
327  updatePoint(id, item.id, calcAngle, calcLength, calcRotation, rotationOrigin,
328  suffix, data, dest.at(i));
329  break;
330  }
331  case GOType::Arc:
332  updateArc<VArc>(id, item.id, calcAngle, calcLength, calcRotation, rotationOrigin,
333  suffix, data, dest.at(i).id);
334  break;
335  case GOType::EllipticalArc:
336  updateArc<VEllipticalArc>(id, item.id, calcAngle, calcLength, calcRotation, rotationOrigin,
337  suffix, data, dest.at(i).id);
338  break;
339  case GOType::Spline:
340  updateCurve<VSpline>(id, item.id, calcAngle, calcLength, calcRotation, rotationOrigin,
341  suffix, data, dest.at(i).id);
342  break;
343  case GOType::SplinePath:
344  updateCurveWithSegments<VSplinePath>(id, item.id, calcAngle, calcLength, calcRotation,
345  rotationOrigin, suffix, data, dest.at(i).id);
346  break;
347  case GOType::CubicBezier:
348  updateCurve<VCubicBezier>(id, item.id, calcAngle, calcLength, calcRotation, rotationOrigin,
349  suffix, data, dest.at(i).id);
350  break;
351  case GOType::CubicBezierPath:
352  updateCurveWithSegments<VCubicBezierPath>(id, item.id, calcAngle, calcLength, calcRotation,
353  rotationOrigin, suffix, data, dest.at(i).id);
354  break;
355  case GOType::Unknown:
356  case GOType::Curve:
357  case GOType::Path:
358  case GOType::AllCurves:
359  default:
360  break;
361  }
362 QT_WARNING_POP
363  }
364  if (parse != Document::FullParse)
365  {
366  doc->UpdateToolData(id, data);
367  }
368  }
369 
370  if (parse == Document::FullParse)
371  {
372  VDrawTool::AddRecord(id, Tool::Move, doc);
373  VToolMove *tool = new VToolMove(doc, data, id, formulaAngle, formulaLength, formulaRotation, originPointId,
374  suffix, source, dest, typeCreation);
375  scene->addItem(tool);
376  initOperationToolConnections(scene, tool);
377  VAbstractPattern::AddTool(id, tool);
378 
379  if (!originPoint.isNull())
380  {
381  doc->IncrementReferens(originPoint->getIdTool());
382  }
383  qCDebug(vTool, "Increment references");
384  for (int i = 0; i < source.size(); ++i)
385  {
386  const SourceItem item = source.at(i);
387  qCDebug(vTool, "Object: %d", item.id);
388  doc->IncrementReferens(data->GetGObject(item.id)->getIdTool());
389  }
390  return tool;
391  }
392  return nullptr;
393 }
