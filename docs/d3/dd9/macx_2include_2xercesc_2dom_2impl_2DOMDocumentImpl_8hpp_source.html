314  // Per-Document heap Variables.
315  // The heap consists of one or more biggish blocks which are
316  // sub-allocated for individual allocations of nodes, strings, etc.
317  // The big blocks form a linked list, allowing them to be located for deletion.
318  //
319  // There is no provision for deleting suballocated blocks, other than
320  // deleting the entire heap when the document is deleted.
321  //
322  // There is no header on individual sub-allocated blocks.
323  // The header on big blocks consists only of a single back pointer to
324  // the previously allocated big block (our linked list of big blocks)

368 inline const XMLCh* DOMDocumentImpl::getPooledString(const XMLCh *in)
369 {
370  if (in == 0)
371  return 0;
372  XMLSize_t n = XMLString::stringLen(in);
373 
374  DOMStringPoolEntry **pspe;
375  DOMStringPoolEntry *spe;
376 
377  XMLSize_t inHash = XMLString::hash(in, fNameTableSize);
378  pspe = &fNameTable[inHash];
379  while (*pspe != 0)
380  {
381  if ((*pspe)->fLength == n && XMLString::equals((*pspe)->fString, in))
382  return (*pspe)->fString;
383  pspe = &((*pspe)->fNext);
384  }
385 
386  // This string hasn't been seen before. Add it to the pool.
387  //
388 
389  // Compute size to allocate. Note that there's 1 char of string
390  // declared in the struct, so we don't need to add one again to
391  // account for the trailing null.
392  //
393  XMLSize_t sizeToAllocate = sizeof(DOMStringPoolEntry) + n*sizeof(XMLCh);
394  *pspe = spe = (DOMStringPoolEntry *)allocate(sizeToAllocate);
395  spe->fLength = n;
396  spe->fNext = 0;
397  XMLString::copyString((XMLCh*)spe->fString, in);
398 
399  return spe->fString;
400 }

402 inline const XMLCh* DOMDocumentImpl::getPooledNString(const XMLCh *in, XMLSize_t n)
403 {
404  if (in == 0)
405  return 0;
406 
407  DOMStringPoolEntry **pspe;
408  DOMStringPoolEntry *spe;
409 
410  XMLSize_t inHash = XMLString::hashN(in, n, fNameTableSize);
411  pspe = &fNameTable[inHash];
412  while (*pspe != 0)
413  {
414  if ((*pspe)->fLength == n && XMLString::equalsN((*pspe)->fString, in, n))
415  return (*pspe)->fString;
416  pspe = &((*pspe)->fNext);
417  }
418 
419  // This string hasn't been seen before. Add it to the pool.
420  //
421 
422  // Compute size to allocate. Note that there's 1 char of string
423  // declared in the struct, so we don't need to add one again to
424  // account for the trailing null.
425  //
426  XMLSize_t sizeToAllocate = sizeof(DOMStringPoolEntry) + n*sizeof(XMLCh);
427  *pspe = spe = (DOMStringPoolEntry *)allocate(sizeToAllocate);
428  spe->fLength = n;
429  spe->fNext = 0;
430  XMLString::copyNString((XMLCh*)spe->fString, in, n);
431 
432  return spe->fString;
433 }

435 inline int DOMDocumentImpl::indexofQualifiedName(const XMLCh* name)
436 {
437  int i = 0;
438  int colon = -1;
439  int colon_count = 0;
440  for (; *name != 0; ++i, ++name)
441  {
442  if (*name == chColon)
443  {
444  ++colon_count;
445  colon = i;
446  }
447  }
448 
449  if (i == 0 || colon == 0 || colon == (i - 1) || colon_count > 1)
450  return -1;
451 
452  return colon != -1 ? colon : 0;
453 }
