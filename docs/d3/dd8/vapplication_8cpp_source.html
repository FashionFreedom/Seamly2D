93 inline void noisyFailureMsgHandler(QtMsgType type, const QMessageLogContext &context, const QString &msg)
94 {
95  // Why on earth didn't Qt want to make failed signal/slot connections qWarning?
96  if ((type == QtDebugMsg) && msg.contains(QStringLiteral("::connect")))
97  {
98  type = QtWarningMsg;
99  }
100 
101 #if defined(V_NO_ASSERT)
102  // I have decided to hide this annoing message for release builds.
103  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QSslSocket: cannot resolve")))
104  {
105  type = QtDebugMsg;
106  }
107 
108  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("setGeometry: Unable to set geometry")))
109  {
110  type = QtDebugMsg;
111  }
112 #endif //defined(V_NO_ASSERT)
113 
114 #if defined(Q_OS_MAC)
115  // Hide Qt bug 'Assertion when reading an icns file'
116  // https://bugreports.qt.io/browse/QTBUG-45537
117  // Remove after Qt fix will be released
118  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QICNSHandler::read()")))
119  {
120  type = QtDebugMsg;
121  }
122 
123  // See issue #568
124  if (msg.contains(QStringLiteral("Error receiving trust for a CA certificate")))
125  {
126  type = QtDebugMsg;
127  }
128 #endif
129 
130  // this is another one that doesn't make sense as just a debug message. pretty serious
131  // sign of a problem
132  // http://www.developer.nokia.com/Community/Wiki/QPainter::begin:Paint_device_returned_engine_%3D%3D_0_(Known_Issue)
133  if ((type == QtDebugMsg) && msg.contains(QStringLiteral("QPainter::begin"))
134  && msg.contains(QStringLiteral("Paint device returned engine")))
135  {
136  type = QtWarningMsg;
137  }
138 
139  // This qWarning about "Cowardly refusing to send clipboard message to hung application..."
140  // is something that can easily happen if you are debugging and the application is paused.
141  // As it is so common, not worth popping up a dialog.
142  if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QClipboard::event"))
143  && msg.contains(QStringLiteral("Cowardly refusing")))
144  {
145  type = QtDebugMsg;
146  }
147 
148  // only the GUI thread should display message boxes. If you are
149  // writing a multithreaded application and the error happens on
150  // a non-GUI thread, you'll have to queue the message to the GUI
151  QCoreApplication *instance = QCoreApplication::instance();
152  const bool isGuiThread = instance && (QThread::currentThread() == instance->thread());
153 
154  {
155  QString debugdate = "[" + QDateTime::currentDateTime().toString(QStringLiteral("yyyy.MM.dd hh:mm:ss"));
156 
157  switch (type)
158  {
159  case QtDebugMsg:
160  debugdate += QString(":DEBUG:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
161  .arg(context.function).arg(context.category).arg(msg);
162  vStdOut() << QApplication::translate("vNoisyHandler", "DEBUG:") << msg << "\n";
163  break;
164  case QtWarningMsg:
165  debugdate += QString(":WARNING:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
166  .arg(context.function).arg(context.category).arg(msg);
167  vStdErr() << QApplication::translate("vNoisyHandler", "WARNING:") << msg << "\n";
168  break;
169  case QtCriticalMsg:
170  debugdate += QString(":CRITICAL:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
171  .arg(context.function).arg(context.category).arg(msg);
172  vStdErr() << QApplication::translate("vNoisyHandler", "CRITICAL:") << msg << "\n";
173  break;
174  case QtFatalMsg:
175  debugdate += QString(":FATAL:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
176  .arg(context.function).arg(context.category).arg(msg);
177  vStdErr() << QApplication::translate("vNoisyHandler", "FATAL:") << msg << "\n";
178  break;
179  #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
180  case QtInfoMsg:
181  debugdate += QString(":INFO:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
182  .arg(context.function).arg(context.category).arg(msg);
183  vStdOut() << QApplication::translate("vNoisyHandler", "INFO:") << msg << "\n";
184  break;
185  #endif
186  default:
187  break;
188  }
189 
190  (*qApp->LogFile()) << debugdate << Qt::endl;
191  }
192 
193  if (isGuiThread)
194  {
195  //fixme: trying to make sure there are no save/load dialogs are opened, because error message during them will
196  //lead to crash
197  const bool topWinAllowsPop = (QApplication::activeModalWidget() == nullptr) ||
198  !QApplication::activeModalWidget()->inherits("QFileDialog");
199 
200  QMessageBox messageBox;
201 
202  switch (type)
203  {
204  case QtWarningMsg:
205  messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Warning"));
206  messageBox.setIcon(QMessageBox::Warning);
207  break;
208  case QtCriticalMsg:
209  messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Critical Error"));
210  messageBox.setIcon(QMessageBox::Critical);
211  break;
212  case QtFatalMsg:
213  messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Fatal Error"));
214  messageBox.setIcon(QMessageBox::Critical);
215  break;
216  #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
217  case QtInfoMsg:
218  messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Information"));
219  messageBox.setIcon(QMessageBox::Information);
220  break;
221  #endif
222  case QtDebugMsg:
223  default:
224  break;
225  }
226 
227  if (type == QtWarningMsg || type == QtCriticalMsg || type == QtFatalMsg)
228  {
229  if (VApplication::IsGUIMode())
230  {
231  if (topWinAllowsPop)
232  {
233  messageBox.setText(msg);
234  messageBox.setStandardButtons(QMessageBox::Ok);
235  messageBox.setWindowModality(Qt::ApplicationModal);
236  messageBox.setModal(true);
237  #ifndef QT_NO_CURSOR
238  QGuiApplication::setOverrideCursor(Qt::ArrowCursor);
239  #endif
240  messageBox.setWindowFlags(messageBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
241  messageBox.exec();
242  #ifndef QT_NO_CURSOR
243  QGuiApplication::restoreOverrideCursor();
244  #endif
245  }
246  }
247  }
248 
249  if (QtFatalMsg == type)
250  {
251  abort();
252  }
253  }
254  else
255  {
256  if( QtDebugMsg != type && QtWarningMsg != type )
257  {
258  abort(); // be NOISY unless overridden!
259  }
260  }
261 }

314 void VApplication::NewSeamly2D(const QString &fileName)
315 {
316  qCDebug(vApp, "Open new detached process.");
317  if (fileName.isEmpty())
318  {
319  qCDebug(vApp, "New process without arguments. program = %s",
320  qUtf8Printable(QCoreApplication::applicationFilePath()));
321  // Path can contain spaces.
322  if (QProcess::startDetached(QCoreApplication::applicationFilePath(), QStringList()))
323  {
324  qCDebug(vApp, "The process was started successfully.");
325  }
326  else
327  {
328  qCWarning(vApp, "Could not run process. The operation timed out or an error occurred.");
329  }
330  }
331  else
332  {
333  const QString run = QString("\"%1\" \"%2\"").arg(QCoreApplication::applicationFilePath()).arg(fileName);
334  qCDebug(vApp, "New process with arguments. program = %s", qUtf8Printable(run));
335 
336  if (QProcess::startDetached(QCoreApplication::applicationFilePath(), QStringList{fileName}))
337  {
338  qCDebug(vApp, "The process was started successfully.");
339  }
340  else
341  {
342  qCWarning(vApp, "Could not run process. The operation timed out or an error occurred.");
343  }
344  }

348 /**
349  * @brief notify Reimplemented from QApplication::notify().
350  * @param receiver receiver.
351  * @param event event.
352  * @return value that is returned from the receiver's event handler.
353  */
354 // reimplemented from QApplication so we can throw exceptions in slots
355 bool VApplication::notify(QObject *receiver, QEvent *event)
356 {
357  try
358  {
359  return QApplication::notify(receiver, event);
360  }
361  catch (const VExceptionObjectError &e)
362  {
363  qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error parsing file. Program will be terminated.")), //-V807
364  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
365  exit(V_EX_DATAERR);
366  }
367  catch (const VExceptionBadId &e)
368  {
369  qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error bad id. Program will be terminated.")),
370  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
371  exit(V_EX_DATAERR);
372  }
373  catch (const VExceptionConversionError &e)
374  {
375  qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error can't convert value. Program will be terminated.")),
376  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
377  exit(V_EX_DATAERR);
378  }
379  catch (const VExceptionEmptyParameter &e)
380  {
381  qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error empty parameter. Program will be terminated.")),
382  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
383  exit(V_EX_DATAERR);
384  }
385  catch (const VExceptionWrongId &e)
386  {
387  qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error wrong id. Program will be terminated.")),
388  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
389  exit(V_EX_DATAERR);
390  }
391  catch (const VExceptionToolWasDeleted &e)
392  {
393  qCCritical(vApp, "%s\n\n%s\n\n%s",
394  qUtf8Printable("Unhadled deleting tool. Continue use object after deleting"),
395  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
396  exit(V_EX_DATAERR);
397  }
398  catch (const VException &e)
399  {
400  qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Something's wrong!!")),
401  qUtf8Printable(e.ErrorMessage()), qUtf8Printable(e.DetailedInformation()));
402  return true;
403  }
404  // These last two cases are special. I found that we can't show here a modal dialog with an error message.
405  // Somehow program doesn't wait until an error dialog will be closed. But if ignore the exception the program will
406  // hang.
407  catch (const qmu::QmuParserError &e)
408  {
409  qCCritical(vApp, "%s", qUtf8Printable(tr("Parser error: %1. Program will be terminated.").arg(e.GetMsg())));
410  exit(V_EX_DATAERR);
411  }
412  catch (std::exception& e)
413  {
414  qCCritical(vApp, "%s", qUtf8Printable(tr("Exception thrown: %1. Program will be terminated.").arg(e.what())));
415  exit(V_EX_SOFTWARE);
416  }
417  return false;
418 }

505 void VApplication::BeginLogging()
506 {
507  VlpCreateLock(lockLog, LogPath(), [this](){return new QFile(LogPath());});
508 
509  if (lockLog->IsLocked())
510  {
511  if (lockLog->GetProtected()->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
512  {
513  out.reset(new QTextStream(lockLog->GetProtected().get()));
514  qInstallMessageHandler(noisyFailureMsgHandler);
515  qCDebug(vApp, "Log file %s was locked.", qUtf8Printable(LogPath()));
516  }
517  else
518  {
519  qCDebug(vApp, "Error opening log file \'%s\'. All debug output redirected to console.",
520  qUtf8Printable(LogPath()));
521  }
522  }
523  else
524  {
525  qCDebug(vApp, "Failed to lock %s", qUtf8Printable(LogPath()));
526  }
527 }

530 void VApplication::ClearOldLogs() const
531 {
532  QDir logsDir(LogDirPath());
533  logsDir.setNameFilters(QStringList("*.log"));
534  logsDir.setCurrent(LogDirPath());
535 
536  const QStringList allFiles = logsDir.entryList(QDir::NoDotAndDotDot | QDir::Files);
537  if (allFiles.isEmpty() == false)
538  {
539  qCDebug(vApp, "Clearing old logs");
540  for (int i = 0, sz = allFiles.size(); i < sz; ++i)
541  {
542  auto fn = allFiles.at(i);
543  QFileInfo info(fn);
544  if (info.birthTime().daysTo(QDateTime::currentDateTime()) >= DAYS_TO_KEEP_LOGS)
545  {
546  VLockGuard<QFile> tmp(info.absoluteFilePath(), [&fn](){return new QFile(fn);});
547  if (tmp.GetProtected() != nullptr)
548  {
549  if (tmp.GetProtected()->remove())
550  {
551  qCDebug(vApp, "Deleted %s", qUtf8Printable(info.absoluteFilePath()));
552  }
553  else
554  {
555  qCDebug(vApp, "Could not delete %s", qUtf8Printable(info.absoluteFilePath()));
556  }
557  }
558  else
559  {
560  qCDebug(vApp, "Failed to lock %s", qUtf8Printable(info.absoluteFilePath()));
561  }
562  }
563  }
564  }
565  else
566  {
567  qCDebug(vApp, "There are no old logs.");
568  }
569 }

665 {
666  switch(e->type())
667  {
668  // In Mac OS X the QFileOpenEvent event is generated when user perform "Open With" from Finder (this event is
669  // Mac specific).
670  case QEvent::FileOpen:
671  {
672  QFileOpenEvent *fileOpenEvent = static_cast<QFileOpenEvent *>(e);

675  {
676  MainWindow *window = qobject_cast<MainWindow*>(mainWindow);
677  if (window)
678  {
679  window->LoadPattern(macFileOpen); // open file in existing window
680  }
681  return true;
682  }
683  break;
684  }
685 #if defined(Q_OS_MAC)
686  case QEvent::ApplicationActivate:
687  {
688  if (mainWindow && not mainWindow->isMinimized())
689  {
690  mainWindow->show();
691  }
692  return true;
693  }
694 #endif //defined(Q_OS_MAC)
695  default:
696  return VAbstractApplication::event(e);
697  }
698  return VAbstractApplication::event(e);

772 void VApplication::GatherLogs() const
773 {
774  QTextStream *out = nullptr;
775  QFile *log = new QFile(QString("%1/seamly2d.log").arg(LogDirPath()));
776  if (log->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
777  {
778  out = new QTextStream(log);
779 
780  QStringList filters{"*.log"};
781  QDir logsDir(LogDirPath());
782  logsDir.setNameFilters(filters);
783  logsDir.setCurrent(LogDirPath());
784 
785  const QStringList allFiles = logsDir.entryList(QDir::NoDotAndDotDot | QDir::Files);
786  if (allFiles.isEmpty() == false)
787  {
788  for (int i = 0, sz = allFiles.size(); i < sz; ++i)
789  {
790  auto fn = allFiles.at(i);
791  QFileInfo info(fn);
792  if (info.fileName() == "seamly2d.log")
793  {
794  continue;
795  }
796 
797  VLockGuard<QFile> tmp(info.absoluteFilePath(), [&fn](){return new QFile(fn);});
798 
799  if (tmp.IsLocked())
800  {
801  *out <<"--------------------------" << Qt::endl;
802  if (tmp.GetProtected()->open(QIODevice::ReadOnly | QIODevice::Text))
803  {
804  QTextStream in(tmp.GetProtected().get());
805  while (!in.atEnd())
806  {
807  *out << in.readLine() << Qt::endl;
808  }
809  tmp.GetProtected()->close();
810  }
811  else
812  {
813  *out << "Log file error:" + tmp.GetProtected()->errorString() << Qt::endl;
814  }
815  }
816  else
817  {
818  qCDebug(vApp, "Failed to lock %s", qUtf8Printable(info.absoluteFilePath()));
819  }
820  }
821  }
822  else
823  {
824  *out << "Could not find logs.";
825  }
826  log->close();
827  }
828  delete out;
829  delete log;
830 }
