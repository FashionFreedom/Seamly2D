73  * @brief CreateEmptyMultisizeFile check if empty multisize measurement file is valid.
74  */
75 void TST_Measurements::CreateEmptyMultisizeFile()
76 {
77  Unit mUnit = Unit::Cm;
78  const int height = 176;
79  const int size = 50;
80 
81  QSharedPointer<VContainer> data = QSharedPointer<VContainer>(new VContainer(nullptr, &mUnit));
82  VContainer::setHeight(height);
83  VContainer::setSize(size);
84 
85  QSharedPointer<MeasurementDoc> m =
86  QSharedPointer<MeasurementDoc>(new MeasurementDoc(mUnit, size, height, data.data()));
87  m->setSize(VContainer::rsize());
88  m->setHeight(VContainer::rheight());
89 
90  QTemporaryFile file;
91  QString fileName;
92  // In Windows we have problems when we try to open QSaveFile when QTemporaryFile with the same name is already open.
93  if (file.open())
94  {
95  // So, before we try to open file in m->SaveDocument function we need to close it and remove.
96  // Just closing - is not enough, if we just close QTemporaryFile we get "access denied" in Windows.
97  fileName = file.fileName();
98  file.close();
99  file.remove();
100  QString error;
101  const bool result = m->SaveDocument(fileName, error);
102 
103  QVERIFY2(result, error.toUtf8().constData());
104  }
105  else
106  {
107  QFAIL("Can't open temporary file.");
108  }
109 
110  try
111  {
112  VDomDocument::ValidateXML(MultiSizeConverter::CurrentSchema, fileName);
113  }
114  catch (VException &error)
115  {
116  QFAIL(error.ErrorMessage().toUtf8().constData());
117  }
118 }
