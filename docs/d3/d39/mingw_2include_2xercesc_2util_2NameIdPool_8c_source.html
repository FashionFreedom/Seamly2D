41 template <class TElem>
42 NameIdPool<TElem>::NameIdPool( const XMLSize_t hashModulus
43  , const XMLSize_t initSize
44  , MemoryManager* const manager) :
45  fMemoryManager(manager)
46  , fIdPtrs(0)
47  , fIdPtrsCount(initSize)
48  , fIdCounter(0)
49  , fBucketList(hashModulus, manager)
50 {
51  if (!hashModulus)
52  ThrowXMLwithMemMgr(IllegalArgumentException, XMLExcepts::Pool_ZeroModulus, fMemoryManager);
53 
54  //
55  // Allocate the initial id pointers array. We don't have to zero them
56  // out since the fIdCounter value tells us which ones are valid. The
57  // zeroth element is never used (and represents an invalid pool id.)
58  //
59  if (!fIdPtrsCount)
60  fIdPtrsCount = 256;
61  fIdPtrs = (TElem**) fMemoryManager->allocate
62  (
63  fIdPtrsCount * sizeof(TElem*)
64  );
65  fIdPtrs[0] = 0;
66 }

69 {

72  // up when we clean the bucket lists.
73  //
74  fMemoryManager->deallocate(fIdPtrs); //delete [] fIdPtrs;
75 }

151 template <class TElem>
152 XMLSize_t NameIdPool<TElem>::put(TElem* const elemToAdopt)
153 {
154  // First see if the key exists already. If so, its an error
155  if(containsKey(elemToAdopt->getKey()))
156  {
157  ThrowXMLwithMemMgr1
158  (
159  IllegalArgumentException
160  , XMLExcepts::Pool_ElemAlreadyExists
161  , elemToAdopt->getKey()
162  , fMemoryManager
163  );
164  }
165 
166  fBucketList.put((void*)elemToAdopt->getKey(), elemToAdopt);
167 
168  //
169  // Give this new one the next available id and add to the pointer list.
170  // Expand the list if that is now required.
171  //
172  if (fIdCounter + 1 == fIdPtrsCount)
173  {
174  // Create a new count 1.5 times larger and allocate a new array
175  XMLSize_t newCount = (XMLSize_t)(fIdPtrsCount * 1.5);
176  TElem** newArray = (TElem**) fMemoryManager->allocate
177  (
178  newCount * sizeof(TElem*)
179  ); //new TElem*[newCount];
180 
181  // Copy over the old contents to the new array
182  memcpy(newArray, fIdPtrs, fIdPtrsCount * sizeof(TElem*));
183 
184  // Ok, toss the old array and store the new data
185  fMemoryManager->deallocate(fIdPtrs); //delete [] fIdPtrs;
186  fIdPtrs = newArray;
187  fIdPtrsCount = newCount;
188  }
189  const XMLSize_t retId = ++fIdCounter;
190  fIdPtrs[retId] = elemToAdopt;
191 
192  // Set the id on the passed element
193  elemToAdopt->setId(retId);
194 
195  // Return the id that we gave to this element
196  return retId;
197 }

261  ThrowXMLwithMemMgr(NoSuchElementException, XMLExcepts::Enum_NoMoreElements, fMemoryManager);

268 template <class TElem> void NameIdPoolEnumerator<TElem>::Reset()
269 {
270  //
271  // Find the next available bucket element in the pool. We use the id
272  // array since its very easy to enumerator through by just maintaining
273  // an index. If the id counter is zero, then its empty and we leave the
274  // current index to zero.
275  //
276  fCurIndex = fToEnum->fIdCounter ? 1:0;
