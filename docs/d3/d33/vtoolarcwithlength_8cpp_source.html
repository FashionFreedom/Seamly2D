138 VToolArcWithLength *VToolArcWithLength::Create(const quint32 _id, const quint32 &center, QString &radius, QString &f1,
139  QString &length, const QString &color, const QString &penStyle,
140  const QString &lineWeight, VMainGraphicsScene *scene, VAbstractPattern *doc, VContainer *data,
141  const Document &parse, const Source &typeCreation)
142 {
143  qreal calcRadius = 0, calcF1 = 0, calcLength = 0;
144 
145  calcRadius = qApp->toPixel(CheckFormula(_id, radius, data));
146  calcLength = qApp->toPixel(CheckFormula(_id, length, data));
147  calcF1 = CheckFormula(_id, f1, data);
148 
149  const VPointF c = *data->GeometricObject<VPointF>(center);
150  VArc *arc = new VArc(calcLength, length, c, calcRadius, radius, calcF1, f1);
151  arc->setLineColor(color);
152  arc->SetPenStyle(penStyle);
153  arc->setLineWeight(lineWeight);
154  quint32 id = _id;
155  if (typeCreation == Source::FromGui)
156  {
157  id = data->AddGObject(arc);
158  data->AddArc(data->GeometricObject<VArc>(id), id);
159  }
160  else
161  {
162  data->UpdateGObject(id, arc);
163  data->AddArc(data->GeometricObject<VArc>(id), id);
164  if (parse != Document::FullParse)
165  {
166  doc->UpdateToolData(id, data);
167  }
168  }
169 
170  if (parse == Document::FullParse)
171  {
172  VDrawTool::AddRecord(id, Tool::ArcWithLength, doc);
173  VToolArcWithLength *toolArc = new VToolArcWithLength(doc, data, id, typeCreation);
174  scene->addItem(toolArc);
175  InitArcToolConnections(scene, toolArc);
176  VAbstractPattern::AddTool(id, toolArc);
177  doc->IncrementReferens(c.getIdTool());
178  return toolArc;
179  }
180  return nullptr;
181 }
