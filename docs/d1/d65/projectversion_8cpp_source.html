74 QString compilerString()
75 {
76 #if defined(Q_CC_INTEL) // must be before GNU, Clang and MSVC because ICC/ICL claim to be them
77  QString iccCompact;
78 #ifdef __INTEL_CLANG_COMPILER
79  iccCompact = QLatin1String("Clang");
80 #elif defined(__INTEL_MS_COMPAT_LEVEL)
81  iccCompact = QLatin1String("Microsoft");
82 #elif defined(__GNUC__)
83  iccCompact = QLatin1String("GCC");
84 #else
85  iccCompact = QLatin1String("no");
86 #endif
87  QString iccVersion;
88  if (__INTEL_COMPILER >= 1300)
89  {
90  iccVersion = QString::number(__INTEL_COMPILER/100);
91  }
92  else
93  {
94  iccVersion = QLatin1String(__INTEL_COMPILER);
95  }
96 #ifdef __INTEL_COMPILER_UPDATE
97  return QLatin1String("Intel(R) C++ ") + iccVersion + QLatin1String(".") + QLatin1String(__INTEL_COMPILER_UPDATE) +
98  QLatin1String(" build ") + QLatin1String(__INTEL_COMPILER_BUILD_DATE) + QLatin1String(" [") +
99  QLatin1String(iccCompact) + QLatin1String(" compatibility]");
100 #else
101  return QLatin1String("Intel(R) C++ ") + iccVersion + QLatin1String(" build ") +
102  QLatin1String(__INTEL_COMPILER_BUILD_DATE) + QLatin1String(" [") + iccCompact +
103  QLatin1String(" compatibility]");
104 #endif
105 #elif defined(Q_CC_CLANG) // must be before GNU, because clang claims to be GNU too
106  QString isAppleString;
107 #if defined(__apple_build_version__) // Apple clang has other version numbers
108  isAppleString = QLatin1String(" (Apple)");
109 #endif
110  return QLatin1String("Clang " ) + QString::number(__clang_major__) + QLatin1Char('.')
111  + QString::number(__clang_minor__) + isAppleString;
112 #elif defined(Q_CC_GNU)
113  return QLatin1String("GCC " ) + QLatin1String(__VERSION__);
114 #elif defined(Q_CC_MSVC)
115  if (_MSC_VER >= 1800) // 1800: MSVC 2013 (yearly release cycle)
116  {
117  return QLatin1String("MSVC ") + QString::number(2008 + ((_MSC_VER / 100) - 13));
118  }
119  if (_MSC_VER >= 1500) // 1500: MSVC 2008, 1600: MSVC 2010, ... (2-year release cycle)
120  {
121  return QLatin1String("MSVC ") + QString::number(2008 + 2 * ((_MSC_VER / 100) - 15));
122  }
123 #else
124  return QStringLiteral("<unknown compiler>");
125 #endif
126 }
