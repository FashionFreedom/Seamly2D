195 QString MeasurementDatabaseDialog::imageUrl(const QString &number)
196 {
197  QString imgUrl("<img src=\"wrong.png\" align=\"center\"/>"); // In case of error
198  const QString filePath = QString("://diagrams/%1.svg").arg(MapDiagrams(qApp->translateVariables(), number));
199  if (QFileInfo::exists(filePath))
200  {
201  // Load your SVG
202  QSvgRenderer renderer;
203  const bool ok = renderer.load(filePath);
204  if (ok)
205  {
206  const QScreen *screen = QGuiApplication::screens().at(0);
207  if (screen)
208  {
209  const QSize defSize = renderer.defaultSize();
210 
211  // Prepare a QImage with desired characteritisc
212  QImage image(defSize, QImage::Format_RGB32);
213  image.fill(Qt::white);
214 
215  const QRect geometry = screen->geometry();
216  const int baseHeight = 1440;
217  const int imgHeight = geometry.height() * defSize.height() / baseHeight;
218 
219  QImage scaledImg = image.scaledToHeight(imgHeight);
220 
221  // Get QPainter that paints to the image
222  QPainter painter(&scaledImg);
223  renderer.render(&painter);
224 
225  QByteArray byteArray;
226  QBuffer buffer(&byteArray);
227  scaledImg.save(&buffer, "PNG");
228  imgUrl = QString("<img src=\"data:image/png;base64,") + byteArray.toBase64() + "\" align=\"center\"/>";
229  }
230  }
231  }
232 
233  return imgUrl;
234 }

250 bool MeasurementDatabaseDialog::eventFilter(QObject *target, QEvent *event)
251 {
252  if (target == ui->treeWidget && event->type() == QEvent::KeyPress)
253  {
254  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
255  switch(keyEvent->key())
256  {
257  case Qt::Key_Up:
258  {
259  const QModelIndex model = ui->treeWidget->indexAbove(ui->treeWidget->currentIndex());
260  QTreeWidgetItem *item = ui->treeWidget->itemAbove(ui->treeWidget->currentItem());
261  showDescription(item, model.column());
262  break;
263  }
264  case Qt::Key_Down:
265  {
266  const QModelIndex model = ui->treeWidget->indexBelow(ui->treeWidget->currentIndex());
267  QTreeWidgetItem *item = ui->treeWidget->itemBelow(ui->treeWidget->currentItem());
268  showDescription(item, model.column());
269  break;
270  }
271  default:
272  break;
273  }
274  }
275  return QDialog::eventFilter(target, event);
276 }

279 void MeasurementDatabaseDialog::updateChecks(QTreeWidgetItem *item, int column)
280 {
281  bool diff = false;
282  if (column != 0 && column != -1)
283  {
284  return;
285  }
286 
287  if (item->childCount() != 0 && item->checkState(0) != Qt::PartiallyChecked && column != -1)
288  {
289  bool flag = false; // Check if we could change atleast one children
290  Qt::CheckState state = item->checkState(0);
291  for (int i = 0; i < item->childCount(); ++i)
292  {
293  if (!measurements.contains(item->child(i)->data(0, Qt::UserRole).toString()))
294  {
295  item->child(i)->setCheckState(0, state);
296  flag = true;
297  }
298 
299  if (flag == false) // All child in the list
300  {
301  item->setCheckState(0, Qt::Checked);
302  }
303  }
304  }
305  else if (item->childCount() == 0 || column == -1)
306  {
307 
308  QString itemName = item->data(0, Qt::UserRole).toString();
309  Qt::CheckState state = item->checkState(0);
310 
311  if (state == Qt::CheckState::Unchecked)
312  {
313  newMeasurements.removeOne(itemName);
314  }
315  else if (state == Qt::CheckState::Checked &&
316  !measurements.contains(itemName) &&
317  !newMeasurements.contains(itemName) &&
318  !itemName.isEmpty())
319  {
320  newMeasurements.append(itemName);
321  }
322 
323  QTreeWidgetItem *parent = item->parent();
324  if (parent == nullptr)
325  {
326  return;
327  }
328  for (int j = 0; j < parent->childCount(); ++j)
329  {
330  if (j != parent->indexOfChild(item)
331  && item->checkState(0) != parent->child(j)->checkState(0))
332  {
333  diff = true;
334  }
335  }
336  if (diff)
337  {
338  parent->setCheckState(0, Qt::PartiallyChecked);
339  }
340  else
341  {
342  parent->setCheckState(0, item->checkState(0));
343  }
344 
345  updateChecks(parent, -1);
346  }
347 }

551 void MeasurementDatabaseDialog::filterGroup(QTreeWidgetItem *group, const QString &searchString)
552 {
553  SCASSERT(group != nullptr)
554 
555  bool isMatch = false;
556  for (int i=0; i < group->childCount(); ++i)
557  {
558  QTreeWidgetItem *item = group->child(i);
559  //const QString description = QTextDocumentFragment::fromHtml(ItemFullDescription(item, false))

561 
562  //const bool isHidden = !item->text(0).contains(searchString, Qt::CaseInsensitive)

564 
565  const bool isHidden = !item->text(0).contains(searchString, Qt::CaseInsensitive);
566 
567  item->setHidden(isHidden);

571  }
572  }
573 
574  group->setHidden(!group->text(0).contains(searchString, Qt::CaseInsensitive) && !isMatch);
