151 /**
152  * @brief updateHandlePositions update handle positions on new parent bounding rect.
153  * @return void.
154  */
155 void ResizeHandlesItem::updateHandlePositions()
156 {
157  foreach (HandleItem *item, m_handleItems)
158  {
159  item->setFlag(ItemSendsGeometryChanges, false);
160 
161  switch (item->position())
162  {
163  case Position::TopLeft:
164  item->setPos(m_parentRect.topLeft());
165  break;
166  case Position::Top:
167  item->setPos(m_parentRect.left() + m_parentRect.width() / 2 - 1,
168  m_parentRect.top());
169  break;
170  case Position::TopRight:
171  item->setPos(m_parentRect.topRight());
172  break;
173  case Position::Right:
174  item->setPos(m_parentRect.right(),
175  m_parentRect.top() + m_parentRect.height() / 2 - 1);
176  break;
177  case Position::BottomRight:
178  item->setPos(m_parentRect.bottomRight());
179  break;
180  case Position::Bottom:
181  item->setPos(m_parentRect.left() + m_parentRect.width() / 2 - 1,
182  m_parentRect.bottom());
183  break;
184  case Position::BottomLeft:
185  item->setPos(m_parentRect.bottomLeft());
186  break;
187  case Position::Left:
188  item->setPos(m_parentRect.left(),
189  m_parentRect.top() + m_parentRect.height() / 2 - 1);
190  break;
191  case Position::Center:
192  break;
193  }
194 
195  item->setFlag(ItemSendsGeometryChanges, true);
196  }
197 
198  if(!m_sizeChangedExternally)
199  {
200  emit sizeChangedFromHandles(m_parentRect);
201  }
202  else
203  {
204  m_sizeChangedExternally = false;
205  }
206 }

482  * @brief limitPosition limits the bounds of the minimum size of the bounding rect.
483  * @param QPointF newPos new position.
484  * @return QPointF limited new position.
485  */
486 QPointF ResizeHandlesItem::HandleItem::limitPosition(const QPointF& newPos)
487 {
488  QPointF point = pos();
489  if (m_handlePosition == Position::TopLeft ||
490  m_handlePosition == Position::BottomRight ||
491  m_handlePosition == Position::TopRight ||
492  m_handlePosition == Position::BottomLeft)
493  {
494  point.setX(newPos.x());
495  point.setY(newPos.y());
496  }
497 
498  if (m_handlePosition == Position::Top || m_handlePosition == Position::Bottom)
499  {
500  point.setY(newPos.y());
501  }
502 
503  if (m_handlePosition == Position::Left || m_handlePosition == Position::Right)
504  {
505  point.setX(newPos.x());
506  }
507 
508  bool isMovingTop = m_handlePosition == Position::Top ||
509  m_handlePosition == Position::TopLeft ||
510  m_handlePosition == Position::TopRight;
511  bool isMovingBottom = m_handlePosition == Position::Bottom ||
512  m_handlePosition == Position::BottomLeft ||
513  m_handlePosition == Position::BottomRight;
514  bool isMovingLeft = m_handlePosition == Position::Left ||
515  m_handlePosition == Position::TopLeft ||
516  m_handlePosition == Position::BottomLeft;
517  bool isMovingRight = m_handlePosition == Position::Right ||
518  m_handlePosition == Position::TopRight ||
519  m_handlePosition == Position::BottomRight;
520 
521  qreal height;
522  qreal width;
523 
524  if (isMovingTop)
525  {
526  height = m_parent->m_parentRect.bottom() - point.y();
527 
528  if (height < m_minDimension)
529  {
530  point.setY(m_parent->m_parentRect.bottom() - m_minDimension);
531  }
532  else if (height > m_maxDimension)
533  {
534  point.setY(m_parent->m_parentRect.bottom() - m_maxDimension);
535  }
536 
537  if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && height * m_scalingFactor < m_minDimension)
538  {
539  point.setY(m_parent->m_parentRect.bottom() - m_minDimension / m_scalingFactor);
540  }
541  else if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && height * m_scalingFactor > m_maxDimension)
542  {
543  point.setY(m_parent->m_parentRect.bottom() - m_maxDimension / m_scalingFactor);
544  }
545  }
546  else if (isMovingBottom)
547  {
548  height = point.y() - m_parent->m_parentRect.top();
549 
550  if (height < m_minDimension)
551  {
552  point.setY(m_parent->m_parentRect.top() + m_minDimension);
553  }
554  else if (height > m_maxDimension)
555  {
556  point.setY(m_parent->m_parentRect.top() + m_maxDimension);
557  }
558 
559  if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && height * m_scalingFactor < m_minDimension)
560  {
561  point.setY(m_parent->m_parentRect.top() + m_minDimension / m_scalingFactor);
562  }
563  else if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && height * m_scalingFactor > m_maxDimension)
564  {
565  point.setY(m_parent->m_parentRect.top() + m_maxDimension / m_scalingFactor);
566  }
567  }
568 
569  if (isMovingLeft)
570  {
571  width = m_parent->m_parentRect.right() - point.x();
572 
573  if (width < m_minDimension)
574  {
575  point.setX(m_parent->m_parentRect.right() - m_minDimension);
576  }
577  else if (width > m_maxDimension)
578  {
579  point.setX(m_parent->m_parentRect.right() - m_maxDimension);
580  }
581 
582  if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && width / m_scalingFactor < m_minDimension)
583  {
584  point.setX(m_parent->m_parentRect.right() - m_minDimension * m_scalingFactor);
585  }
586  else if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && width / m_scalingFactor > m_maxDimension)
587  {
588  point.setX(m_parent->m_parentRect.right() - m_maxDimension * m_scalingFactor);
589  }
590  }
591  else if (isMovingRight)
592  {
593  width = point.x() - m_parent->m_parentRect.left();
594 
595  if (width < m_minDimension)
596  {
597  point.setX(m_parent->m_parentRect.left() + m_minDimension);
598  }
599  else if (width > m_maxDimension)
600  {
601  point.setX(m_parent->m_parentRect.left() + m_maxDimension);
602  }
603 
604  if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && width / m_scalingFactor < m_minDimension)
605  {
606  point.setX(m_parent->m_parentRect.left() + m_minDimension * m_scalingFactor);
607  }
608  else if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && width / m_scalingFactor > m_maxDimension)
609  {
610  point.setX(m_parent->m_parentRect.left() + m_maxDimension * m_scalingFactor);
611  }
612  }
613 
614  return point;
615 }
