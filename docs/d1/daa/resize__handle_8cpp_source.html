150 /**
151  * @brief updateHandlePositions update handle positions on new parent bounding rect.
152  * @return void.
153  */
154 void ResizeHandlesItem::updateHandlePositions()
155 {
156  foreach (HandleItem *item, m_handleItems)
157  {
158  item->setFlag(ItemSendsGeometryChanges, false);
159 
160  switch (item->position())
161  {
162  case Position::TopLeft:
163  item->setPos(m_parentRect.topLeft());
164  break;
165  case Position::Top:
166  item->setPos(m_parentRect.left() + m_parentRect.width() / 2 - 1,
167  m_parentRect.top());
168  break;
169  case Position::TopRight:
170  item->setPos(m_parentRect.topRight());
171  break;
172  case Position::Right:
173  item->setPos(m_parentRect.right(),
174  m_parentRect.top() + m_parentRect.height() / 2 - 1);
175  break;
176  case Position::BottomRight:
177  item->setPos(m_parentRect.bottomRight());
178  break;
179  case Position::Bottom:
180  item->setPos(m_parentRect.left() + m_parentRect.width() / 2 - 1,
181  m_parentRect.bottom());
182  break;
183  case Position::BottomLeft:
184  item->setPos(m_parentRect.bottomLeft());
185  break;
186  case Position::Left:
187  item->setPos(m_parentRect.left(),
188  m_parentRect.top() + m_parentRect.height() / 2 - 1);
189  break;
190  case Position::Center:
191  break;
192  }
193 
194  item->setFlag(ItemSendsGeometryChanges, true);
195  }
196 
197  emit sizeChanged(m_parentRect);
198 }

475  * @param QPointF newPos new position.
476  * @return QPointF limited new position.
477  */
478 QPointF ResizeHandlesItem::HandleItem::limitPosition(const QPointF& newPos)
479 {
480  QPointF point = pos();
481  if (m_handlePosition == Position::TopLeft ||
482  m_handlePosition == Position::BottomRight ||
483  m_handlePosition == Position::TopRight ||
484  m_handlePosition == Position::BottomLeft)
485  {
486  point.setX(newPos.x());
487  point.setY(newPos.y());
488  }
489 
490  if (m_handlePosition == Position::Top || m_handlePosition == Position::Bottom)
491  {
492  point.setY(newPos.y());
493  }
494 
495  if (m_handlePosition == Position::Left || m_handlePosition == Position::Right)
496  {
497  point.setX(newPos.x());
498  }
499 
500  bool isMovingTop = m_handlePosition == Position::Top ||
501  m_handlePosition == Position::TopLeft ||
502  m_handlePosition == Position::TopRight;
503  bool isMovingBottom = m_handlePosition == Position::Bottom ||
504  m_handlePosition == Position::BottomLeft ||
505  m_handlePosition == Position::BottomRight;
506  bool isMovingLeft = m_handlePosition == Position::Left ||
507  m_handlePosition == Position::TopLeft ||
508  m_handlePosition == Position::BottomLeft;
509  bool isMovingRight = m_handlePosition == Position::Right ||
510  m_handlePosition == Position::TopRight ||
511  m_handlePosition == Position::BottomRight;
512 
513  qreal height;
514  qreal width;
515 
516  if (isMovingTop)
517  {
518  height = m_parent->m_parentRect.bottom() - point.y();
519 
520  if (height < m_minDimension)
521  {
522  point.setY(m_parent->m_parentRect.bottom() - m_minDimension);
523  }
524  else if (height > m_maxDimension)
525  {
526  point.setY(m_parent->m_parentRect.bottom() - m_maxDimension);
527  }
528 
529  if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && height * m_scalingFactor < m_minDimension)
530  {
531  point.setY(m_parent->m_parentRect.bottom() - m_minDimension / m_scalingFactor);
532  }
533  else if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && height * m_scalingFactor > m_maxDimension)
534  {
535  point.setY(m_parent->m_parentRect.bottom() - m_maxDimension / m_scalingFactor);
536  }
537  }
538  else if (isMovingBottom)
539  {
540  height = point.y() - m_parent->m_parentRect.top();
541 
542  if (height < m_minDimension)
543  {
544  point.setY(m_parent->m_parentRect.top() + m_minDimension);
545  }
546  else if (height > m_maxDimension)
547  {
548  point.setY(m_parent->m_parentRect.top() + m_maxDimension);
549  }
550 
551  if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && height * m_scalingFactor < m_minDimension)
552  {
553  point.setY(m_parent->m_parentRect.top() + m_minDimension / m_scalingFactor);
554  }
555  else if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && height * m_scalingFactor > m_maxDimension)
556  {
557  point.setY(m_parent->m_parentRect.top() + m_maxDimension / m_scalingFactor);
558  }
559  }
560 
561  if (isMovingLeft)
562  {
563  width = m_parent->m_parentRect.right() - point.x();
564 
565  if (width < m_minDimension)
566  {
567  point.setX(m_parent->m_parentRect.right() - m_minDimension);
568  }
569  else if (width > m_maxDimension)
570  {
571  point.setX(m_parent->m_parentRect.right() - m_maxDimension);
572  }
573 
574  if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && width / m_scalingFactor < m_minDimension)
575  {
576  point.setX(m_parent->m_parentRect.right() - m_minDimension * m_scalingFactor);
577  }
578  else if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && width / m_scalingFactor > m_maxDimension)
579  {
580  point.setX(m_parent->m_parentRect.right() - m_maxDimension * m_scalingFactor);
581  }
582  }
583  else if (isMovingRight)
584  {
585  width = point.x() - m_parent->m_parentRect.left();
586 
587  if (width < m_minDimension)
588  {
589  point.setX(m_parent->m_parentRect.left() + m_minDimension);
590  }
591  else if (width > m_maxDimension)
592  {
593  point.setX(m_parent->m_parentRect.left() + m_maxDimension);
594  }
595 
596  if (m_parent->m_lockAspectRatio && m_scalingFactor > 1 && width / m_scalingFactor < m_minDimension)
597  {
598  point.setX(m_parent->m_parentRect.left() + m_minDimension * m_scalingFactor);
599  }
600  else if (m_parent->m_lockAspectRatio && m_scalingFactor < 1 && width / m_scalingFactor > m_maxDimension)
601  {
602  point.setX(m_parent->m_parentRect.left() + m_maxDimension * m_scalingFactor);
603  }
604  }
605 
606  return point;
607 }
