133 //---------------------------------------------------------------------------------------------------------------------
134 IntersectCircleTangentTool *IntersectCircleTangentTool::Create(const quint32 _id, const QString &pointName,
135  quint32 circleCenterId, QString &circleRadius,
136  quint32 tangentPointId,
137  CrossCirclesPoint crossPoint,
138  qreal mx, qreal my,
139  bool showPointName,
140  VMainGraphicsScene *scene,
141  VAbstractPattern *doc, VContainer *data,
142  const Document &parse,
143  const Source &typeCreation)
144 {
145  const qreal radius = qApp->toPixel(CheckFormula(_id, circleRadius, data));
146  const VPointF cPoint = *data->GeometricObject<VPointF>(circleCenterId);
147  const VPointF tPoint = *data->GeometricObject<VPointF>(tangentPointId);
148 
149  const QPointF point = IntersectCircleTangentTool::FindPoint(static_cast<QPointF>(tPoint),
150  static_cast<QPointF>(cPoint), radius, crossPoint);
151 
152  if (point == QPointF())
153  {
154  const QString msg = tr("<b><big>Can't find intersection point %1 of</big></b><br>"
155  "<b><big>Circle and Tangent</big></b><br><br>"
156  "Using origin point as a place holder until pattern is corrected.")
157  .arg(pointName);
158 
159  QMessageBox msgBox(qApp->getMainWindow());
160  msgBox.setWindowTitle(tr("Intersect Circle and Tangent"));
161  msgBox.setWindowFlags(msgBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
162  msgBox.setWindowIcon(QIcon(":/toolicon/32x32/point_from_circle_and_tangent.png"));
163  msgBox.setIcon(QMessageBox::Warning);
164  msgBox.setText(msg);
165  msgBox.setStandardButtons(QMessageBox::Ok);
166  msgBox.exec();
167  }
168 
169  quint32 id = _id;
170 
171  VPointF *p = new VPointF(point, pointName, mx, my);
172  p->setShowPointName(showPointName);
173 
174  if (typeCreation == Source::FromGui)
175  {
176  id = data->AddGObject(p);
177  }
178  else
179  {
180  data->UpdateGObject(id, p);
181  if (parse != Document::FullParse)
182  {
183  doc->UpdateToolData(id, data);
184  }
185  }
186 
187  if (parse == Document::FullParse)
188  {
189  VDrawTool::AddRecord(id, Tool::PointFromCircleAndTangent, doc);
190  IntersectCircleTangentTool *point = new IntersectCircleTangentTool(doc, data, id, circleCenterId,
191  circleRadius, tangentPointId,
192  crossPoint, typeCreation);
193  scene->addItem(point);
194  InitToolConnections(scene, point);
195  VAbstractPattern::AddTool(id, point);
196  doc->IncrementReferens(cPoint.getIdTool());
197  doc->IncrementReferens(tPoint.getIdTool());
198  return point;
199  }
200  return nullptr;
201 }

204 QPointF IntersectCircleTangentTool::FindPoint(const QPointF &p, const QPointF &center, qreal radius,
205  const CrossCirclesPoint crossPoint)
206 {
207  QPointF p1, p2;
208  const int res = VGObject::ContactPoints (p, center, radius, p1, p2);
209 
210  switch(res)
211  {
212  case 2:
213  if (crossPoint == CrossCirclesPoint::FirstPoint)
214  {
215  return p1;
216  }
217  else
218  {
219  return p2;
220  }
221  case 1:
222  return p1;
223  case 3:
224  case 0:
225  default:
226  return QPointF();
227  }
228 }
