199  m_bBuiltInOp = a_Parser.m_bBuiltInOp;

204  m_vStringVarBuf = a_Parser.m_vStringVarBuf;
205  m_nIfElseCounter = a_Parser.m_nIfElseCounter;
206  m_pTokenReader.reset(a_Parser.m_pTokenReader->Clone(this));
207 
208  // Copy function and operator callbacks
209  m_FunDef = a_Parser.m_FunDef; // Copy function definitions
210  m_PostOprtDef = a_Parser.m_PostOprtDef; // post value unary operators
211  m_InfixOprtDef = a_Parser.m_InfixOprtDef; // unary operators for infix notation

359  * @brief Add a function or operator callback to the parser.
360  */
361 void QmuParserBase::AddCallback(const QString &a_strName, const QmuParserCallback &a_Callback,
362  funmap_type &a_Storage, const QString &a_szCharSet )
363 {
364  if (a_Callback.GetAddr() == nullptr)
365  {
366  Error(ecINVALID_FUN_PTR);
367  }
368 
369  const funmap_type *pFunMap = &a_Storage;
370 
371  // Check for conflicting operator or function names
372  if ( pFunMap!=&m_FunDef && m_FunDef.find(a_strName)!=m_FunDef.end() )
373  {
374  Error(ecNAME_CONFLICT, -1, a_strName);
375  }
376 
377  if ( pFunMap!=&m_PostOprtDef && m_PostOprtDef.find(a_strName)!=m_PostOprtDef.end() )
378  {
379  Error(ecNAME_CONFLICT, -1, a_strName);
380  }
381 
382  if ( pFunMap!=&m_InfixOprtDef && pFunMap!=&m_OprtDef && m_InfixOprtDef.find(a_strName)!=m_InfixOprtDef.end() )
383  {
384  Error(ecNAME_CONFLICT, -1, a_strName);
385  }
386 
387  if ( pFunMap!=&m_InfixOprtDef && pFunMap!=&m_OprtDef && m_OprtDef.find(a_strName)!=m_OprtDef.end() )
388  {
389  Error(ecNAME_CONFLICT, -1, a_strName);
390  }
391 
392  CheckOprt(a_strName, a_Callback, a_szCharSet);
393  a_Storage[a_strName] = a_Callback;
394  ReInit();
395 }

403 // cppcheck-suppress
404 void QmuParserBase::CheckOprt(const QString &a_sName, const QmuParserCallback &a_Callback,
405  const QString &a_szCharSet) const
406 {
407  const std::wstring a_sNameStd = a_sName.toStdWString();
408  const std::wstring a_szCharSetStd = a_szCharSet.toStdWString();
409 
410  if ( a_sNameStd.length() == false || (a_sNameStd.find_first_not_of(a_szCharSetStd)!=string_type::npos) ||
411  (a_sNameStd.at(0)>='0' && a_sNameStd.at(0)<='9'))
412  {
413  switch (a_Callback.GetCode())
414  {
415  case cmOPRT_POSTFIX:
416  Error(ecINVALID_POSTFIX_IDENT, -1, a_sName);
417  break;
418  case cmOPRT_INFIX:
419  Error(ecINVALID_INFIX_IDENT, -1, a_sName);
420  break;
421  default:
422  Error(ecINVALID_NAME, -1, a_sName);
423  break;
424  }
425  }
426 }

642  * @throw ParserException if a_Oprt is no operator code
643  */
644 int QmuParserBase::GetOprtPrecedence(const token_type &a_Tok) const
645 {
646  switch (a_Tok.GetCode())
647  {
648  // built in operators
649  case cmEND:
650  return -5;
651  case cmARG_SEP:
652  return -4;
653  case cmASSIGN:
654  return -1;
655  case cmELSE:
656  case cmIF:
657  return 0;
658  case cmLAND:
659  return prLAND;
660  case cmLOR:
661  return prLOR;
662  case cmLT:
663  case cmGT:
664  case cmLE:
665  case cmGE:
666  case cmNEQ:
667  case cmEQ:
668  return prCMP;
669  case cmADD:
670  case cmSUB:
671  return prADD_SUB;
672  case cmMUL:
673  case cmDIV:
674  return prMUL_DIV;
675  case cmPOW:
676  return prPOW;
677  // user defined binary operators
678  case cmOPRT_INFIX:
679  case cmOPRT_BIN:
680  return a_Tok.GetPri();
681  default:
682  Error(ecINTERNAL_ERROR, 5);
683  return 999;
684  }

801 /**
802  * @brief Apply a function token.
803  * @param a_iArgCount Number of Arguments actually gathered used only for multiarg functions.
804  * @post The result is pushed to the value stack
805  * @post The function token is removed from the stack
806  * @throw QmuParserError if Argument count does not mach function requirements.
807  */
808 void QmuParserBase::ApplyFunc( QStack<token_type> &a_stOpt, QStack<token_type> &a_stVal, int a_iArgCount) const
809 {
810  assert(m_pTokenReader.get());
811 
812  // Operator stack empty or does not contain tokens with callback functions
813  if (a_stOpt.empty() || a_stOpt.top().GetFuncAddr() == nullptr)
814  {
815  return;
816  }
817 
818  token_type funTok = a_stOpt.pop();
819  assert(funTok.GetFuncAddr());
820 
821  // Binary operators must rely on their internal operator number
822  // since counting of operators relies on commas for function arguments
823  // binary operators do not have commas in their expression
824  int iArgCount = (funTok.GetCode()==cmOPRT_BIN) ? funTok.GetArgCount() : a_iArgCount;
825 
826  // determine how many parameters the function needs. To remember iArgCount includes the
827  // string parameter whilst GetArgCount() counts only numeric parameters.
828  int iArgRequired = funTok.GetArgCount() + ((funTok.GetType()==tpSTR) ? 1 : 0);
829 
830  // Thats the number of numerical parameters
831  int iArgNumerical = iArgCount - ((funTok.GetType()==tpSTR) ? 1 : 0);
832 
833  if (funTok.GetCode()==cmFUNC_STR && iArgCount-iArgNumerical>1)
834  {
835  Error(ecINTERNAL_ERROR);
836  }
837 
838  if (funTok.GetArgCount()>=0 && iArgCount>iArgRequired)
839  {
840  Error(ecTOO_MANY_PARAMS, m_pTokenReader->GetPos()-1, funTok.GetAsString());
841  }
842 
843  if (funTok.GetCode()!=cmOPRT_BIN && iArgCount<iArgRequired )
844  {
845  Error(ecTOO_FEW_PARAMS, m_pTokenReader->GetPos()-1, funTok.GetAsString());
846  }
847 
848  if (funTok.GetCode()==cmFUNC_STR && iArgCount>iArgRequired )
849  {
850  Error(ecTOO_MANY_PARAMS, m_pTokenReader->GetPos()-1, funTok.GetAsString());
851  }
852 
853  // Collect the numeric function arguments from the value stack and store them
854  // in a vector
855  QVector<token_type> stArg;
856  for (int i=0; i<iArgNumerical; ++i)
857  {
858  if (a_stVal.isEmpty())// Check if stack is empty like in origin muparser.
859  {
860  Error(ecUNASSIGNABLE_TOKEN, m_pTokenReader->GetPos(), funTok.GetAsString());
861  }
862  stArg.push_back( a_stVal.pop() );
863  if ( stArg.back().GetType()==tpSTR && funTok.GetType()!=tpSTR )
864  {
865  Error(ecVAL_EXPECTED, m_pTokenReader->GetPos(), funTok.GetAsString());
866  }
867  }
868 
869  switch (funTok.GetCode())
870  {
871  case cmFUNC_STR:
872  stArg.push_back(a_stVal.pop());
873 
874  if ( stArg.back().GetType()==tpSTR && funTok.GetType()!=tpSTR )
875  {
876  Error(ecVAL_EXPECTED, m_pTokenReader->GetPos(), funTok.GetAsString());
877  }
878 
879  ApplyStrFunc(funTok, stArg);
880  break;
881  case cmFUNC_BULK:
882  m_vRPN.AddBulkFun(funTok.GetFuncAddr(), stArg.size());
883  break;
884  case cmOPRT_BIN:
885  case cmOPRT_POSTFIX:
886  case cmOPRT_INFIX:
887  case cmFUNC:
888  if (funTok.GetArgCount()==-1 && iArgCount==0)
889  {
890  Error(ecTOO_FEW_PARAMS, m_pTokenReader->GetPos(), funTok.GetAsString());
891  }
892 
893  m_vRPN.AddFun(funTok.GetFuncAddr(), (funTok.GetArgCount()==-1) ? -iArgNumerical : iArgNumerical);
894  break;
895  default:
896  break;
897  }
898  // Push dummy value representing the function result to the stack
899  token_type token;
900  token.SetVal(1);
901  a_stVal.push(token);
902 }

905 void QmuParserBase::ApplyIfElse(QStack<token_type> &a_stOpt, QStack<token_type> &a_stVal) const
906 {
907  // Check if there is an if Else clause to be calculated
908  while (a_stOpt.size() && a_stOpt.top().GetCode()==cmELSE)
909  {
910  token_type opElse = a_stOpt.pop();
911  Q_ASSERT(a_stOpt.size()>0);
912 
913  // Take the value associated with the else branch from the value stack
914  token_type vVal2 = a_stVal.pop();
915 
916  Q_ASSERT(a_stOpt.size()>0);
917  Q_ASSERT(a_stVal.size()>=2);
918 
919  // it then else is a ternary operator Pop all three values from the value s
920  // tack and just return the right value
921  token_type vVal1 = a_stVal.pop();
922  token_type vExpr = a_stVal.pop();
923 
924  a_stVal.push( not qFuzzyIsNull(vExpr.GetVal()) ? vVal1 : vVal2);
925 
926  token_type opIf = a_stOpt.pop();
927  Q_ASSERT(opElse.GetCode()==cmELSE);
928  Q_ASSERT(opIf.GetCode()==cmIF);
929 
930  m_vRPN.AddIfElse(cmENDIF);
931  } // while pending if-else-clause found
932 }

935 /**
936  * @brief Performs the necessary steps to write code for the execution of binary operators into the bytecode.
937  */
938 void QmuParserBase::ApplyBinOprt(QStack<token_type> &a_stOpt, QStack<token_type> &a_stVal) const
939 {
940  // is it a user defined binary operator?
941  if (a_stOpt.top().GetCode()==cmOPRT_BIN)
942  {
943  ApplyFunc(a_stOpt, a_stVal, 2);
944  }
945  else
946  {
947  if(a_stVal.size()<2)
948  {
949  Error(ecUNEXPECTED_OPERATOR);
950  }
951 
952  token_type valTok1 = a_stVal.pop(),
953  valTok2 = a_stVal.pop(),
954  optTok = a_stOpt.pop(),
955  resTok;
956 
957  if ( valTok1.GetType()!=valTok2.GetType() || (valTok1.GetType()==tpSTR && valTok2.GetType()==tpSTR) )
958  {
959  Error(ecOPRT_TYPE_CONFLICT, m_pTokenReader->GetPos(), optTok.GetAsString());
960  }
961 
962  if (optTok.GetCode()==cmASSIGN)
963  {
964  if (valTok2.GetCode()!=cmVAR)
965  {
966  Error(ecUNEXPECTED_OPERATOR, -1, "=");
967  }
968  m_vRPN.AddAssignOp(valTok2.GetVar());
969  }
970  else
971  {
972  m_vRPN.AddOp(optTok.GetCode());
973  }
974  resTok.SetVal(1);
975  a_stVal.push(resTok);
976  }
977 }

983  * @param stVal The value stack

1003  {

1005  }

1009  }
1010  }

1074  continue;
1075  case cmSUB:

1077  Stack[sidx] -= Stack[1+sidx];
1078  continue;
1079  case cmMUL:

1368 void QmuParserBase::CreateRPN() const
1369 {
1370  if (m_pTokenReader->GetExpr().length() == false)
1371  {
1372  Error(ecUNEXPECTED_EOF, 0);
1373  }
1374 
1375  QStack<token_type> stOpt, stVal;
1376  QStack<int> stArgCount;
1377  token_type opta, opt; // for storing operators
1378  //token_type val, tval; // for storing value
1379  //string_type strBuf; // buffer for string function arguments
1380 
1381  ReInit();
1382 
1383  // The outermost counter counts the number of seperated items
1384  // such as in "a=10,b=20,c=c+a"
1385  stArgCount.push(1);
1386 
1387  for (;;)
1388  {
1389  opt = m_pTokenReader->ReadNextToken(m_locale, m_decimalPoint, m_thousandsSeparator);
1390 
1391  switch (opt.GetCode())
1392  {
1393  //
1394  // Next three are different kind of value entries
1395  //
1396  case cmSTRING:
1397  {
1398  opt.SetIdx(m_vStringBuf.size()); // Assign buffer index to token
1399  stVal.push(opt);
1400  const QString &str = opt.GetAsString();
1401  m_vStringBuf.push_back(str); // Store string in internal buffer
1402  m_Tokens.insert(m_pTokenReader->GetPos()-str.length(), str);
1403  break;
1404  }
1405  case cmVAR:
1406  {
1407  stVal.push(opt);
1408  m_vRPN.AddVar( static_cast<qreal*>(opt.GetVar()) );
1409  const QString &str = opt.GetAsString();
1410  m_Tokens.insert(m_pTokenReader->GetPos()-str.length(), str);
1411  break;
1412  }
1413  case cmVAL:
1414  {
1415  stVal.push(opt);
1416  m_vRPN.AddVal( opt.GetVal() );
1417  const QString &str = opt.GetAsString();
1418  m_Numbers.insert(m_pTokenReader->GetPos()-str.length(), str);
1419  break;
1420  }
1421  case cmELSE:
1422  m_nIfElseCounter--;
1423  if (m_nIfElseCounter<0)
1424  {
1425  Error(ecMISPLACED_COLON, m_pTokenReader->GetPos());
1426  }
1427  ApplyRemainingOprt(stOpt, stVal);
1428  m_vRPN.AddIfElse(cmELSE);
1429  stOpt.push(opt);
1430  break;
1431  case cmARG_SEP:
1432  if (stArgCount.empty())
1433  {
1434  Error(ecUNEXPECTED_ARG_SEP, m_pTokenReader->GetPos());
1435  }
1436  if (stOpt.empty() && allowSubexpressions == false)
1437  {
1438  Error(ecUNEXPECTED_ARG_SEP, m_pTokenReader->GetPos());
1439  }
1440  ++stArgCount.top();
1441  // fallthrough intentional (no break!)
1442  QMUP_FALLTHROUGH
1443  case cmEND:
1444  ApplyRemainingOprt(stOpt, stVal);
1445  break;
1446  case cmBC:
1447  {
1448  // The argument count for parameterless functions is zero
1449  // by default an opening bracket sets parameter count to 1
1450  // in preparation of arguments to come. If the last token
1451  // was an opening bracket we know better...
1452  if (opta.GetCode()==cmBO)
1453  {
1454  --stArgCount.top();
1455  }
1456 
1457  ApplyRemainingOprt(stOpt, stVal);
1458 
1459  // Check if the bracket content has been evaluated completely
1460  if (stOpt.size() && stOpt.top().GetCode()==cmBO)
1461  {
1462  // if opt is ")" and opta is "(" the bracket has been evaluated, now its time to check
1463  // if there is either a function or a sign pending
1464  // neither the opening nor the closing bracket will be pushed back to
1465  // the operator stack
1466  // Check if a function is standing in front of the opening bracket,
1467  // if yes evaluate it afterwards check for infix operators
1468  assert(stArgCount.size());
1469  int iArgCount = stArgCount.pop();
1470 
1471  stOpt.pop(); // Take opening bracket from stack
1472 
1473  if (iArgCount>1 && ( stOpt.size()==0 || (stOpt.top().GetCode()!=cmFUNC &&
1474  stOpt.top().GetCode()!=cmFUNC_BULK &&
1475  stOpt.top().GetCode()!=cmFUNC_STR) ) )
1476  {
1477  Error(ecUNEXPECTED_ARG, m_pTokenReader->GetPos());
1478  }
1479 
1480  // The opening bracket was popped from the stack now check if there
1481  // was a function before this bracket
1482  if (stOpt.size() && stOpt.top().GetCode()!=cmOPRT_INFIX && stOpt.top().GetCode()!=cmOPRT_BIN &&
1483  stOpt.top().GetFuncAddr()!=nullptr)
1484  {
1485  ApplyFunc(stOpt, stVal, iArgCount);
1486  }
1487  }
1488  } // if bracket content is evaluated
1489  break;
1490  //
1491  // Next are the binary operator entries
1492  //
1493  //case cmAND: // built in binary operators
1494  //case cmOR:
1495  //case cmXOR:
1496  case cmIF:
1497  m_nIfElseCounter++;
1498  // fallthrough intentional (no break!)
1499  QMUP_FALLTHROUGH
1500  case cmLE:
1501  case cmGE:
1502  case cmNEQ:
1503  case cmEQ:
1504  case cmLT:
1505  case cmGT:
1506  case cmADD:
1507  case cmSUB:
1508  case cmMUL:
1509  case cmDIV:
1510  case cmPOW:
1511  case cmLAND:
1512  case cmLOR:
1513  case cmASSIGN:
1514  case cmOPRT_BIN:
1515  // A binary operator (user defined or built in) has been found.
1516  while ( stOpt.size() && stOpt.top().GetCode() != cmBO && stOpt.top().GetCode() != cmELSE &&
1517  stOpt.top().GetCode() != cmIF)
1518  {
1519  const token_type &topToken = stOpt.top();
1520  int nPrec1 = GetOprtPrecedence(topToken),
1521  nPrec2 = GetOprtPrecedence(opt);
1522 
1523  const ECmdCode code = topToken.GetCode();
1524  if (code==opt.GetCode())
1525  {
1526  // Deal with operator associativity
1527  EOprtAssociativity eOprtAsct = GetOprtAssociativity(opt);
1528  if ( (eOprtAsct==oaRIGHT && (nPrec1 <= nPrec2)) ||
1529  (eOprtAsct==oaLEFT && (nPrec1 < nPrec2)) )
1530  {
1531  break;
1532  }
1533  }
1534  else if (nPrec1 < nPrec2)
1535  {
1536  // In case the operators are not equal the precedence decides alone...
1537  break;
1538  }
1539  if (code==cmOPRT_INFIX)
1540  {
1541  ApplyFunc(stOpt, stVal, 1);
1542  }
1543  else
1544  {
1545  ApplyBinOprt(stOpt, stVal);
1546  }
1547  } // while ( ... )
1548 
1549  if (opt.GetCode()==cmIF)
1550  {
1551  m_vRPN.AddIfElse(opt.GetCode());
1552  }
1553 
1554  // The operator can't be evaluated right now, push back to the operator stack
1555  stOpt.push(opt);
1556  break;
1557  //
1558  // Last section contains functions and operators implicitely mapped to functions
1559  //
1560  case cmBO:
1561  stArgCount.push(1);
1562  stOpt.push(opt);
1563  break;
1564  case cmOPRT_INFIX:
1565  case cmFUNC:
1566  case cmFUNC_BULK:
1567  case cmFUNC_STR:
1568  stOpt.push(opt);
1569  m_Tokens.insert(m_pTokenReader->GetPos()-opt.GetAsString().length(), opt.GetAsString());
1570  break;
1571  case cmOPRT_POSTFIX:
1572  stOpt.push(opt);
1573  ApplyFunc(stOpt, stVal, 1); // this is the postfix operator
1574  m_Tokens.insert(m_pTokenReader->GetPos()-opt.GetAsString().length(), opt.GetAsString());
1575  break;
1576  case cmENDIF:
1577  case cmVARPOW2:
1578  case cmVARPOW3:
1579  case cmVARPOW4:
1580  case cmVARMUL:
1581  case cmPOW2:
1582  case cmUNKNOWN:
1583  default:
1584  Error(ecINTERNAL_ERROR, 3);
1585  } // end of switch operator-token
1586 
1587  opta = opt;
1588 
1589  if ( opt.GetCode() == cmEND )
1590  {
1591  m_vRPN.Finalize();
1592  break;
1593  }
1594 
1595  if (QmuParserBase::g_DbgDumpStack)
1596  {
1597  StackDump(stVal, stOpt);
1598  m_vRPN.AsciiDump();
1599  }
1600  } // while (true)
1601 
1602  if (QmuParserBase::g_DbgDumpCmdCode)
1603  {
1604  m_vRPN.AsciiDump();
1605  }
1606 
1607  if (m_nIfElseCounter>0)
1608  {
1609  Error(ecMISSING_ELSE_CLAUSE);
1610  }
1611 
1612  // get the last value (= final result) from the stack
1613  Q_ASSERT(stArgCount.size()==1);
1614  m_nFinalResultIdx = stArgCount.top();
1615  if (m_nFinalResultIdx==0)
1616  {
1617  Error(ecINTERNAL_ERROR, 9);
1618  }
1619 
1620  if (stVal.size()==0)
1621  {
1622  Error(ecEMPTY_EXPRESSION);
1623  }
1624 
1625  if (stVal.top().GetType()!=tpDBL)
1626  {
1627  Error(ecSTR_RESULT);
1628  }
1629 
1630  m_vStackBuffer.resize(m_vRPN.GetMaxStackSize() * s_MaxNumOpenMPThreads);
1631 }
