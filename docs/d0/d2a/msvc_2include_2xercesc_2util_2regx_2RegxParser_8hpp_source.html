257 inline int RegxParser::hexChar(const XMLInt32 ch) {
258 
259  if (ch < chDigit_0 || ch > chLatin_f)
260  return -1;
261 
262  if (ch <= chDigit_9)
263  return ch - chDigit_0;
264 
265  if (ch < chLatin_A)
266  return -1;
267 
268  if (ch <= chLatin_F)
269  return ch - chLatin_A + 10;
270 
271  if (ch < chLatin_a)
272  return -1;
273 
274  return ch - chLatin_a + 10;
275 }
