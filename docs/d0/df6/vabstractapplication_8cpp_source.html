268 void VAbstractApplication::loadTranslations(const QString &locale)
269 {
270  if (locale.isEmpty())
271  {
272  qInfo() << "Locale is empty.";
273  return;
274  }
275  qInfo() << "Checked locale:" << locale;
276 
277  ClearTranslation();
278 
279  qtTranslator = new QTranslator(this);
280  qtxmlTranslator = new QTranslator(this);
281  qtBaseTranslator = new QTranslator(this);
282  appTranslator = new QTranslator(this);
283  pmsTranslator = new QTranslator(this);
284 
285 #if defined(Q_OS_WIN) || defined(Q_OS_MAC)
286  qtTranslator->load("qt_" + locale, translationsPath(locale));
287  qtxmlTranslator->load("qtxmlpatterns_" + locale, translationsPath(locale));
288  qtBaseTranslator->load("qtbase_" + locale, translationsPath(locale));
289 #else
290  qtTranslator->load("qt_" + locale, QLibraryInfo::location(QLibraryInfo::TranslationsPath));
291  qtxmlTranslator->load("qtxmlpatterns_" + locale, QLibraryInfo::location(QLibraryInfo::TranslationsPath));
292  qtBaseTranslator->load("qtbase_" + locale, QLibraryInfo::location(QLibraryInfo::TranslationsPath));
293 #endif
294 
295  appTranslator->load("seamly2d_" + locale, translationsPath(locale));
296  pmsTranslator->load("measurements_" + locale, translationsPath(locale));
297 
298  installTranslator(qtTranslator);
299  installTranslator(qtxmlTranslator);
300  installTranslator(qtBaseTranslator);
301  installTranslator(appTranslator);
302  installTranslator(pmsTranslator);
303 
304  InitTrVars();//Very important do it after load QM files.
305 }
