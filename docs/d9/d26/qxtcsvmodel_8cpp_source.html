186 /*!
187  Reads in a CSV file from the provided \a file using \a codec.
188 
189  The value of \a separator will be used to delimit fields, subject to the specified \a quoteMode.
190  If \a withHeader is set to true, the first line of the file will be used to populate the model's
191  horizontal header.
192 
193  \sa quoteMode
194  */
195 void QxtCsvModel::setSource(QIODevice *file, bool withHeader, QChar separator, QTextCodec* codec)
196 {
197  QxtCsvModelPrivate* d_ptr = &qxt_d();
198  bool headerSet = !withHeader;
199  if (not file->isOpen())
200  {
201  file->open(QIODevice::ReadOnly);
202  }
203  if (withHeader)
204  {
205  d_ptr->maxColumn = 0;
206  }
207  else
208  {
209  d_ptr->maxColumn = d_ptr->header.size();
210  }
211  d_ptr->csvData.clear();
212  QStringList row;
213  QString field;
214  QChar quote;
215  QChar ch, buffer(0);
216  bool readCR = false;
217  QTextStream stream(file);
218  std::optional<QStringConverter::Encoding> text_encoding = QStringConverter::encodingForName(codec->name());
219  if (text_encoding.has_value())
220  {
221  stream.setEncoding(text_encoding.value());
222  }
223  else
224  {
225  stream.setAutoDetectUnicode(true);
226  }
227  while (not stream.atEnd())
228  {
229  if (buffer != QChar(0))
230  {
231  ch = buffer;
232  buffer = QChar(0);
233  }
234  else
235  {
236  stream >> ch;
237  }
238  if (ch == '\n' && readCR)
239  {
240  continue;
241  }
242  else if (ch == '\r')
243  {
244  readCR = true;
245  }
246  else
247  {
248  readCR = false;
249  }
250  if (ch != separator && (ch.category() == QChar::Separator_Line || ch.category() == QChar::Separator_Paragraph
251  || ch.category() == QChar::Other_Control))
252  {
253  row << field;
254  field.clear();
255  if (not row.isEmpty())
256  {
257  if (not headerSet)
258  {
259  d_ptr->header = row;
260  headerSet = true;
261  }
262  else
263  {
264  d_ptr->csvData.append(row);
265  }
266  if (row.length() > d_ptr->maxColumn)
267  {
268  d_ptr->maxColumn = row.length();
269  }
270  }
271  row.clear();
272  }
273  else if ((d_ptr->quoteMode & DoubleQuote && ch == '"') || (d_ptr->quoteMode & SingleQuote && ch == '\''))
274  {
275  quote = ch;
276  do
277  {
278  stream >> ch;
279  if (ch == '\\' && d_ptr->quoteMode & BackslashEscape)
280  {
281  stream >> ch;
282  }
283  else if (ch == quote)
284  {
285  if (d_ptr->quoteMode & TwoQuoteEscape)
286  {
287  stream >> buffer;
288  if (buffer == quote)
289  {
290  buffer = QChar(0);
291  field.append(ch);
292  continue;
293  }
294  }
295  break;
296  }
297  field.append(ch);
298  } while (!stream.atEnd());
299  }
300  else if (ch == separator)
301  {
302  row << field;
303  field.clear();
304  }
305  else
306  {
307  field.append(ch);
308  }
309  }
310  if (not field.isEmpty())
311  {
312  row << field;
313  }
314  if (not row.isEmpty())
315  {
316  if (not headerSet)
317  {
318  d_ptr->header = row;
319  }
320  else
321  {
322  d_ptr->csvData.append(row);
323  }
324  }
325  file->close();
326 }

548 static QString qxt_addCsvQuotes(QxtCsvModel::QuoteMode mode, QString field)
549 {
550  bool addDoubleQuotes = ((mode & QxtCsvModel::DoubleQuote) && field.contains('"'));
551  bool addSingleQuotes = ((mode & QxtCsvModel::SingleQuote) && field.contains('\''));
552  bool quoteField = (mode & QxtCsvModel::AlwaysQuoteOutput) || addDoubleQuotes || addSingleQuotes;
553  if (quoteField && !addDoubleQuotes && !addSingleQuotes)
554  {
555  if (mode & QxtCsvModel::DoubleQuote)
556  {
557  addDoubleQuotes = true;
558  }
559  else if(mode & QxtCsvModel::SingleQuote)
560  {
561  addSingleQuotes = true;
562  }
563  }
564  if (mode & QxtCsvModel::BackslashEscape)
565  {
566  if (addDoubleQuotes)
567  {
568  return '"' + field.replace("\\", "\\\\").replace("\"", "\\\"") + '"';
569  }
570  if (addSingleQuotes)
571  {
572  return '\'' + field.replace("\\", "\\\\").replace("'", "\\'") + '\'';
573  }
574  }
575  else
576  {
577  if (addDoubleQuotes)
578  {
579  return '"' + field.replace("\"", "\"\"") + '"';
580  }
581  if (addSingleQuotes)
582  {
583  return '\'' + field.replace("'", "''") + '\'';
584  }
585  }
586  return field;
587 }

589 /*!
590  Outputs the content of the model as a CSV file to the device \a dest using \a codec.
591 
592  Fields in the output file will be separated by \a separator. Set \a withHeader to true
593  to output a row of headers at the top of the file.
594  */
595 // cppcheck-suppress funcArgNamesDifferent
596 void QxtCsvModel::toCSV(QIODevice* dest, bool withHeader, QChar separator, QTextCodec* codec) const
597 {
598  const QxtCsvModelPrivate& d_ptr = qxt_d();
599  int row, col, rows, cols;
600  rows = rowCount();
601  cols = columnCount();
602  QString data;
603  if (not dest->isOpen())
604  {
605  dest->open(QIODevice::WriteOnly | QIODevice::Truncate);
606  }
607  QTextStream stream(dest);
608  std::optional<QStringConverter::Encoding> text_encoding = QStringConverter::encodingForName(codec->name());
609  if (text_encoding.has_value())
610  {
611  stream.setEncoding(text_encoding.value());
612  }
613  if (withHeader)
614  {
615  data = "";
616  for (col = 0; col < cols; ++col)
617  {
618  if (col > 0)
619  {
620  data += separator;
621  }
622  data += qxt_addCsvQuotes(d_ptr.quoteMode, d_ptr.header.at(col));
623  }
624  stream << data << Qt::endl;
625  }
626  for (row = 0; row < rows; ++row)
627  {
628  const QStringList& rowData = d_ptr.csvData[row];
629  data = "";
630  for (col = 0; col < cols; ++col)
631  {
632  if (col > 0)
633  {
634  data += separator;
635  }
636  if (col < rowData.length())
637  {
638  data += qxt_addCsvQuotes(d_ptr.quoteMode, rowData.at(col));
639  }
640  else
641  {
642  data += qxt_addCsvQuotes(d_ptr.quoteMode, QString());
643  }
644  }
645  stream << data << Qt::endl;
646  }
647  stream << Qt::flush;
648  dest->close();
649 }
