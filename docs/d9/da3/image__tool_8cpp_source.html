100 ImageTool::ImageTool(QObject *parent, VAbstractPattern *doc, VMainGraphicsScene *draftScene, DraftImage img)
101  : QObject(parent),
102  image(img),
103  m_doc(doc),
104  m_draftScene(draftScene)
105 {
106  qCDebug(vImageTool, "Image filename: %s", qUtf8Printable(image.filename));
107 
108  QFileInfo fileInfo(image.filename);
109 
110  if (!fileInfo.exists())
111  {
112  const QString text = tr("The image <br/><br/> <b>%1</b> <br/><br/> could not be found. Do you "
113  "want to update the file location?").arg(image.filename);
114  QMessageBox::StandardButton result = QMessageBox::question(nullptr, tr("Loading image"), text,
115  QMessageBox::Yes | QMessageBox::No,
116  QMessageBox::Yes);
117  if (result == QMessageBox::No)
118  {
119  return;
120  }
121  else
122  {
123  bool askAgain = true;
124  while(askAgain)
125  {
126  image.filename = getImageFilename(nullptr);
127  if (image.filename.isEmpty())
128  {
129  qCDebug(vImageTool, "No image selected");
130  QMessageBox::critical(nullptr, tr("Import Image"), tr("No image was selected..."), QMessageBox::Ok);
131  continue;
132  }
133 
134  QImageReader imageReader(image.filename);
135 
136  if(!imageReader.canRead())
137  {
138  qCDebug(vImageTool, "Can't read image");
139  QMessageBox::critical(nullptr, tr("Import Image"), tr("Could not read the image.") + "\n" + tr("File may be corrupted..."), QMessageBox::Ok);
140  continue;
141  }
142  askAgain = false;
143  }
144  image.name = fileInfo.baseName();
145 
146  QDomElement domElement = m_doc->elementById(image.id, VAbstractPattern::TagDraftImage);
147  QDomElement newDomElement = domElement.cloneNode().toElement();
148  m_doc->SetAttribute(newDomElement, VAbstractPattern::AttrName, image.name);
149  m_doc->SetAttribute(newDomElement, VAbstractPattern::AttrSource, image.filename);
150  domElement.parentNode().replaceChild(newDomElement, domElement);
151  }
152  }
153 
154  if (image.name.isEmpty())
155  {
156  image.name = fileInfo.baseName();
157  }
158 
159  addImage();
160 }
