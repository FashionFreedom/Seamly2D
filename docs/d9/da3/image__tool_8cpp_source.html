106

160

162 
163  if (dialogResult != QDialog::Accepted)
164  {
165  deleteConfirmed = false;
166  }
167 
168  qApp->Settings()->setConfirmItemDelete(not msgBox.isChecked());
169 
170  }
171 
172  if (deleteConfirmed)
173  {
174  m_doc->removeBackgroundImage(id);
175  imageItem->deleteImageItem();
176  deleteLater();
