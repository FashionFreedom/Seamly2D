220 void TST_BuitInRegExp::TestCheckInternalVaribleRegExp()
221 {
222  QFETCH(QString, var);
223  QFETCH(QString, originalName);
224 
225  static const QString regex = QStringLiteral("(.){1,}_(.){1,}$");
226 
227  const QString sourceRegex = QLatin1String("^") + var + regex;
228  const QRegularExpression sourceRe(sourceRegex);
229 
230  const QString translated = m_trMs->InternalVarToUser(var);
231  const QString translationRegex = QLatin1String("^") + translated + regex;
232  const QRegularExpression translationRe(translationRegex);
233 
234  {
235  if (sourceRe.match(originalName).hasMatch() || translationRe.match(originalName).hasMatch())
236  {
237  const QString message = QString("Invalid original string '%1'").arg(originalName);
238  QFAIL(qUtf8Printable(message));
239  }
240 
241  const QString translated = m_trMs->VarToUser(originalName);
242  if (sourceRe.match(translated).hasMatch() || translationRe.match(translated).hasMatch())
243  {
244  const QString message = QString("Invalid translation string '%1'").arg(translated);
245  QFAIL(qUtf8Printable(message));
246  }
247  }
248 }
