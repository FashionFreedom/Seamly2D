176

178  settings->SetPMSystemCode(code);

180 
181  qApp->loadTranslations(locale);
182  qApp->processEvents();// force to call changeEvent
183 
184  // Part about measurments will not be updated automatically
185  qApp->RetranslateTables();
186  qApp->retranslateGroups();
187  }
188 
189  if (m_defGradationChanged)
190  {
191  settings->SetDefHeight(ui->defHeightCombo->currentText().toInt());
192  settings->SetDefSize(ui->defSizeCombo->currentText().toInt());
