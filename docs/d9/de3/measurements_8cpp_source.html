287 void MeasurementDoc::readMeasurements() const
288 {
289  // For conversion values we must first calculate all data in measurement file's unit.
290  // That's why we need two containers: one for converted values, second for real data.
291 
292  // Container for values in measurement file's unit
293  QScopedPointer<VContainer> tempData(new VContainer(data->getTranslateVariables(), data->GetPatternUnit()));
294 
295  const QDomNodeList list = elementsByTagName(TagMeasurement);
296  for (int i=0; i < list.size(); ++i)
297  {
298  const QDomElement dom = list.at(i).toElement();
299 
300  const QString name = GetParametrString(dom, AttrName);
301 
302  QString description;
303  try
304  {
305  description = GetParametrString(dom, AttrDescription);
306  }
307  catch (VExceptionEmptyParameter &error)
308  {
309  Q_UNUSED(error)
310  }
311 
312  QString fullName;
313  try
314  {
315  fullName = GetParametrString(dom, AttrFullName);
316  }
317  catch (VExceptionEmptyParameter &error)
318  {
319  Q_UNUSED(error)
320  }
321 
322  QSharedPointer<MeasurementVariable> meash;
323  QSharedPointer<MeasurementVariable> tempMeash;
324  if (type == MeasurementsType::Multisize)
325  {
326  qreal base = GetParametrDouble(dom, AttrBase, "0");
327  qreal ksize = GetParametrDouble(dom, AttrSizeIncrease, "0");
328  qreal kheight = GetParametrDouble(dom, AttrHeightIncrease, "0");
329 
330  tempMeash = QSharedPointer<MeasurementVariable>(new MeasurementVariable(static_cast<quint32>(i), name, BaseSize(),
331  BaseHeight(), base, ksize, kheight));
332  tempMeash->setSize(m_currentSize);
333  tempMeash->setHeight(m_currentHeight);
334  tempMeash->SetUnit(data->GetPatternUnit());
335 
336  base = UnitConvertor(base, measurementUnits(), *data->GetPatternUnit());
337  ksize = UnitConvertor(ksize, measurementUnits(), *data->GetPatternUnit());
338  kheight = UnitConvertor(kheight, measurementUnits(), *data->GetPatternUnit());
339 
340  const qreal baseSize = UnitConvertor(BaseSize(), measurementUnits(), *data->GetPatternUnit());
341  const qreal baseHeight = UnitConvertor(BaseHeight(), measurementUnits(), *data->GetPatternUnit());
342 
343  meash = QSharedPointer<MeasurementVariable>(new MeasurementVariable(static_cast<quint32>(i), name, baseSize, baseHeight,
344  base, ksize, kheight, fullName, description));
345  meash->setSize(m_currentSize);
346  meash->setHeight(m_currentHeight);
347  meash->SetUnit(data->GetPatternUnit());
348  }
349  else
350  {
351  const QString formula = GetParametrString(dom, AttrValue, "0");
352  bool ok = false;
353  qreal value = EvalFormula(tempData.data(), formula, &ok);
354 
355  tempMeash = QSharedPointer<MeasurementVariable>(new MeasurementVariable(tempData.data(), static_cast<quint32>(i), name,
356  value, formula, ok));
357 
358  value = UnitConvertor(value, measurementUnits(), *data->GetPatternUnit());
359  meash = QSharedPointer<MeasurementVariable>(new MeasurementVariable(data, static_cast<quint32>(i), name, value, formula,
360  ok, fullName, description));
361  }
362  tempData->AddVariable(name, tempMeash);
363  data->AddVariable(name, meash);
364  }
365 }

746 void MeasurementDoc::CreateEmptyMultisizeFile(Unit unit, int baseSize, int baseHeight)
747 {
748  this->clear();
749  QDomElement mElement = this->createElement(TagSMMS);
750 
751  mElement.appendChild(createComment(FileComment()));
752 
753  QDomElement version = createElement(TagVersion);
754  const QDomText newNodeText = createTextNode(MultiSizeConverter::MeasurementMaxVerStr);
755  version.appendChild(newNodeText);
756  mElement.appendChild(version);
757 
758  QDomElement ro = createElement(TagReadOnly);
759  const QDomText roNodeText = createTextNode("false");
760  ro.appendChild(roNodeText);
761  mElement.appendChild(ro);
762 
763  mElement.appendChild(createElement(TagNotes));
764 
765  QDomElement mUnit = createElement(TagUnit);
766  const QDomText unitText = createTextNode(UnitsToStr(unit));
767  mUnit.appendChild(unitText);
768  mElement.appendChild(mUnit);
769 
770  QDomElement system = createElement(TagPMSystem);
771  system.appendChild(createTextNode(ClearPMCode(p998_S)));
772  mElement.appendChild(system);
773 
774  QDomElement size = createElement(TagSize);
775  SetAttribute(size, AttrBase, QString().setNum(baseSize));
776  mElement.appendChild(size);
777 
778  QDomElement height = createElement(TagHeight);
779  SetAttribute(height, AttrBase, QString().setNum(baseHeight));
