165 template <typename T>
166 inline void VAbstractSpline::ShowToolVisualization(bool show)
167 {
168  if (show)
169  {
170  if (vis.isNull())
171  {
172  AddVisualization<T>();
173  SetVisualization();
174  }
175  else
176  {
177  if (T *visual = qobject_cast<T *>(vis))
178  {
179  visual->show();
180  }
181  }
182  }
183  else
184  {
185  delete vis;
186  }
187 
188  if (m_piecesMode)
189  {
190  showHandles(m_piecesMode);
191  }
192  else
193  {
194  showHandles(show);
195  }
196 
197  if (QGraphicsScene *sc = scene())
198  { // Showing/hiding control points require recalculation scene size.
199  VMainGraphicsView::NewSceneRect(sc, qApp->getSceneView(), this);
200  }
201 }
