112 QSizeF AbstractLayoutDialog::RoundTemplateSize(qreal width, qreal height, Unit unit) const
113 {
114  qreal w = 0;

116 
117  switch (unit)

123  h = qRound(height * 100.0) / 100.0;
124  return QSizeF(w, h);

127  h = qRound(height * 100000.0) / 100000.0;

131  }
