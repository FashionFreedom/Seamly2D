90 void SaveNodeCanonically(QXmlStreamWriter &stream, const QDomNode &domNode)
91 {
92  if (stream.hasError())
93  {
94  return;
95  }
96 
97  if (domNode.isElement())
98  {
99  const QDomElement domElement = domNode.toElement();
100  if (not domElement.isNull())
101  {
102  stream.writeStartElement(domElement.tagName());
103 
104  if (domElement.hasAttributes())
105  {
106  QMap<QString, QString> attributes;
107  const QDomNamedNodeMap attributeMap = domElement.attributes();
108  for (int i = 0; i < attributeMap.count(); ++i)
109  {
110  const QDomNode attribute = attributeMap.item(i);
111  attributes.insert(attribute.nodeName(), attribute.nodeValue());
112  }
113 
114  QMap<QString, QString>::const_iterator i = attributes.constBegin();
115  while (i != attributes.constEnd())
116  {
117  stream.writeAttribute(i.key(), i.value());
118  ++i;
119  }
120  }
121 
122  if (domElement.hasChildNodes())
123  {
124  QDomNode elementChild = domElement.firstChild();
125  while (not elementChild.isNull())
126  {
127  SaveNodeCanonically(stream, elementChild);
128  elementChild = elementChild.nextSibling();
129  }
130  }
131 
132  stream.writeEndElement();
133  }
134  }
135  else if (domNode.isComment())
136  {
137  stream.writeComment(domNode.nodeValue());
138  }
139  else if (domNode.isText())
140  {
141  stream.writeCharacters(domNode.nodeValue());
142  }
143 }
144 }

224 {
225  if (id == 0)
226  {
227  return QDomElement();
228  }
229 
230  if (map.contains(id))
231  {
232  const QDomElement e = map[id];
233  if (e.parentNode().nodeType() != QDomNode::BaseNode)
234  {
235  return e;
236  }
237  map.remove(id);
238  }
239 
240  if (tagName.isEmpty())
241  {
242  if (this->find(this->documentElement(), id))
243  {
244  return map[id];
245  }
246  }
247  else
248  {
249  const QDomNodeList list = elementsByTagName(tagName);
250  for (int i=0; i < list.size(); ++i)
251  {
252  const QDomElement domElement = list.at(i).toElement();
253  if (not domElement.isNull() && domElement.hasAttribute(AttrId))
254  {
255  try
256  {
257  const quint32 elementId = GetParametrUInt(domElement, AttrId, NULL_ID_STR);
258 
259  this->map[elementId] = domElement;
260  if (elementId == id)
261  {
262  return domElement;
263  }
264  }
265  catch (const VExceptionConversionError &)
266  {
267  // do nothing
268  }
269  }
270  }
271  }
272 
273  return QDomElement();

637 /**
638  * @brief ValidateXML validate xml file by xsd schema.
639  * @param schema path to schema file.
640  * @param fileName name of xml file.
641  */
642 void VDomDocument::ValidateXML(const QString &schema, const QString &fileName)
643 {
644  qCDebug(vXML, "Validation xml file %s.", qUtf8Printable(fileName));
645  QFile pattern(fileName);
646  // cppcheck-suppress ConfigurationNotChecked
647  if (pattern.open(QIODevice::ReadOnly) == false)
648  {
649  const QString errorMsg(tr("Can't open file %1:\n%2.").arg(fileName).arg(pattern.errorString()));
650  throw VException(errorMsg);
651  }
652 
653  QFile fileSchema(schema);
654  // cppcheck-suppress ConfigurationNotChecked
655  if (fileSchema.open(QIODevice::ReadOnly) == false)
656  {
657  pattern.close();
658  const QString errorMsg(tr("Can't open schema file %1:\n%2.").arg(schema).arg(fileSchema.errorString()));
659  throw VException(errorMsg);
660  }
661 
662  MessageHandler messageHandler;
663  QXmlSchema sch;
664  sch.setMessageHandler(&messageHandler);
665  if (sch.load(&fileSchema, QUrl::fromLocalFile(fileSchema.fileName()))==false)
666  {
667  pattern.close();
668  fileSchema.close();
669  VException e(messageHandler.statusMessage());
670  e.AddMoreInformation(tr("Could not load schema file '%1'.").arg(fileSchema.fileName()));
671  throw e;
672  }
673  qCDebug(vXML, "Schema loaded.");
674 
675  bool errorOccurred = false;
676  if (sch.isValid() == false)
677  {
678  errorOccurred = true;
679  }
680  else
681  {
682  QXmlSchemaValidator validator(sch);
683  if (validator.validate(&pattern, QUrl::fromLocalFile(pattern.fileName())) == false)
684  {
685  errorOccurred = true;
686  }
687  }
688 
689  if (errorOccurred)
690  {
691  pattern.close();
692  fileSchema.close();
693  VException e(messageHandler.statusMessage());
694  e.AddMoreInformation(tr("Validation error file %3 in line %1 column %2").arg(messageHandler.line())
695  .arg(messageHandler.column()).arg(fileName));
696  throw e;
697  }
698  pattern.close();
699  fileSchema.close();
700 }

883 {
884  bool result = false;
885 
886 #ifdef Q_OS_WIN32
887  qt_ntfs_permission_lookup++; // turn checking on
888 #endif /*Q_OS_WIN32*/
889 
890  QTemporaryFile destFile(destination + QLatin1String(".XXXXXX"));
891  destFile.setAutoRemove(false);// Will be renamed to be destination file
892  // cppcheck-suppress ConfigurationNotChecked
893  if (not destFile.open())
894  {
895  error = destFile.errorString();
896  }
897  else
898  {
899  QFile sourceFile(source);
900  // cppcheck-suppress ConfigurationNotChecked
901  if (sourceFile.open(QIODevice::ReadOnly))
902  {
903  result = true;
904  char block[4096];
905  qint64 bytes;
906  while ((bytes = sourceFile.read(block, sizeof(block))) > 0)
907  {
908  if (bytes != destFile.write(block, bytes))
909  {
910  error = destFile.errorString();
911  result = false;
912  break;
913  }
914  }
915 
916  if (bytes == -1)
917  {
918  error = sourceFile.errorString();
919  result = false;
920  }
921 
922  if (result)
923  {
924  QFile::remove(destination);
925  if (not destFile.rename(destination))
926  {
927  error = destFile.errorString();
928  result = false;
929  }
930  else
931  {
932  result = true;
933  }
934  }
935  }
936  else
937  {
938  error = sourceFile.errorString();
939  }
940  }
941 
942 #ifdef Q_OS_WIN32
943  qt_ntfs_permission_lookup--; // turn off check permission again
944 #endif /*Q_OS_WIN32*/
945 
946  return result;
