39 // ---------------------------------------------------------------------------
40 // Instances of this class are used to manage the content of entities. The
41 // scanner maintains a stack of these, one for each entity (this means entity
42 // in the sense of any parsed file or internal entity) currently being
43 // scanned. This class, given a binary input stream will handle reading in
44 // the data and decoding it from its external decoding into the internal
45 // Unicode format. Once internallized, this class provides the access
46 // methods to read in the data in various ways, maintains line and column
47 // information, and provides high performance character attribute checking
48 // methods.
49 //
50 // This is NOT to be derived from.
51 //
52 // ---------------------------------------------------------------------------

287  // -----------------------------------------------------------------------
288  // Data members
289  //
290  // fCharIndex
291  // The index into the character buffer. When this hits fCharsAvail
292  // then its time to refill.
293  //
294  // fCharBuf
295  // A buffer that the reader manager fills up with transcoded
296  // characters a small amount at a time.
297  //
298  // fCharsAvail
299  // The characters currently available in the character buffer.
300  //
301  // fCharSizeBuf
302  // This buffer is an array that contains the number of source chars
303  // eaten to create each char in the fCharBuf buffer. So the entry
304  // fCharSizeBuf[x] is the number of source chars that were eaten
305  // to make the internalized char fCharBuf[x]. This only contains
306  // useful data if fSrcOfsSupported is true.
307  //
308  // fCharOfsBuf
309  // This buffer is an array that contains the offset in the
310  // fRawByteBuf buffer of each char in the fCharBuf buffer. It
311  // only contains useful data if fSrcOfsSupported is true.
312  //
313  // fCurCol
314  // fCurLine
315  // The current line and column that we are in within this reader's
316  // text.
317  //
318  // fEncoding
319  // This is the rough encoding setting. This enum is set during
320  // construction and just tells us the rough family of encoding that
321  // we are doing.
322  //
323  // fEncodingStr
324  // This is the name of the encoding we are using. It will be
325  // provisionally set during construction, from the auto-sensed
326  // encoding. But it might be overridden when the XMLDecl is finally
327  // seen by the scanner. It can also be forced to a particular
328  // encoding, in which case fForcedEncoding is set.
329  //
330  // fForcedEncoding
331  // If the encoding if forced then this is set and all other
332  // information will be ignored. This encoding will be taken as
333  // gospel. This is done by calling an alternate constructor.
334  //
335  // fNoMore
336  // This is set when the source text is exhausted. It lets us know
337  // quickly that no more text is available.
338  //
339  // fRawBufIndex
340  // The current index into the raw byte buffer. When its equal to
341  // fRawBytesAvail then we need to read another buffer.
342  //
343  // fRawByteBuf
344  // This is the raw byte buffer that is used to spool out bytes
345  // from into the fCharBuf buffer, as we transcode in blocks.
346  //
347  // fRawBytesAvail
348  // The number of bytes currently available in the raw buffer. This
349  // helps deal with the last buffer's worth, which will usually not
350  // be a full one.
351  //
352  // fLowWaterMark
353  // The low water mark for the raw byte buffer.
354  //
355  //
356  // fReaderNum
357  // Each reader from a particular reader manager (which means from a
358  // particular document) is given a unique number. The reader manager
359  // sets these numbers. They are used to catch things like partial
360  // markup errors.
361  //
362  // fRefFrom
363  // This flag is provided in the ctor, and tells us if we represent
364  // some entity being expanded inside a literal. Sometimes things
365  // happen differently inside and outside literals.
366  //
367  // fPublicId
368  // fSystemId
369  // These are the system and public ids of the source that this
370  // reader is reading.
371  //
372  // fSentTrailingSpace
373  // If we are a PE entity being read and we not referenced from a
374  // literal, then a leading and trailing space must be faked into the
375  // data. This lets us know we've done the trailing space already (so
376  // we don't just keep doing it again and again.)
377  //
378  // fSource
379  // Indicates whether the content this reader is spooling as already
380  // been internalized. This will prevent multiple processing of
381  // whitespace when an already internalized entity is being spooled
382  // out.
383  //
384  // fSpareChar
385  // Some encodings can create two chars in an atomic way, e.g.
386  // surrogate pairs. We might not be able to store both, so we store
387  // it here until the next buffer transcoding operation.
388  //
389  // fSrcOfsBase
390  // This is the base offset within the source of this entity. Values
391  // in the curent fCharSizeBuf array are relative to this value.
392  //
393  // fSrcOfsSupported
394  // This flag is set to indicate whether source byte offset info
395  // is supported. For intrinsic encodings, its always set since we
396  // can always support it. For transcoder based encodings, we ask
397  // the transcoder if it supports it or not.
398  //
399  // fStream
400  // This is the input stream that provides the data for the reader.
401  // Its always treated as a raw byte stream. The derived class will
402  // ask for buffers of text from it and will handle making some
403  // sense of it.
404  //
405  // fSwapped
406  // If the encoding is one of the ones we do intrinsically, and its
407  // in a different byte order from our native order, then this is
408  // set to remind us to byte swap it during transcoding.
409  //
410  // fThrowAtEnd
411  // Indicates whether the reader manager should throw an end of entity
412  // exception at the end of this reader instance. This is usually
413  // set for top level external entity references. It overrides the
414  // reader manager's global flag that controls throwing at the end
415  // of entities. Defaults to false.
416  //
417  // fTranscoder
418  // If the encoding is not one that we handle intrinsically, then
419  // we use an an external transcoder to do it. This class is an
420  // abstraction that allows us to use pluggable external transcoding
421  // services (via XMLTransService in util.)
422  //
423  // fType
424  // Indicates whether this reader represents a PE or not. If this
425  // flag is true and the fInLiteral flag is false, then we will put
426  // out an extra space at the end.
427  //
428  // fgCharCharsTable;
429  // Pointer to XMLChar table, depends on XML version
430  //
431  // fNEL
432  // Boolean indicates if NEL and LSEP should be recognized as NEL
433  //
434  // fXMLVersion
435  // Enum to indicate if this Reader is conforming to XML 1.0 or XML 1.1
436  // -----------------------------------------------------------------------
437  XMLSize_t fCharIndex;
438  XMLCh fCharBuf[kCharBufSize];
439  XMLSize_t fCharsAvail;
440  unsigned char fCharSizeBuf[kCharBufSize];
441  unsigned int fCharOfsBuf[kCharBufSize];
442  XMLFileLoc fCurCol;

625 // ---------------------------------------------------------------------------
626 //
627 // XMLReader: movePlainContentChars()
628 //
629 // Move as many plain (no special handling of any sort required) content
630 // characters as possible from this reader to the supplied destination buffer.
631 //
632 // This is THE hottest performance spot in the parser.
633 //
634 // ---------------------------------------------------------------------------
635 inline void XMLReader::movePlainContentChars(XMLBuffer &dest)
636 {
637  const XMLSize_t chunkSize = fCharsAvail - fCharIndex;
638  const XMLCh* cursor = &fCharBuf[fCharIndex];
639  XMLSize_t count=0;
640  for(;count<chunkSize && (fgCharCharsTable[*cursor++] & gPlainContentCharMask) != 0;++count) /*noop*/ ;
641 
642  if (count!=0)
643  {
644  dest.append(&fCharBuf[fCharIndex], count);
645  fCharIndex += count;
646  fCurCol += (XMLFileLoc)count;
647  }
648 }

651 // ---------------------------------------------------------------------------
652 // XMLReader: getNextCharIfNot() method inlined for speed
653 // ---------------------------------------------------------------------------
654 inline bool XMLReader::getNextCharIfNot(const XMLCh chNotToGet, XMLCh& chGotten)
655 {
656  //
657  // See if there is at least a char in the buffer. Else, do the buffer
658  // reload logic.
659  //
660  if (fCharIndex >= fCharsAvail)
661  {
662  // If fNoMore is set, then we have nothing else to give
663  if (fNoMore)
664  return false;
665 
666  // Try to refresh
667  if (!refreshCharBuffer())
668  return false;
669  }
670 
671  // Check the next char
672  if (fCharBuf[fCharIndex] == chNotToGet)
673  return false;
674 
675  // Its not the one we want to skip so bump the index
676  chGotten = fCharBuf[fCharIndex++];
677 
678  // Handle end of line normalization and line/col member maintenance.
679  //
680  // we can have end-of-line combinations with a leading
681  // chCR(xD), chLF(xA), chNEL(x85), or chLineSeparator(x2028)
682  //
683  // 0000000000001101 chCR
684  // 0000000000001010 chLF
685  // 0000000010000101 chNEL
686  // 0010000000101000 chLineSeparator
687  // -----------------------
688  // 1101111101010000 == ~(chCR|chLF|chNEL|chLineSeparator)
689  //
690  // if the result of the logical-& operation is
691  // true : 'curCh' can not be chCR, chLF, chNEL or chLineSeparator
692  // false : 'curCh' can be chCR, chLF, chNEL or chLineSeparator
693  //
694  if ( chGotten & (XMLCh) ~(chCR|chLF|chNEL|chLineSeparator) )
695  {
696  fCurCol++;
697  } else
698  {
699  handleEOL(chGotten, false);
700  }
701 
702  return true;
703 }

756 // XMLReader: peekNextChar() method inlined for speed
757 // ---------------------------------------------------------------------------
758 inline bool XMLReader::peekNextChar(XMLCh& chGotten)
759 {
760  //
761  // If there is something still in the buffer, get it. Else do the reload
762  // scenario.
763  //
764  if (fCharIndex >= fCharsAvail)
765  {
766  // Try to refresh the buffer
