222  {
223  const QSharedPointer<CustomVariable> variables = data->getVariable<CustomVariable>(item->text());

227  break;

237  {
238  setDescription(item->text(),
239  *data->getVariable<VCurveLength>(qApp->translateVariables()->VarFromUser(item->text()))->GetValue(),

241  break;
242  }
243  case VariableTab::LineAngles:
244  {
245  setDescription(item->text(),
246  *data->getVariable<VLineAngle>(qApp->translateVariables()->VarFromUser(item->text()))->GetValue(),
247  degreeSymbol, tr("Line Angle"), "");
248  break;
249  }
250  case VariableTab::ArcRadii:
251  {
252  setDescription(item->text(),

608  ui->tableWidget->setItem(ui->tableWidget->rowCount()-1, ColumnName, item);

691 void EditFormulaDialog::filterVariables(const QString &filter)
692 {
693  ui->tableWidget->blockSignals(true);
694 
695  // If filter is empty findItems() for unknown reason returns nullptr items.
696  if (filter.isEmpty())
697  {
698  // show all rows
699  for (auto i = 0; i < ui->tableWidget->rowCount(); ++i)
700  {
701  ui->tableWidget->showRow(i);
702  }
703  }
704  else
705  {
706  // hide all rows
707  for (auto i = 0; i < ui->tableWidget->rowCount(); i++)
708  {
709  ui->tableWidget->hideRow(i);
710  }
711 
712  // show rows with matched filter
713  for (auto item : ui->tableWidget->findItems(filter, Qt::MatchContains))
714  {
715  // If filter is empty findItems() for unknown reason returns nullptr items.
716  if (item)
717  {
718  ui->tableWidget->showRow(item->row());
719  }
720  }
721  }
722 
723  ui->tableWidget->blockSignals(false);
724 }
