134 template <class TElem> void ValueVectorOf<TElem>::
135 insertElementAt(const TElem& toInsert, const XMLSize_t insertAt)
136 {
137  if (insertAt == fCurCount)
138  {
139  addElement(toInsert);
140  return;
141  }
142 
143  if (insertAt > fCurCount)
144  ThrowXMLwithMemMgr(ArrayIndexOutOfBoundsException, XMLExcepts::Vector_BadIndex, fMemoryManager);
145 
146  // Make room for the newbie
147  ensureExtraCapacity(1);
148  for (XMLSize_t index = fCurCount; index > insertAt; index--)
149  fElemList[index] = fElemList[index-1];
150 
151  // And stick it in and bump the count
152  fElemList[insertAt] = toInsert;
153  fCurCount++;
154 }

161

227 template <class TElem> void ValueVectorOf<TElem>::
228 ensureExtraCapacity(const XMLSize_t length)
229 {
230  XMLSize_t newMax = fCurCount + length;
231 
232  if (newMax > fMaxCount)
233  {
234  // Avoid too many reallocations by expanding by a percentage
235  XMLSize_t minNewMax = (XMLSize_t)((double)fCurCount * 1.25);
236  if (newMax < minNewMax)
237  newMax = minNewMax;
238 
239  TElem* newList = (TElem*) fMemoryManager->allocate
240  (
241  newMax * sizeof(TElem)
242  ); //new TElem[newMax];
243  for (XMLSize_t index = 0; index < fCurCount; index++)
244  newList[index] = fElemList[index];
245 
246  fMemoryManager->deallocate(fElemList); //delete [] fElemList;
247  fElemList = newList;
248  fMaxCount = newMax;
249  }
250 }
