236  // fElemState

239  // Stores an element next state from DFA content model - used for
240  // wildcard validation
241  //
242  // fDTDElemNonDeclPool
243  // registry of "faulted-in" DTD element decls
244  // fSchemaElemNonDeclPool
245  // registry for elements without decls in the grammar
246  // fElemCount
247  // count of the number of start tags seen so far (starts at 1).
248  // Used for duplicate attribute detection/processing of required/defaulted attributes
249  // fAttDefRegistry
250  // mapping from XMLAttDef instances to the count of the last
251  // start tag where they were utilized.
252  // fUndeclaredAttrRegistry
253  // set of attr QNames to detect duplicates
254  // fPSVIAttrList
255  // PSVI attribute list implementation that needs to be
256  // filled when a PSVIHandler is registered
257  // fSchemaInfoList
258  // Transient schema info list that is passed to TraverseSchema instances.
259  // fCachedSchemaInfoList
260  // Cached Schema info list that is passed to TraverseSchema instances.
261  //
